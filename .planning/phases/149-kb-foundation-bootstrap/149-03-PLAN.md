// app/src/lib/services/knowledge-sync.ts
/**
 * knowledge-sync — Servicio de sincronización bidireccional DB ↔ .docflow-kb/.
 *
 * Responsabilidades:
 * - Escribir/actualizar archivos .md desde DB rows (create, update).
 * - Soft-delete via markDeprecated (status: deprecated).
 * - Auto-sync de _index.json y _header.md tras cada op.
 * - detectBumpLevel según §5.2 del PRD (patch/minor/major).
 * - touchAccess — incrementa access_count para TTL management.
 *
 * NO responsabilidades en esta fase (Phase 149):
 * - Integración real con prompt-assembler → Fase 4 del PRD (invalidateLLMCache es no-op).
 * - Lectura de DB real en el CLI → Fase 2 del PRD (syncResource recibe row ya leída).
 *
 * Convenciones:
 * - NO usar process.env directo; recibir `kbRoot` por context.
 * - NO importar mejor-sqlite3 aquí; el servicio opera sobre filesystem + row ya leída.
 * - Errores: throw Error con mensaje descriptivo; el caller decide logging.
 */
import * as fs from 'fs';
import * as path from 'path';

export type Entity = 'catpaw' | 'connector' | 'catbrain' | 'template' | 'skill' | 'canvas';
export type Op = 'create' | 'update' | 'delete' | 'access';
export type BumpLevel = 'patch' | 'minor' | 'major';

export interface DBRow {
  id: string;
  [k: string]: unknown;
}

export interface SyncContext {
  author?: string;
  kbRoot?: string;
  reason?: string;
  superseded_by?: string;
}

const DEFAULT_KB_ROOT = path.resolve(__dirname, '../../../../.docflow-kb');

function getKbRoot(ctx?: SyncContext): string {
  return ctx?.kbRoot ?? DEFAULT_KB_ROOT;
}

// --- Mapping entity → subdir en resources/ ---
const ENTITY_SUBDIR: Record<Entity, string> = {
  catpaw: 'resources/catpaws',
  connector: 'resources/connectors',
  catbrain: 'resources/catbrains',
  template: 'resources/email-templates',
  skill: 'resources/skills',
  canvas: 'resources/canvases',
};

// --- Campos por entity que vienen de DB (fields_from_db) ---
const FIELDS_FROM_DB: Record<Entity, string[]> = {
  catpaw: ['name', 'description', 'mode', 'model', 'system_prompt', 'temperature', 'max_tokens', 'is_active', 'department'],
  connector: ['name', 'description', 'type', 'config', 'is_active'],
  catbrain: ['name', 'description', 'collection', 'is_active'],
  template: ['name', 'description', 'subject', 'body', 'product'],
  skill: ['name', 'description', 'category', 'is_active'],
  canvas: ['name', 'description', 'canvas_data', 'is_active'],
};

// --- Utilidades slugify, idShort ---
function slugify(name: string): string {
  return (name || 'unnamed').toLowerCase().replace(/[^a-z0-9]+/g, '-').replace(/^-|-$/g, '').slice(0, 50);
}

function idShort(id: string): string {
  return id.slice(0, 8);
}

function kbFilePath(kbRoot: string, entity: Entity, row: { id: string; name?: string }): string {
  const subdir = ENTITY_SUBDIR[entity];
  const slug = slugify(row.name ?? 'unnamed');
  const fname = `${idShort(row.id)}-${slug}.md`;
  return path.join(kbRoot, subdir, fname);
}

function findExistingFileByIdShort(kbRoot: string, entity: Entity, id: string): string | null {
  const subdir = path.join(kbRoot, ENTITY_SUBDIR[entity]);
  if (!fs.existsSync(subdir)) return null;
  const prefix = idShort(id);

}

// --- detectBumpLevel ---
export function detectBumpLevel(current: Record<string, unknown>, newRow: DBRow): BumpLevel {
  // major first
  if (current.subtype !== undefined && (newRow as any).subtype !== undefined && current.subtype !== (newRow as any).subtype) return 'major';
  if (current.mode !== undefined && newRow.mode !== undefined && current.mode !== newRow.mode) return 'major';
  if (current.status === 'active' && (newRow.is_active === 0 || newRow.is_active === false)) return 'major';
  // contract I/O change: si hay un campo explicit `io_contract_hash` o similar; no lo tenemos en DB, skip.

  // minor
  if (current.system_prompt !== undefined && newRow.system_prompt !== undefined && current.system_prompt !== newRow.system_prompt) return 'minor';
  if ((newRow as any).connectors_linked && JSON.stringify((newRow as any).connectors_linked) !== JSON.stringify(current.connectors_linked)) return 'minor';
  if ((newRow as any).skills_linked && JSON.stringify((newRow as any).skills_linked) !== JSON.stringify(current.skills_linked)) return 'minor';
  if (current.lang === 'es' && (newRow as any).lang === 'es+en') return 'minor';
  // related cambios: si detectamos diffs significativos
  // ...

  // patch (default)
  return 'patch';
}

function bumpVersion(current: string, level: BumpLevel): string {
  const [M, m, p] = current.split('.').map(Number);
  if (level === 'major') return `${M + 1}.0.0`;
  if (level === 'minor') return `${M}.${m + 1}.0`;
  return `${M}.${m}.${p + 1}`;
}

function truncateChangeLog(log: any[]): any[] {
  if (!Array.isArray(log)) return [];
  return log.slice(-5); // últimos 5
}

export async function syncResource(
  entity: Entity,
  op: Op,
  row: DBRow | { id: string },
  context?: SyncContext
): Promise<void> {
  const ctx = context ?? {};
  const kbRoot = getKbRoot(ctx);

  switch (op) {
    case 'create': {
      const fm = buildFrontmatterForCreate(entity, row as DBRow, ctx);
      const fpath = kbFilePath(kbRoot, entity, row as any);
      fs.mkdirSync(path.dirname(fpath), { recursive: true });
      const content = serializeFrontmatter(fm) + `\n# ${fm.title}\n\n${fm.summary}\n`;
      fs.writeFileSync(fpath, content);
      break;
    }
    case 'update': {
      const fpath = findExistingFileByIdShort(kbRoot, entity, row.id);
      if (!fpath) {
        // Si no existe, tratar como create
        return syncResource(entity, 'create', row as DBRow, ctx);
      }
      const existing = fs.readFileSync(fpath, 'utf8');
      const { frontmatter: current, body } = parseFrontmatter(existing);
      const bump = detectBumpLevel(current, row as DBRow);
      const newVersion = bumpVersion(String(current.version ?? '1.0.0'), bump);
      const now = new Date().toISOString();
      // Merge: fields_from_db sobrescriben; enriched_fields (body con ## Use cases etc.) se preservan
      const updated = { ...current };
      for (const k of FIELDS_FROM_DB[entity]) {
        if ((row as any)[k] !== undefined) {
          if (k === 'name') updated.title = String((row as any)[k]);
          else if (k === 'description') updated.summary = String((row as any)[k]).slice(0, 200);
          // otros campos específicos se reflejan en el body por template (minimal MVP)
        }
      }
      updated.version = newVersion;
      updated.updated_at = now;
      updated.updated_by = ctx.author ?? 'unknown';
      const prevLog = Array.isArray(current.change_log) ? current.change_log : [];
      updated.change_log = truncateChangeLog([
        ...prevLog,
        { version: newVersion, date: now.slice(0,10), author: ctx.author ?? 'unknown', change: `Auto-sync ${bump} bump` },
      ]);
      const newContent = serializeFrontmatter(updated) + body;
      fs.writeFileSync(fpath, newContent);
      break;
    }
    case 'delete': {
      const fpath = findExistingFileByIdShort(kbRoot, entity, row.id);
      if (!fpath) return; // nada que deprecar
      await markDeprecated(fpath, row, ctx.author ?? 'unknown', ctx.reason, ctx.superseded_by);
      break;
    }
    case 'access': {
      const fpath = findExistingFileByIdShort(kbRoot, entity, row.id);
      if (!fpath) return;
      await touchAccess(fpath);
      break;
    }
  }

  await updateIndexEntry(kbRoot);
  await regenerateHeader(kbRoot);
  await invalidateLLMCache(); // no-op en Phase 149
}

export async function touchAccess(filePath: string): Promise<void> {
  const content = fs.readFileSync(filePath, 'utf8');
  const { frontmatter, body } = parseFrontmatter(content);
  frontmatter.access_count = (typeof frontmatter.access_count === 'number' ? frontmatter.access_count : 0) + 1;
  frontmatter.last_accessed_at = new Date().toISOString();
  fs.writeFileSync(filePath, serializeFrontmatter(frontmatter) + body);
}

export async function markDeprecated(
  filePath: string,
  row: { id: string },
  author: string,
  reason?: string,
  supersededBy?: string
): Promise<void> {
  const content = fs.readFileSync(filePath, 'utf8');
  const { frontmatter, body } = parseFrontmatter(content);
  const now = new Date().toISOString();
  frontmatter.status = 'deprecated';
  frontmatter.deprecated_at = now;
  frontmatter.deprecated_by = author;
  frontmatter.deprecated_reason = reason ?? `DB row removed at ${now}`;
  if (supersededBy) frontmatter.superseded_by = supersededBy;
  // bump major por deprecation
  const newVersion = bumpVersion(String(frontmatter.version ?? '1.0.0'), 'major');
  frontmatter.version = newVersion;
  frontmatter.updated_at = now;
  frontmatter.updated_by = author;
  const prevLog = Array.isArray(frontmatter.change_log) ? frontmatter.change_log : [];
  frontmatter.change_log = truncateChangeLog([
    ...prevLog,
    { version: newVersion, date: now.slice(0,10), author, change: 'DEPRECATED — ' + (reason ?? 'DB removed') },
  ]);
  fs.writeFileSync(filePath, serializeFrontmatter(frontmatter) + body);
}
