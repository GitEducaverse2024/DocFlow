```javascript
#!/usr/bin/env node
/**
 * kb-sync — CLI de mantenimiento del Knowledge Base (.docflow-kb/).
 *
 * Comandos:
 *   --full-rebuild                        Regenera _index.json desde frontmatters.
 *   --audit-stale                         Genera _audit_stale.md.
 *   --archive --confirm                   Mueve elegibles a _archived/YYYY-MM-DD/.
 *   --purge --confirm --older-than-archived=365d   Borra físicamente archivos viejos.
 *
 * Flags destructivos requieren --confirm explícito. Sin él, aborta exit 1.
 *
 * Phase 149: implementa workflow completo de mantenimiento. NO lee DB
 * (eso es Fase 2 del PRD: `--full-rebuild --source db`). Comando rechaza --source db con mensaje "wired in Phase 2".
 */
const fs = require('fs');
const path = require('path');

const KB_ROOT = path.resolve(__dirname, '..', '.docflow-kb');
const ARCHIVE_THRESHOLD_DAYS = 180;
const WARNING_THRESHOLD_DAYS = 150;
const VISIBLE_WARNING_DAYS = 170;

// --- walkKB helper ---
function walkKB(root, opts = {}) {
  const { excludeDirs = ['_archived'], includeExt = '.md' } = opts;

// --- Helpers de frontmatter ---
function readFrontmatter(filePath) { /* ... */ }
function writeFrontmatter(filePath, fm, body) { /* ... */ }

// --- Cálculo de días ---
function daysSince(isoDateString) {
  if (!isoDateString) return Infinity;
  const then = new Date(isoDateString).getTime();
  const now = Date.now();
  return Math.floor((now - then) / (1000 * 60 * 60 * 24));
}

function cmdFullRebuild(args) {
  if (args.includes('--source') || args.includes('--source=db') || args.some(a => a.startsWith('--source='))) {
    console.error('ERROR: --source db no implementado en Phase 149. Planeado para Fase 2 del PRD.');
    process.exit(1);
  }
  // Recorrer .docflow-kb/resources/**/*.md, parsear frontmatter, construir entries[] y indexes{}.
  const files = walkKB(KB_ROOT);
  const entries = [];
  const byType = {};
  const byTag = {};
  const byAudience = {};
  for (const f of files) {
    const rel = path.relative(KB_ROOT, f);
    try {
      const { frontmatter } = readFrontmatter(f);
      if (!frontmatter.id) continue;
      const entry = {
        id: frontmatter.id,
        path: rel,
        type: frontmatter.type,
        subtype: frontmatter.subtype ?? null,
        title: frontmatter.title,
        summary: frontmatter.summary,
        tags: frontmatter.tags ?? [],
        audience: frontmatter.audience ?? [],
        status: frontmatter.status,
        updated: frontmatter.updated_at ?? frontmatter.created_at,
        search_hints: frontmatter.search_hints ?? null,
      };
      entries.push(entry);
      (byType[entry.type] ||= []).push(entry.id);
      for (const t of entry.tags) (byTag[t] ||= []).push(entry.id);
      for (const a of entry.audience) (byAudience[a] ||= []).push(entry.id);
    } catch (e) {
      console.warn(`WARN ${rel}: ${e.message}`);
    }
  }

  const counts = {
    catpaws_active: entries.filter(e => e.subtype === 'catpaw' && e.status === 'active').length,
    connectors_active: entries.filter(e => e.subtype === 'connector' && e.status === 'active').length,
    catbrains_active: entries.filter(e => e.subtype === 'catbrain' && e.status === 'active').length,
    templates_active: entries.filter(e => e.subtype === 'email-template' && e.status === 'active').length,
    skills_active: entries.filter(e => e.subtype === 'skill' && e.status === 'active').length,
    rules: entries.filter(e => e.type === 'rule').length,
    incidents_resolved: entries.filter(e => e.type === 'incident' && e.status === 'active').length,
    features_documented: entries.filter(e => e.type === 'feature').length,
  };

  const topTags = Object.entries(byTag).sort((a, b) => b[1].length - a[1].length).slice(0, 10).map(([t]) => t);

  const idx = {
    schema_version: '2.0',
    generated_at: new Date().toISOString(),
    generated_by: 'kb-sync-cli',
    entry_count: entries.length,
    header: { counts, top_tags: topTags, last_changes: [] },
    entries,
    indexes: { by_type: byType, by_tag: byTag, by_audience: byAudience },
  };
  fs.writeFileSync(path.join(KB_ROOT, '_index.json'), JSON.stringify(idx, null, 2));
  console.log(`OK: _index.json regenerado con ${entries.length} entries`);
}

function cmdAuditStale() {
  const files = walkKB(KB_ROOT);
  const eligible = []; // >=180d, deprecated, sin refs
  const warning = []; // 150-180d

  // Escanear todos los archivos para detectar `related` cross-references
  const incomingRefs = {}; // id -> count
  for (const f of files) {
    try {
      const { frontmatter } = readFrontmatter(f);
      if (Array.isArray(frontmatter.related)) {
        for (const r of frontmatter.related) {
          incomingRefs[r] = (incomingRefs[r] || 0) + 1;
        }
      }
    } catch { /* skip */ }
  }

  for (const f of files) {
    try {
      const { frontmatter } = readFrontmatter(f);
      if (frontmatter.status !== 'deprecated') continue;
      const days = daysSince(frontmatter.last_accessed_at);
      const refs = incomingRefs[frontmatter.id] || 0;
      const rec = {
        id: frontmatter.id,
        title: typeof frontmatter.title === 'string' ? frontmatter.title : (frontmatter.title?.es || frontmatter.title?.en || '(no title)'),
        deprecated_at: frontmatter.deprecated_at,
        last_access: frontmatter.last_accessed_at,
        days,
        refs,
        path: path.relative(KB_ROOT, f),
      };
      if (days >= ARCHIVE_THRESHOLD_DAYS && refs === 0) {
        eligible.push(rec);
      } else if (days >= WARNING_THRESHOLD_DAYS) {
        warning.push(rec);
      }
    } catch { /* skip */ }
  }

  const now = new Date();
  const isoDate = now.toISOString().slice(0, 10);
  const warningVisible = warning.some(w => w.days >= VISIBLE_WARNING_DAYS);

  const auditPath = path.join(KB_ROOT, '_audit_stale.md');
  const fm = [
    '---',
    `id: audit-stale-${isoDate}`,
    'type: audit',
    'lang: es',
    `title: Audit stale ${isoDate}`,
    'summary: Archivos elegibles para archivado/purga',
    'tags: [ops]',
    'audience: [developer]',
    'status: active',
    `created_at: "${now.toISOString()}"`,
    'created_by: kb-sync-cli',
    'version: 1.0.0',
    `updated_at: "${now.toISOString()}"`,
    'updated_by: kb-sync-cli',
    'source_of_truth: null',
    'change_log:',
    `  - { version: 1.0.0, date: "${isoDate}", author: kb-sync-cli, change: "Generado por --audit-stale" }`,
    'ttl: never',
    `eligible_for_purge: ${eligible.length}`,
    `warning_only: ${warning.length}`,
    `warning_visible: ${warningVisible}`,
    '---',
    '',
    `# Audit de archivos stale — ${isoDate}`,
    '',
    '## Elegibles para archivado (180+ días sin acceso, deprecated)',
    '',
    '| ID | Title | Deprecated since | Last access | Refs in-coming | Action |',
    '|----|-------|------------------|-------------|----------------|--------|',
  ];
  if (eligible.length === 0) fm.push('| _(ninguno)_ | | | | | |');
  else for (const e of eligible) fm.push(`| ${e.id} | ${e.title} | ${e.deprecated_at} (${daysSince(e.deprecated_at)}d) | ${e.last_access} | ${e.refs} | **Elegible archivar** |`);
  fm.push('', '## Warning only (150+ días, aviso previo)', '');
  if (warning.length === 0) fm.push('_(ninguno)_');
  else {
    fm.push('| ID | Title | Días | Action |');
    fm.push('|----|-------|------|--------|');
    for (const w of warning) fm.push(`| ${w.id} | ${w.title} | ${w.days} | ${w.days >= VISIBLE_WARNING_DAYS ? '**Warning visible**' : 'Aviso informativo'} |`);
  }
  fm.push('', '---', '', '**Para archivar los elegibles:**', '', '    node scripts/kb-sync.cjs --archive --confirm', '');
  fs.writeFileSync(auditPath, fm.join('\n'));
  console.log(`OK: _audit_stale.md generado (${eligible.length} elegibles, ${warning.length} warnings${warningVisible ? ', warning_visible=true' : ''})`);
}

function cmdArchive(args) {
  if (!args.includes('--confirm')) {
    console.error('ERROR: --archive requiere --confirm explícito. Aborted.');
    console.error('Uso: node scripts/kb-sync.cjs --archive --confirm');
    process.exit(1);
  }
  // Re-ejecutar lógica de elegibilidad; mover elegibles a _archived/YYYY-MM-DD/
  const files = walkKB(KB_ROOT);
  const incomingRefs = {};
  for (const f of files) {
    try {
      const { frontmatter } = readFrontmatter(f);
      if (Array.isArray(frontmatter.related)) {
        for (const r of frontmatter.related) incomingRefs[r] = (incomingRefs[r] || 0) + 1;
      }
    } catch { /* skip */ }
  }

  const isoDate = new Date().toISOString().slice(0, 10);
  const archiveDir = path.join(KB_ROOT, '_archived', isoDate);
  let moved = 0;

  for (const f of files) {
    try {
      const { frontmatter, body } = readFrontmatter(f);
      if (frontmatter.status !== 'deprecated') continue;
      const days = daysSince(frontmatter.last_accessed_at);
      const refs = incomingRefs[frontmatter.id] || 0;
      if (days < ARCHIVE_THRESHOLD_DAYS || refs > 0) continue;

      fs.mkdirSync(archiveDir, { recursive: true });
      const newPath = path.join(archiveDir, path.basename(f));
      // Actualizar status a archived antes del move
      frontmatter.status = 'archived';
      const newContent = writeFrontmatter(null, frontmatter, body); // devuelve string
      fs.writeFileSync(newPath, newContent);
      fs.unlinkSync(f);
      console.log(`ARCHIVED: ${path.relative(KB_ROOT, f)} -> _archived/${isoDate}/${path.basename(f)}`);
      moved++;
    } catch (e) {
      console.warn(`WARN ${f}: ${e.message}`);
    }
  }
  console.log(`OK: ${moved} archivos archivados a _archived/${isoDate}/`);
  // Re-rebuild _index.json tras mover
  cmdFullRebuild([]);
}

function cmdPurge(args) {
  if (!args.includes('--confirm')) {
    console.error('ERROR: --purge requiere --confirm explícito. Aborted.');
    console.error('Uso: node scripts/kb-sync.cjs --purge --confirm --older-than-archived=365d');
    process.exit(1);
  }
  let thresholdDays = 365;
  const flag = args.find(a => a.startsWith('--older-than-archived='));
  if (flag) {
    const m = flag.match(/--older-than-archived=(\d+)d?/);
    if (m) thresholdDays = parseInt(m[1], 10);
  }
  const archiveRoot = path.join(KB_ROOT, '_archived');
  if (!fs.existsSync(archiveRoot)) {
    console.log('OK: _archived/ no existe — nada que purgar');
    return;
  }
  let purged = 0;
  for (const dateDir of fs.readdirSync(archiveRoot)) {
    const dirPath = path.join(archiveRoot, dateDir);
    if (!fs.statSync(dirPath).isDirectory()) continue;
    const days = daysSince(new Date(dateDir + 'T00:00:00Z').toISOString());
    if (days < thresholdDays) continue;
    const files = fs.readdirSync(dirPath);
    for (const f of files) {
      fs.unlinkSync(path.join(dirPath, f));
      console.log(`PURGED: _archived/${dateDir}/${f}`);
      purged++;
    }
    // Borrar el dir vacío
    fs.rmdirSync(dirPath);
  }
  console.log(`OK: ${purged} archivos purgados físicamente (>${thresholdDays}d en _archived)`);
}

// --- Main dispatcher ---
function main() {
  const args = process.argv.slice(2);
  if (args.includes('--full-rebuild')) return cmdFullRebuild(args);
  if (args.includes('--audit-stale')) return cmdAuditStale();
  if (args.includes('--archive')) return cmdArchive(args);
  if (args.includes('--purge')) return cmdPurge(args);

  console.error('Uso:');
  console.error('  node scripts/kb-sync.cjs --full-rebuild');
  console.error('  node scripts/kb-sync.cjs --audit-stale');
  console.error('  node scripts/kb-sync.cjs --archive --confirm');
  console.error('  node scripts/kb-sync.cjs --purge --confirm --older-than-archived=365d');
  process.exit(1);
}

main();
```

```typescript
import * as os from 'os';

const REPO_ROOT = path.resolve(__dirname, '../../../..');
const CLI = path.join(REPO_ROOT, 'scripts/kb-sync.cjs');

let tmpRepo: string;
let tmpKb: string;

function runCli(args: string[], expectCode: number = 0): { code: number; stdout: string; stderr: string } {
  try {
    const stdout = execSync(`node "${CLI}" ${args.join(' ')}`, {
      cwd: tmpRepo,
      encoding: 'utf8',
      stdio: ['ignore', 'pipe', 'pipe'],
    });
    return { code: 0, stdout, stderr: '' };
  } catch (e: any) {
    return { code: e.status ?? 1, stdout: e.stdout?.toString() ?? '', stderr: e.stderr?.toString() ?? '' };
  }
}

function writeFixture(subdir: string, filename: string, fm: Record<string, any>, body = 'body') {
  const fpath = path.join(tmpKb, subdir, filename);
  fs.mkdirSync(path.dirname(fpath), { recursive: true });
  const yaml = ['---'];
  for (const [k, v] of Object.entries(fm)) {
    if (Array.isArray(v)) yaml.push(`${k}: [${v.map(x => `"${x}"`).join(', ')}]`);
    else if (v === null) yaml.push(`${k}: null`);
    else if (typeof v === 'object') {
      yaml.push(`${k}:`);
      for (const [sk, sv] of Object.entries(v)) yaml.push(`  ${sk}: ${typeof sv === 'string' ? `"${sv}"` : sv}`);
    } else yaml.push(`${k}: ${typeof v === 'string' ? `"${v}"` : v}`);
  }
  yaml.push('---');
  yaml.push(body);
  fs.writeFileSync(fpath, yaml.join('\n'));
  return fpath;
}
```
