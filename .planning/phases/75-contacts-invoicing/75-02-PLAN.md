Document API (invoicing module, default — no module parameter):
- GET /documents/invoice — list invoices. Query params: starttmp, endtmp, contactid, paid (0/1/2), sort (created-asc/created-desc)
- POST /documents/invoice — create invoice. Body: { contactId, items: [{name, units, subtotal, tax}], date?, notes?, currency? }
- Response fields: id, contact, contactName, date, dueDate, total, subtotal, tax, status, paid (0/1/2), items, ...

Invoice item fields: name, desc, units, subtotal, discount, tax, retention, sku, ...
API calculates total = (subtotal * units) + tax%. No need to compute client-side.

```typescript
export const quickInvoiceItemSchema = z.object({
  name: z.string().min(1).describe('Item/service name'),
  units: z.number().positive().optional().describe('Quantity (default: 1)'),
  subtotal: z.number().nonnegative().describe('Unit price (before tax)'),
  tax: z.number().min(0).max(100).optional().describe('Tax percentage (default: 21 for Spain IVA)'),
});

export const quickInvoiceSchema = z.object({
  contact: z.string().min(1).describe('Contact name (fuzzy matched) or contact ID (24-char hex)'),
  items: z.array(quickInvoiceItemSchema).min(1).describe('Line items for the invoice'),
  date: z.number().int().optional().describe('Invoice date as Unix timestamp (seconds). Defaults to now.'),
  notes: z.string().optional().describe('Invoice notes'),
  currency: z.string().optional().describe('Currency code (e.g., EUR, USD). Defaults to account currency.'),
});

export const listInvoicesSchema = z.object({
  contact: z.string().min(1).describe('Contact name or ID'),
  paid: z.enum(['0', '1', '2']).optional().describe('Filter: 0=unpaid, 1=paid, 2=partially paid'),
  startDate: z.number().int().optional().describe('Start date filter (Unix timestamp seconds)'),
  endDate: z.number().int().optional().describe('End date filter (Unix timestamp seconds)'),
  page: z.number().int().positive().optional(),
  limit: z.number().int().positive().max(500).optional(),
});

export const invoiceSummarySchema = z.object({
  contact: z.string().min(1).describe('Contact name or ID'),
  months: z.number().int().positive().optional().describe('Number of months to look back (default: 12)'),
});
```

**2. Create `src/tools/invoice-helpers.ts`:**

```typescript
import { HoldedClient } from '../holded-client.js';
import {
  quickInvoiceSchema,
  listInvoicesSchema,
  invoiceSummarySchema,
  withValidation,
} from '../validation.js';
import { resolveContactId } from '../utils/id-resolver.js';

interface InvoiceListItem {
  id: string;
  contact: string;
  contactName: string;
  date: number;
  dueDate?: number;
  total: number;
  subtotal: number;
  tax: number;
  status: number;
  paid?: number;
  [key: string]: unknown;
}

export function getInvoiceHelperTools(client: HoldedClient) {
  return {
    holded_quick_invoice: {
      description:
        'Create a Holded invoice with simplified inputs. Accepts a contact name (fuzzy matched) or contact ID, and simple line items with optional tax (defaults to 21% Spain IVA). Returns the created invoice ID.',
      inputSchema: {
        type: 'object' as const,
        properties: {
          contact: {
            type: 'string',
            description: 'Contact name (fuzzy matched) or contact ID (24-char hex)',
          },
          items: {
            type: 'array',
            description: 'Line items: [{name, subtotal, units?, tax?}]',
            items: {
              type: 'object',
              properties: {
                name: { type: 'string', description: 'Item/service name' },
                units: { type: 'number', description: 'Quantity (default: 1)' },
                subtotal: { type: 'number', description: 'Unit price before tax' },
                tax: { type: 'number', description: 'Tax % (default: 21)' },
              },
              required: ['name', 'subtotal'],
            },
          },
          date: {
            type: 'number',
            description: 'Invoice date (Unix timestamp seconds). Defaults to now.',
          },
          notes: {
            type: 'string',
            description: 'Invoice notes',
          },
          currency: {
            type: 'string',
            description: 'Currency code (e.g., EUR)',
          },
        },
        required: ['contact', 'items'],
      },
      destructiveHint: true,
      handler: withValidation(quickInvoiceSchema, async (args) => {
        const contactId = await resolveContactId(client, args.contact);

        const items = args.items.map((item) => ({
          name: item.name,
          units: item.units ?? 1,
          subtotal: item.subtotal,
          tax: item.tax ?? 21,
        }));

        const body: Record<string, unknown> = { contactId, items };
        if (args.date) body.date = args.date;
        if (args.notes) body.notes = args.notes;
        if (args.currency) body.currency = args.currency;

        return client.post('/documents/invoice', body);
      }),
    },

    holded_list_invoices: {
      description:
        'List Holded invoices for a specific contact. Accepts contact name (fuzzy matched) or ID. Supports filtering by payment status (paid/unpaid/partial) and date range.',
      inputSchema: {
        type: 'object' as const,
        properties: {
          contact: {
            type: 'string',
            description: 'Contact name (fuzzy matched) or contact ID',
          },
          paid: {
            type: 'string',
            enum: ['0', '1', '2'],
            description: 'Filter: 0=unpaid, 1=paid, 2=partially paid',
          },
          startDate: {
            type: 'number',
            description: 'Start date filter (Unix timestamp seconds)',
          },
          endDate: {
            type: 'number',
            description: 'End date filter (Unix timestamp seconds)',
          },
          page: {
            type: 'number',
            description: 'Page number (default: 1)',
          },
          limit: {
            type: 'number',
            description: 'Items per page (default: 50, max: 500)',
          },
        },
        required: ['contact'],
      },
      readOnlyHint: true,
      handler: withValidation(listInvoicesSchema, async (args) => {
        const contactId = await resolveContactId(client, args.contact);

        const queryParams: Record<string, string | number> = {
          contactid: contactId,
        };
        if (args.paid) queryParams.paid = args.paid;
        if (args.startDate) queryParams.starttmp = String(args.startDate);
        if (args.endDate) queryParams.endtmp = String(args.endDate);

        const invoices = await client.get<InvoiceListItem[]>(
          '/documents/invoice',
          queryParams
        );

        const page = args.page ?? 1;
        const limit = Math.min(args.limit ?? 50, 500);
        const startIndex = (page - 1) * limit;
        const endIndex = startIndex + limit;
        const items = invoices.slice(startIndex, endIndex);

        return {
          items: items.map((inv) => ({
            id: inv.id,
            contactName: inv.contactName,
            date: inv.date,
            dueDate: inv.dueDate,
            total: inv.total,
            subtotal: inv.subtotal,
            tax: inv.tax,
            status: inv.status,
            paid: inv.paid,
          })),
          page,
          pageSize: items.length,
          totalItems: invoices.length,
          totalPages: Math.ceil(invoices.length / limit),
          hasMore: endIndex < invoices.length,
        };
      }),
    },

    holded_invoice_summary: {
      description:
        'Get invoice summary for a Holded contact: total invoiced, total paid, total pending, invoice count. Accepts contact name (fuzzy matched) or ID. Looks back a configurable number of months (default: 12).',
      inputSchema: {
        type: 'object' as const,
        properties: {
          contact: {
            type: 'string',
            description: 'Contact name (fuzzy matched) or contact ID',
          },
          months: {
            type: 'number',
            description: 'Months to look back (default: 12)',
          },
        },
        required: ['contact'],
      },
      readOnlyHint: true,
      handler: withValidation(invoiceSummarySchema, async (args) => {
        const contactId = await resolveContactId(client, args.contact);

        const months = args.months ?? 12;
        const now = Math.floor(Date.now() / 1000);
        const startDate = now - months * 30 * 24 * 60 * 60;

        const invoices = await client.get<InvoiceListItem[]>(
          '/documents/invoice',
          {
            contactid: contactId,
            starttmp: String(startDate),
            endtmp: String(now),
          }
        );

        let totalInvoiced = 0;
        let totalPaid = 0;
        let totalPending = 0;
        let countPaid = 0;
        let countUnpaid = 0;
        let countPartial = 0;

        for (const inv of invoices) {
          const total = inv.total ?? 0;
          totalInvoiced += total;

          if (inv.paid === 1) {
            totalPaid += total;
            countPaid++;
          } else if (inv.paid === 2) {
            // Partially paid — we don't know exact paid amount from list endpoint
            // Count as pending for conservative estimate
            totalPending += total;
            countPartial++;
          } else {
            totalPending += total;
            countUnpaid++;
          }
        }

        return {
          contactId,
          period: `Last ${months} months`,
          totalInvoices: invoices.length,
          totalInvoiced: Math.round(totalInvoiced * 100) / 100,
          totalPaid: Math.round(totalPaid * 100) / 100,
          totalPending: Math.round(totalPending * 100) / 100,
          breakdown: {
            paid: countPaid,
            unpaid: countUnpaid,
            partiallyPaid: countPartial,
          },
        };
      }),
    },
  };
}
```

```typescript
import { createMockClient } from './mock-client.js';
import { getInvoiceHelperTools } from '../tools/invoice-helpers.js';

import { resolveContactId } from '../utils/id-resolver.js';
const mockResolveContactId = vi.mocked(resolveContactId);

const mockInvoices = [
  {
    id: 'inv-1',
    contact: 'contact-123',
    contactName: 'Acme Corp',
    date: 1710000000,
    dueDate: 1712592000,
    total: 1210,
    subtotal: 1000,
    tax: 210,
    status: 1,
    paid: 1,
  },
  {
    id: 'inv-2',
    contact: 'contact-123',
    contactName: 'Acme Corp',
    date: 1711000000,
    dueDate: 1713592000,
    total: 605,
    subtotal: 500,
    tax: 105,
    status: 0,
    paid: 0,
  },
  {
    id: 'inv-3',
    contact: 'contact-123',
    contactName: 'Acme Corp',
    date: 1712000000,
    dueDate: 1714592000,
    total: 242,
    subtotal: 200,
    tax: 42,
    status: 0,
    paid: 2,
  },
];

describe('Invoice Helper Tools', () => {
  let client: ReturnType<typeof createMockClient>;
  let tools: ReturnType<typeof getInvoiceHelperTools>;

  beforeEach(() => {
    vi.clearAllMocks();
    mockResolveContactId.mockResolvedValue('contact-123');
    client = createMockClient();
    tools = getInvoiceHelperTools(client);
  });

  describe('holded_quick_invoice', () => {
    it('should create an invoice with contact name resolution', async () => {
      client.post = vi.fn().mockResolvedValue({ status: 1, info: 'Created', id: 'new-inv' });

      const result = await tools.holded_quick_invoice.handler({
        contact: 'Acme Corp',
        items: [{ name: 'Consulting', subtotal: 1000 }],
      });

      expect(mockResolveContactId).toHaveBeenCalledWith(client, 'Acme Corp');
      expect(client.post).toHaveBeenCalledWith('/documents/invoice', {
        contactId: 'contact-123',
        items: [{ name: 'Consulting', units: 1, subtotal: 1000, tax: 21 }],
      });
      expect(result).toEqual({ status: 1, info: 'Created', id: 'new-inv' });
    });

    it('should apply default tax 21% and units 1', async () => {
      client.post = vi.fn().mockResolvedValue({ status: 1, id: 'inv-x' });

      await tools.holded_quick_invoice.handler({
        contact: 'Acme',
        items: [{ name: 'Service A', subtotal: 500 }],
      });

      const callArgs = (client.post as any).mock.calls[0][1];
      expect(callArgs.items[0].tax).toBe(21);
      expect(callArgs.items[0].units).toBe(1);
    });

    it('should respect custom tax and units per item', async () => {
      client.post = vi.fn().mockResolvedValue({ status: 1, id: 'inv-x' });

      await tools.holded_quick_invoice.handler({
        contact: 'Acme',
        items: [
          { name: 'Product A', subtotal: 100, units: 5, tax: 10 },
          { name: 'Product B', subtotal: 200, tax: 0 },
        ],
      });

      const callArgs = (client.post as any).mock.calls[0][1];
      expect(callArgs.items[0]).toEqual({ name: 'Product A', units: 5, subtotal: 100, tax: 10 });
      expect(callArgs.items[1]).toEqual({ name: 'Product B', units: 1, subtotal: 200, tax: 0 });
    });

    it('should pass optional date, notes, currency', async () => {
      client.post = vi.fn().mockResolvedValue({ status: 1, id: 'inv-x' });

      await tools.holded_quick_invoice.handler({
        contact: 'Acme',
        items: [{ name: 'X', subtotal: 100 }],
        date: 1710000000,
        notes: 'Test invoice',
        currency: 'USD',
      });

      const callArgs = (client.post as any).mock.calls[0][1];
      expect(callArgs.date).toBe(1710000000);
      expect(callArgs.notes).toBe('Test invoice');
      expect(callArgs.currency).toBe('USD');
    });

    it('should reject empty items array', async () => {
      await expect(
        tools.holded_quick_invoice.handler({ contact: 'Acme', items: [] })
      ).rejects.toThrow();
    });
  });

  describe('holded_list_invoices', () => {
    it('should list invoices for a contact', async () => {
      client.get = vi.fn().mockResolvedValue(mockInvoices);

      const result = (await tools.holded_list_invoices.handler({
        contact: 'Acme Corp',
      })) as any;

      expect(mockResolveContactId).toHaveBeenCalledWith(client, 'Acme Corp');
      expect(client.get).toHaveBeenCalledWith('/documents/invoice', {
        contactid: 'contact-123',

      expect(result.totalItems).toBe(3);
    });

    it('should pass paid filter', async () => {
      client.get = vi.fn().mockResolvedValue([mockInvoices[1]]);

        contact: 'Acme',
        paid: '0',

        contact: 'Acme',
```
