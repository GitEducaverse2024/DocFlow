```typescript
/**
 * Resolve a contact name or ID to a contact ID.
 * If nameOrId is a 24-char hex string, returns it directly.
 * Otherwise, fetches all contacts and does fuzzy matching by name.
 */
export async function resolveContactId(
  client: HoldedClient,
  nameOrId: string
): Promise<string> {
  if (looksLikeId(nameOrId)) {
    return nameOrId;
  }

  const contacts = await client.get<Array<{ id: string; name: string }>>('/contacts');

  const matches = fuzzyMatch(contacts, nameOrId);

  if (matches.length === 0) {
    throw new Error(`No contact found matching "${nameOrId}"`);
  }

  if (matches.length === 1) {
    return matches[0].id;
  }

  throw new AmbiguousMatchError(
    'contact',
    nameOrId,
    matches.map((c) => ({ id: c.id, name: c.name }))
  );
}
```

```typescript
import { HoldedClient } from '../holded-client.js';
import {
  searchContactSchema,
  resolveContactSchema,
  withValidation,
} from '../validation.js';
import { resolveContactId } from '../utils/id-resolver.js';

interface HoldedContact {
  id: string;
  name: string;
  email?: string;
  vatnumber?: string;
  tradename?: string;
  phone?: string;
  mobile?: string;
  type?: string;
  [key: string]: unknown;
}

export function getContactSearchTools(client: HoldedClient) {
  return {
    holded_search_contact: {
      description:
        'Search Holded contacts by name, email, VAT number (NIF/CIF), or trade name. Case-insensitive fuzzy match with pagination. Use this to find a contact before creating invoices or looking up history.',
      inputSchema: {
        type: 'object' as const,
        properties: {
          query: {
            type: 'string',
            description: 'Search query (matches name, email, vatnumber, or tradename)',
          },
          page: {
            type: 'number',
            description: 'Page number (default: 1)',
          },
          limit: {
            type: 'number',
            description: 'Items per page (default: 50, max: 500)',
          },
        },
        required: ['query'],
      },
      readOnlyHint: true,
      handler: withValidation(searchContactSchema, async (args) => {
        const allContacts = await client.get<HoldedContact[]>('/contacts');
        const q = args.query.toLowerCase();

        const filtered = allContacts.filter((c) => {
          const nameMatch = c.name?.toLowerCase().includes(q);
          const emailMatch = c.email?.toLowerCase().includes(q);
          const vatMatch = c.vatnumber?.toLowerCase().includes(q);
          const tradeMatch = c.tradename?.toLowerCase().includes(q);
          return nameMatch || emailMatch || vatMatch || tradeMatch;
        });

        const page = args.page ?? 1;
        const limit = Math.min(args.limit ?? 50, 500);
        const startIndex = (page - 1) * limit;
        const endIndex = startIndex + limit;
        const items = filtered.slice(startIndex, endIndex);

        return {
          items: items.map((c) => ({
            id: c.id,
            name: c.name,
            email: c.email,
            vatnumber: c.vatnumber,
            tradename: c.tradename,
            phone: c.phone,
            type: c.type,
          })),
          page,
          pageSize: items.length,
          totalItems: filtered.length,
          totalPages: Math.ceil(filtered.length / limit),
          hasMore: endIndex < filtered.length,
        };
      }),
    },

    holded_resolve_contact: {
      description:
        'Resolve a contact name to a Holded contact ID. Pass a name (fuzzy matched) or a 24-char hex ID (returned as-is). Throws an error with candidates if multiple contacts match. Use this before creating invoices by contact name.',
      inputSchema: {
        type: 'object' as const,
        properties: {
          contact: {
            type: 'string',
            description: 'Contact name or ID. Names are fuzzy-matched (exact > startsWith > includes).',
          },
        },
        required: ['contact'],
      },
      readOnlyHint: true,
      handler: withValidation(resolveContactSchema, async (args) => {
        const contactId = await resolveContactId(client, args.contact);
        return { contactId };
      }),
    },
  };
}
```
