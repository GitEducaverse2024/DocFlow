Add the following tool definition inside the return object of `getContactSearchTools`, after `holded_resolve_contact`:

```typescript
    holded_contact_context: {
      description:
        'Get full business context for a Holded contact: details (name, email, VAT, type, address) + recent invoices (last 90 days by default) + outstanding balance. Single call for complete relationship overview. Accepts contact name (fuzzy matched) or ID.',
      inputSchema: {
        type: 'object' as const,
        properties: {
          contact: {
            type: 'string',
            description: 'Contact name (fuzzy matched) or contact ID (24-char hex)',
          },
          days: {
            type: 'number',
            description: 'Days to look back for invoices (default: 90)',
          },
        },
        required: ['contact'],
      },
      readOnlyHint: true,
      handler: withValidation(contactContextSchema, async (args) => {
        const contactId = await resolveContactId(client, args.contact);

        // Fetch contact details and recent invoices in parallel
        const days = args.days ?? 90;
        const now = Math.floor(Date.now() / 1000);
        const startDate = now - days * 24 * 60 * 60;

        const [contactDetails, invoices] = await Promise.all([
          client.get<Record<string, unknown>>(`/contacts/${contactId}`),
          client.get<Array<{
            id: string;
            contactName: string;
            date: number;
            dueDate?: number;
            total: number;
            subtotal: number;
            tax: number;
            paid?: number;
            [key: string]: unknown;
          }>>('/documents/invoice', {
            contactid: contactId,
            starttmp: String(startDate),
            endtmp: String(now),
          }),
        ]);

        // Compute balance
        let totalInvoiced = 0;
        let totalPaid = 0;
        let totalPending = 0;

        for (const inv of invoices) {
          const total = inv.total ?? 0;
          totalInvoiced += total;
          if (inv.paid === 1) {
            totalPaid += total;
          } else {
            totalPending += total;
          }
        }

        // Limit recent invoices to 20 most recent (sorted by date desc)
        const recentInvoices = [...invoices]
          .sort((a, b) => (b.date ?? 0) - (a.date ?? 0))
          .slice(0, 20)
          .map((inv) => ({
            id: inv.id,
            date: inv.date,
            dueDate: inv.dueDate,
            total: inv.total,
            paid: inv.paid,
          }));

        return {
          contact: {
            id: contactId,
            name: contactDetails.name,
            email: contactDetails.email,
            vatnumber: contactDetails.vatnumber,
            tradename: contactDetails.tradename,
            type: contactDetails.type,
            phone: contactDetails.phone,
            mobile: contactDetails.mobile,
            billAddress: contactDetails.billAddress,
          },
          invoices: {
            period: `Last ${days} days`,
            count: invoices.length,
            totalInvoiced: Math.round(totalInvoiced * 100) / 100,
            totalPaid: Math.round(totalPaid * 100) / 100,
            totalPending: Math.round(totalPending * 100) / 100,
            recent: recentInvoices,
          },
        };
      }),
    },
```

describe('holded_contact_context', () => {
  let client: ReturnType<typeof createMockClient>;
  let tools: ReturnType<typeof getContactSearchTools>;

  const mockContactDetail = {
    id: '507f1f77bcf86cd799439011',
    name: 'Acme Corp',
    email: 'info@acme.com',
    vatnumber: 'B12345678',
    tradename: 'Acme',
    type: 'client',
    phone: '912345678',
    mobile: '612345678',
    billAddress: { address: 'Calle Mayor 1', city: 'Madrid', postalCode: '28001' },
  };

  const mockInvoices = [
    { id: 'inv-1', contactName: 'Acme Corp', date: 1710000000, dueDate: 1712592000, total: 1210, subtotal: 1000, tax: 210, paid: 1 },
    { id: 'inv-2', contactName: 'Acme Corp', date: 1711000000, dueDate: 1713592000, total: 605, subtotal: 500, tax: 105, paid: 0 },
    { id: 'inv-3', contactName: 'Acme Corp', date: 1712000000, dueDate: 1714592000, total: 242, subtotal: 200, tax: 42, paid: 2 },
  ];

  beforeEach(() => {
    vi.clearAllMocks();
    client = createMockClient();
    tools = getContactSearchTools(client);
  });

  it('should return contact details and invoice summary', async () => {
    client.get = vi.fn()
      .mockResolvedValueOnce(mockContacts) // resolveContactId fetches contacts
      .mockResolvedValueOnce(mockContactDetail) // get contact details
      .mockResolvedValueOnce(mockInvoices); // get invoices

    const result = (await tools.holded_contact_context.handler({
      contact: 'Beta Industries',
    })) as any;

    expect(result.contact.name).toBe('Acme Corp');
    expect(result.contact.email).toBe('info@acme.com');
    expect(result.contact.vatnumber).toBe('B12345678');
    expect(result.invoices.count).toBe(3);
    expect(result.invoices.totalInvoiced).toBe(2057);
    expect(result.invoices.totalPaid).toBe(1210);
    expect(result.invoices.totalPending).toBe(847);
  });

  it('should pass through hex ID without API call for resolution', async () => {
    client.get = vi.fn()
      .mockResolvedValueOnce(mockContactDetail) // get contact details
      .mockResolvedValueOnce([]); // get invoices (empty)

    const result = (await tools.holded_contact_context.handler({
      contact: '507f1f77bcf86cd799439011',
    })) as any;

    // First call should be /contacts/{id}, not /contacts (list all)
    expect((client.get as any).mock.calls[0][0]).toBe('/contacts/507f1f77bcf86cd799439011');
    expect(result.contact.id).toBe('507f1f77bcf86cd799439011');
  });

  it('should limit recent invoices to 20', async () => {
    const manyInvoices = Array.from({ length: 30 }, (_, i) => ({
      id: `inv-${i}`,
      contactName: 'Acme',
      date: 1710000000 + i * 86400,
      total: 100,
      subtotal: 82.64,
      tax: 17.36,
      paid: 0,
    }));

    client.get = vi.fn()
      .mockResolvedValueOnce(mockContactDetail)
      .mockResolvedValueOnce(manyInvoices);

    const result = (await tools.holded_contact_context.handler({
      contact: '507f1f77bcf86cd799439011',
    })) as any;

    expect(result.invoices.count).toBe(30); // total count
    expect(result.invoices.recent).toHaveLength(20); // limited
  });

  it('should sort recent invoices by date descending', async () => {
    client.get = vi.fn()
      .mockResolvedValueOnce(mockContactDetail)
      .mockResolvedValueOnce(mockInvoices);

    const result = (await tools.holded_contact_context.handler({
      contact: '507f1f77bcf86cd799439011',
    })) as any;

    const dates = result.invoices.recent.map((i: any) => i.date);
    expect(dates[0]).toBeGreaterThanOrEqual(dates[1]);
    expect(dates[1]).toBeGreaterThanOrEqual(dates[2]);
  });

  it('should handle empty invoices', async () => {
    client.get = vi.fn()
      .mockResolvedValueOnce(mockContactDetail)
      .mockResolvedValueOnce([]);

    const result = (await tools.holded_contact_context.handler({
      contact: '507f1f77bcf86cd799439011',
    })) as any;

    expect(result.invoices.count).toBe(0);
    expect(result.invoices.totalInvoiced).toBe(0);
    expect(result.invoices.totalPaid).toBe(0);
    expect(result.invoices.totalPending).toBe(0);
    expect(result.invoices.recent).toHaveLength(0);
  });

  it('should use custom days parameter', async () => {
    client.get = vi.fn()
      .mockResolvedValueOnce(mockContactDetail)
      .mockResolvedValueOnce([]);

    const result = (await tools.holded_contact_context.handler({
      contact: '507f1f77bcf86cd799439011',
      days: 30,
    })) as any;

    expect(result.invoices.period).toBe('Last 30 days');
  });

  it('should default to 90 days', async () => {
    client.get = vi.fn()
      .mockResolvedValueOnce(mockContactDetail)
      .mockResolvedValueOnce([]);

    const result = (await tools.holded_contact_context.handler({
      contact: '507f1f77bcf86cd799439011',
    })) as any;

    expect(result.invoices.period).toBe('Last 90 days');
  });
});
