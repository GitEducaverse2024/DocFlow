import { HoldedClient } from '../holded-client.js';
import { funnelIdSchema, withValidation } from '../validation.js';

export function getFunnelTools(client: HoldedClient) {
  return {
    holded_list_funnels: {
      description:
        'List all CRM funnels (sales pipelines) with their stages. Each funnel contains stages that leads move through.',
      inputSchema: {
        type: 'object' as const,
        properties: {},
        required: [],
      },
      readOnlyHint: true,
      handler: async () => {
        return client.get('/funnels', undefined, 'crm');
      },
    },

    holded_get_funnel: {
      description:
        'Get a specific CRM funnel by ID, including all its stages with IDs and names.',
      inputSchema: {
        type: 'object' as const,
        properties: {
          funnelId: {
            type: 'string',
            description: 'Funnel ID',
          },
        },
        required: ['funnelId'],
      },
      readOnlyHint: true,
      handler: withValidation(funnelIdSchema, async (args) => {
        return client.get(`/funnels/${args.funnelId}`, undefined, 'crm');
      }),
    },
  };
}
```

import { describe, it, expect, beforeEach, vi } from 'vitest';
import { createMockClient } from './mock-client.js';
import { getFunnelTools } from '../tools/funnels.js';

describe('Funnel Tools', () => {
  let client: ReturnType<typeof createMockClient>;
  let tools: ReturnType<typeof getFunnelTools>;

  beforeEach(() => {
    vi.clearAllMocks();
    client = createMockClient();
    tools = getFunnelTools(client);
  });

  describe('holded_list_funnels', () => {
    it('should list all funnels using CRM module', async () => {
      const mockFunnels = [
        { id: 'funnel-1', name: 'Sales Pipeline', stages: [{ stageId: 's1', name: 'New' }] },
        { id: 'funnel-2', name: 'Support Pipeline', stages: [] },
      ];
      client.get = vi.fn().mockResolvedValue(mockFunnels);

      const result = await tools.holded_list_funnels.handler({});

      expect(client.get).toHaveBeenCalledWith('/funnels', undefined, 'crm');
      expect(result).toEqual(mockFunnels);
    });

    it('should return empty array when no funnels exist', async () => {
      client.get = vi.fn().mockResolvedValue([]);

      const result = await tools.holded_list_funnels.handler({});

      expect(result).toEqual([]);
    });
  });

  describe('holded_get_funnel', () => {
    it('should get a funnel by ID using CRM module', async () => {
      const mockFunnel = {
        id: 'funnel-1',
        name: 'Sales Pipeline',
        stages: [
          { stageId: 's1', key: 'new', name: 'New', desc: '' },
          { stageId: 's2', key: 'negotiation', name: 'Negotiation', desc: '' },
        ],
      };
      client.get = vi.fn().mockResolvedValue(mockFunnel);

      const result = await tools.holded_get_funnel.handler({ funnelId: 'funnel-1' });

      expect(client.get).toHaveBeenCalledWith('/funnels/funnel-1', undefined, 'crm');
      expect(result).toEqual(mockFunnel);
    });

    it('should reject empty funnelId', async () => {
      await expect(tools.holded_get_funnel.handler({ funnelId: '' })).rejects.toThrow();
    });
  });
});
```
