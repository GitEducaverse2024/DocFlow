```typescript
import { HoldedClient } from '../holded-client.js';

interface FunnelStage {
  stageId: string;
  key: string;
  name: string;
  desc: string;
}

interface Funnel {
  id: string;
  name: string;
  stages: FunnelStage[];
}

export class AmbiguousMatchError extends Error {
  candidates: Array<{ id: string; name: string }>;

  constructor(type: string, query: string, candidates: Array<{ id: string; name: string }>) {
    const candidateList = candidates.map((c) => `  - "${c.name}" (${c.id})`).join('\n');
    super(`Ambiguous ${type} match for "${query}". Candidates:\n${candidateList}`);
    this.name = 'AmbiguousMatchError';
    this.candidates = candidates;
  }
}

/**
 * Check if a string looks like a Holded ID (24-char hex).
 */
function looksLikeId(value: string): boolean {
  return /^[a-f0-9]{24}$/.test(value);
}

/**
 * Fuzzy match: case-insensitive. Prioritizes exact match, then startsWith, then includes.
 */
function fuzzyMatch<T extends { name: string }>(
  items: T[],
  query: string
): T[] {
  const q = query.toLowerCase().trim();

  // 1. Exact match (case-insensitive)
  const exact = items.filter((item) => item.name.toLowerCase() === q);
  if (exact.length > 0) return exact;

  // 2. Starts with
  const startsWith = items.filter((item) => item.name.toLowerCase().startsWith(q));
  if (startsWith.length > 0) return startsWith;

  // 3. Includes
  const includes = items.filter((item) => item.name.toLowerCase().includes(q));
  return includes;
}

/**
 * Resolve a funnel name or ID to a funnel ID.
 * If nameOrId is a 24-char hex string, returns it directly (assumed to be an ID).
 * Otherwise, fetches all funnels and does fuzzy matching by name.
 */
export async function resolveFunnelId(
  client: HoldedClient,
  nameOrId: string
): Promise<string> {
  if (looksLikeId(nameOrId)) {
    return nameOrId;
  }

  const funnels = await client.get<Funnel[]>('/funnels', undefined, 'crm');

  const matches = fuzzyMatch(funnels, nameOrId);

  if (matches.length === 0) {
    throw new Error(`No funnel found matching "${nameOrId}"`);
  }

  if (matches.length === 1) {
    return matches[0].id;
  }

  throw new AmbiguousMatchError(
    'funnel',
    nameOrId,
    matches.map((f) => ({ id: f.id, name: f.name }))
  );
}

/**
 * Resolve a stage name or ID to a stageId within a specific funnel.
 * If stageNameOrId is a 24-char hex string, returns it directly.
 * Otherwise, fetches the funnel and does fuzzy matching on stage names.
 */
export async function resolveLeadStageId(
  client: HoldedClient,
  funnelId: string,
  stageNameOrId: string
): Promise<string> {
  if (looksLikeId(stageNameOrId)) {
    return stageNameOrId;
  }

  const funnel = await client.get<Funnel>(`/funnels/${funnelId}`, undefined, 'crm');

  const stages = funnel.stages || [];
  const matches = fuzzyMatch(
    stages.map((s) => ({ ...s, name: s.name })),
    stageNameOrId
  );

  if (matches.length === 0) {
    throw new Error(`No stage found matching "${stageNameOrId}" in funnel "${funnel.name}"`);
  }

  if (matches.length === 1) {
    return matches[0].stageId;
  }

  throw new AmbiguousMatchError(
    'stage',
    stageNameOrId,
    matches.map((s) => ({ id: s.stageId, name: s.name }))
  );
}
```

```typescript
import { describe, it, expect, beforeEach, vi } from 'vitest';
import { createMockClient } from './mock-client.js';
import { resolveFunnelId, resolveLeadStageId, AmbiguousMatchError } from '../utils/id-resolver.js';

const mockFunnels = [
  {
    id: '6650a1b2c3d4e5f678901234',
    name: 'Sales Pipeline',
    stages: [
      { stageId: 'stage-new', key: 'new', name: 'New Lead', desc: '' },
      { stageId: 'stage-nego', key: 'negotiation', name: 'Negotiation', desc: '' },
      { stageId: 'stage-won', key: 'won', name: 'Won', desc: '' },
    ],
  },
  {
    id: '6650a1b2c3d4e5f678905678',
    name: 'Support Pipeline',
    stages: [
      { stageId: 'stage-open', key: 'open', name: 'Open', desc: '' },
      { stageId: 'stage-closed', key: 'closed', name: 'Closed', desc: '' },
    ],
  },
];

// For ambiguity testing
const ambiguousFunnels = [
  ...mockFunnels,
  {
    id: '6650a1b2c3d4e5f678909999',
    name: 'Sales Pipeline International',
    stages: [],
  },
];

describe('ID Resolver', () => {
  let client: ReturnType<typeof createMockClient>;

  beforeEach(() => {
    vi.clearAllMocks();
    client = createMockClient();
  });

  describe('resolveFunnelId', () => {
    it('should pass through a 24-char hex ID without API call', async () => {
      const id = '6650a1b2c3d4e5f678901234';
      const result = await resolveFunnelId(client, id);
      expect(result).toBe(id);
      expect(client.get).not.toHaveBeenCalled();
    });

    it('should resolve exact funnel name (case-insensitive)', async () => {
      client.get = vi.fn().mockResolvedValue(mockFunnels);
      const result = await resolveFunnelId(client, 'sales pipeline');
      expect(result).toBe('6650a1b2c3d4e5f678901234');
      expect(client.get).toHaveBeenCalledWith('/funnels', undefined, 'crm');
    });

    it('should resolve partial funnel name (startsWith)', async () => {
      client.get = vi.fn().mockResolvedValue(mockFunnels);
      const result = await resolveFunnelId(client, 'Support');
      expect(result).toBe('6650a1b2c3d4e5f678905678');
    });

    it('should resolve partial funnel name (includes)', async () => {
      client.get = vi.fn().mockResolvedValue(mockFunnels);
      const result = await resolveFunnelId(client, 'Pipeline');
      // Both match "Pipeline" — should be ambiguous
      // But with only mockFunnels (2 items), both contain "Pipeline"
    });

    it('should throw error when no funnel matches', async () => {
      client.get = vi.fn().mockResolvedValue(mockFunnels);
      await expect(resolveFunnelId(client, 'Nonexistent')).rejects.toThrow(
        'No funnel found matching "Nonexistent"'
      );
    });

    it('should throw AmbiguousMatchError when multiple funnels match', async () => {
      client.get = vi.fn().mockResolvedValue(ambiguousFunnels);
      try {
        await resolveFunnelId(client, 'Sales');
        expect.fail('Should have thrown');
      } catch (error) {
        expect(error).toBeInstanceOf(AmbiguousMatchError);
        const ambError = error as AmbiguousMatchError;
        expect(ambError.candidates).toHaveLength(2);
        expect(ambError.candidates[0].name).toBe('Sales Pipeline');
        expect(ambError.candidates[1].name).toBe('Sales Pipeline International');
      }
    });
  });

  describe('resolveLeadStageId', () => {
    it('should pass through a 24-char hex stageId without API call', async () => {
      const stageId = 'aabbccdd11223344aabb1122';
      const result = await resolveLeadStageId(client, 'funnel-1', stageId);
      expect(result).toBe(stageId);
      expect(client.get).not.toHaveBeenCalled();
    });

    it('should resolve stage name within a funnel', async () => {
      client.get = vi.fn().mockResolvedValue(mockFunnels[0]);
      const result = await resolveLeadStageId(client, '6650a1b2c3d4e5f678901234', 'Negotiation');
      expect(result).toBe('stage-nego');
      expect(client.get).toHaveBeenCalledWith(
        '/funnels/6650a1b2c3d4e5f678901234',
        undefined,
        'crm'
      );
    });

    it('should resolve stage name case-insensitively', async () => {
      client.get = vi.fn().mockResolvedValue(mockFunnels[0]);
      const result = await resolveLeadStageId(client, 'funnel-1', 'new lead');
      expect(result).toBe('stage-new');
    });

    it('should throw error when no stage matches', async () => {
      client.get = vi.fn().mockResolvedValue(mockFunnels[0]);
      await expect(
        resolveLeadStageId(client, 'funnel-1', 'Nonexistent Stage')
      ).rejects.toThrow('No stage found matching "Nonexistent Stage"');
    });

    it('should throw AmbiguousMatchError for ambiguous stage match', async () => {
      const funnelWithAmbiguousStages = {
        id: 'funnel-ambig',
        name: 'Test Funnel',
        stages: [
          { stageId: 's1', key: 'new', name: 'New Lead', desc: '' },
          { stageId: 's2', key: 'new-qualified', name: 'New Qualified', desc: '' },
        ],
      };
      client.get = vi.fn().mockResolvedValue(funnelWithAmbiguousStages);

      try {
        await resolveLeadStageId(client, 'funnel-ambig', 'New');
        expect.fail('Should have thrown');
      } catch (error) {
        expect(error).toBeInstanceOf(AmbiguousMatchError);
        const ambError = error as AmbiguousMatchError;
        expect(ambError.candidates).toHaveLength(2);
      }
    });
  });
});
```
