import { HoldedClient } from '../holded-client.js';
import {
  leadIdSchema,
  createLeadSchema,
  updateLeadSchema,
  createLeadNoteSchema,
  updateLeadNoteSchema,
  createLeadTaskSchema,
  searchLeadSchema,
  listLeadsSchema,
  withValidation,
} from '../validation.js';
import { resolveFunnelId } from '../utils/id-resolver.js';

interface Lead {
  id: string;
  name: string;
  contactName?: string;
  funnelId?: string;
  stageId?: string;
  value?: number;
  status?: number;
  [key: string]: unknown;
}

interface Funnel {
  id: string;
  name: string;
  stages: Array<{ stageId: string; name: string; [key: string]: unknown }>;
}

/**
 * Enrich leads with funnel and stage names by looking up funnels.
 */
async function enrichLeadsWithNames(
  client: HoldedClient,
  leads: Lead[]
): Promise<Lead[]> {
  // Collect unique funnelIds
  const funnelIds = [...new Set(leads.map((l) => l.funnelId).filter(Boolean))] as string[];
  if (funnelIds.length === 0) return leads;

  // Fetch all funnels once
  const funnels = await client.get<Funnel[]>('/funnels', undefined, 'crm');
  const funnelMap = new Map(funnels.map((f) => [f.id, f]));

  return leads.map((lead) => {
    const funnel = lead.funnelId ? funnelMap.get(lead.funnelId) : undefined;
    const stage = funnel?.stages.find((s) => s.stageId === lead.stageId);
    return {
      ...lead,
      _funnelName: funnel?.name || null,
      _stageName: stage?.name || null,
    };
  });
}

export function getLeadTools(client: HoldedClient) {
  return {
    holded_list_leads: {
      description:
        'List CRM leads with optional funnelId filter. Returns paginated results enriched with funnel and stage names.',
      inputSchema: {
        type: 'object' as const,
        properties: {
          funnelId: {
            type: 'string',
            description: 'Filter by funnel ID (optional)',
          },
          page: {
            type: 'number',
            description: 'Page number (default: 1)',
          },
          limit: {
            type: 'number',
            description: 'Items per page (default: 50, max: 500)',
          },
        },
        required: [],
      },
      readOnlyHint: true,
      handler: withValidation(listLeadsSchema, async (args) => {
        const allLeads = await client.get<Lead[]>('/leads', undefined, 'crm');

        // Client-side filter by funnelId
        let filtered = allLeads;
        if (args.funnelId) {
          filtered = allLeads.filter((l) => l.funnelId === args.funnelId);
        }

        // Enrich with funnel/stage names
        const enriched = await enrichLeadsWithNames(client, filtered);

        // Client-side pagination
        const page = args.page ?? 1;
        const limit = Math.min(args.limit ?? 50, 500);
        const startIndex = (page - 1) * limit;
        const endIndex = startIndex + limit;
        const items = enriched.slice(startIndex, endIndex);

        return {
          items,
          page,
          pageSize: items.length,
          totalItems: enriched.length,
          totalPages: Math.ceil(enriched.length / limit),
          hasMore: endIndex < enriched.length,
        };
      }),
    },

    holded_search_lead: {
      description:
        'Search CRM leads by name or contact name (fuzzy, case-insensitive). Optionally filter by funnelId.',
      inputSchema: {
        type: 'object' as const,
        properties: {
          query: {
            type: 'string',
            description: 'Search query (matches lead name or contactName)',
          },
          funnelId: {
            type: 'string',
            description: 'Filter by funnel ID (optional)',
          },
          page: {
            type: 'number',
            description: 'Page number (default: 1)',
          },
          limit: {
            type: 'number',
            description: 'Items per page (default: 50, max: 500)',
          },
        },
        required: ['query'],
      },
      readOnlyHint: true,
      handler: withValidation(searchLeadSchema, async (args) => {
        const allLeads = await client.get<Lead[]>('/leads', undefined, 'crm');
        const q = args.query.toLowerCase();

        let filtered = allLeads.filter((lead) => {
          const nameMatch = lead.name?.toLowerCase().includes(q);
          const contactMatch = lead.contactName?.toLowerCase().includes(q);
          return nameMatch || contactMatch;
        });

        if (args.funnelId) {
          filtered = filtered.filter((l) => l.funnelId === args.funnelId);
        }

        // Enrich with names
        const enriched = await enrichLeadsWithNames(client, filtered);

        // Paginate
        const page = args.page ?? 1;
        const limit = Math.min(args.limit ?? 50, 500);
        const startIndex = (page - 1) * limit;
        const endIndex = startIndex + limit;
        const items = enriched.slice(startIndex, endIndex);

        return {
          items,
          page,
          pageSize: items.length,
          totalItems: enriched.length,
          totalPages: Math.ceil(enriched.length / limit),
          hasMore: endIndex < enriched.length,
        };
      }),
    },

    holded_get_lead: {
      description:
        'Get full detail of a CRM lead by ID, including notes, tasks, and events.',
      inputSchema: {
        type: 'object' as const,
        properties: {
          leadId: {
            type: 'string',
            description: 'Lead ID',
          },
        },
        required: ['leadId'],
      },
      readOnlyHint: true,
      handler: withValidation(leadIdSchema, async (args) => {
        return client.get(`/leads/${args.leadId}`, undefined, 'crm');
      }),
    },

    holded_create_lead: {
      description:
        'Create a new CRM lead. funnelId can be a name (auto-resolved to ID). stageId can be a stage name.',
      inputSchema: {
        type: 'object' as const,
        properties: {
          funnelId: {
            type: 'string',
            description: 'Funnel ID or name (required, will be auto-resolved)',
          },
          name: {
            type: 'string',
            description: 'Lead name',
          },
          contactId: {
            type: 'string',
            description: 'Associated contact ID',
          },
          contactName: {
            type: 'string',
            description: 'Contact name (if no contactId)',
          },
          value: {
            type: 'number',
            description: 'Deal value',
          },
          potential: {
            type: 'number',
            description: 'Potential value',
          },
          dueDate: {
            type: 'number',
            description: 'Due date as Unix timestamp (seconds)',
          },
          stageId: {
            type: 'string',
            description: 'Stage ID or name within the funnel',
          },
        },
        required: ['funnelId'],
      },
      destructiveHint: true,
      handler: withValidation(createLeadSchema, async (args) => {
        // Resolve funnelId from name if needed
        const resolvedFunnelId = await resolveFunnelId(client, args.funnelId);
        const body: Record<string, unknown> = { ...args, funnelId: resolvedFunnelId };
        return client.post('/leads', body, 'crm');
      }),
    },

    holded_update_lead: {
      description:
        'Update a CRM lead. If stageId is provided, the lead is also moved to that stage. stageId accepts stage name.',
      inputSchema: {
        type: 'object' as const,
        properties: {
          leadId: {
            type: 'string',
            description: 'Lead ID',
          },
          name: {
            type: 'string',
            description: 'Updated lead name',
          },
          value: {
            type: 'number',
            description: 'Updated deal value',
          },
          dueDate: {
            type: 'number',
            description: 'Updated due date (Unix timestamp seconds)',
          },
          status: {
            type: 'number',
            description: 'Lead status: 0=open, 1=won, 2=lost',
          },
          stageId: {
            type: 'string',
            description: 'Move to this stage (ID or name)',
          },
        },
        required: ['leadId'],
      },
      destructiveHint: true,
      handler: withValidation(updateLeadSchema, async (args) => {
        const { leadId, stageId, ...updateFields } = args;

        const results: unknown[] = [];

        // Update lead fields if any provided
        if (Object.keys(updateFields).length > 0) {
          const updateResult = await client.put(`/leads/${leadId}`, updateFields, 'crm');
          results.push(updateResult);
        }

        // Move stage if stageId provided (separate API endpoint)
        if (stageId) {
          const stageResult = await client.put(
            `/leads/${leadId}/stages`,
            { stageId },
            'crm'
          );
          results.push(stageResult);
        }

        return results.length === 1 ? results[0] : { updates: results };
      }),
    },

    holded_create_lead_note: {
      description: 'Add a note to a CRM lead for follow-up tracking.',
      inputSchema: {
        type: 'object' as const,
        properties: {
          leadId: {
            type: 'string',
            description: 'Lead ID',
          },
          title: {
            type: 'string',
            description: 'Note title',
          },
          desc: {
            type: 'string',
            description: 'Note description/body',
          },
        },
        required: ['leadId', 'title'],
      },
      destructiveHint: true,
      handler: withValidation(createLeadNoteSchema, async (args) => {
        const { leadId, ...body } = args;
        return client.post(`/leads/${leadId}/notes`, body, 'crm');
      }),
    },

    holded_update_lead_note: {
      description:
        'Update an existing note on a CRM lead. QUIRK: noteId goes in the body, not the URL path.',
      inputSchema: {
        type: 'object' as const,
        properties: {
          leadId: {
            type: 'string',
            description: 'Lead ID',
          },
          noteId: {
            type: 'string',
            description: 'Note ID to update',
          },
          title: {
            type: 'string',
            description: 'Updated note title',
          },
          desc: {
            type: 'string',
            description: 'Updated note description',
          },
        },
        required: ['leadId', 'noteId'],
      },
      destructiveHint: true,
      handler: withValidation(updateLeadNoteSchema, async (args) => {
        const { leadId, ...body } = args;
        // API quirk: noteId goes in body, not path
        return client.put(`/leads/${leadId}/notes`, body, 'crm');
      }),
    },

    holded_create_lead_task: {
      description: 'Create a task on a CRM lead.',
      inputSchema: {
        type: 'object' as const,
        properties: {
          leadId: {
            type: 'string',
            description: 'Lead ID',
          },
          name: {
            type: 'string',
            description: 'Task name',
          },
        },
        required: ['leadId', 'name'],
      },
      destructiveHint: true,
      handler: withValidation(createLeadTaskSchema, async (args) => {
        const { leadId, ...body } = args;
        return client.post(`/leads/${leadId}/tasks`, body, 'crm');
      }),
    },
  };
}
