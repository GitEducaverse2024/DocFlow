**2. Create `src/tools/events.ts`:**

```typescript
import { HoldedClient } from '../holded-client.js';
import { listEventsSchema, createEventSchema, withValidation } from '../validation.js';

interface HoldedEvent {
  id: string;
  name: string;
  startDate?: number;
  kind?: string;
  [key: string]: unknown;
}

export function getEventTools(client: HoldedClient) {
  return {
    holded_list_events: {
      description:
        'List CRM events (calls, meetings, emails, etc.) with optional date range filtering. Dates are Unix timestamps in seconds.',
      inputSchema: {
        type: 'object' as const,
        properties: {
          dateFrom: {
            type: 'number',
            description: 'Filter events starting from this Unix timestamp (seconds)',
          },
          dateTo: {
            type: 'number',
            description: 'Filter events until this Unix timestamp (seconds)',
          },
          page: {
            type: 'number',
            description: 'Page number (default: 1)',
          },
          limit: {
            type: 'number',
            description: 'Items per page (default: 50, max: 500)',
          },
        },
        required: [],
      },
      readOnlyHint: true,
      handler: withValidation(listEventsSchema, async (args) => {
        const allEvents = await client.get<HoldedEvent[]>('/events', undefined, 'crm');

        // Client-side date filtering
        let filtered = allEvents;
        if (args.dateFrom) {
          filtered = filtered.filter((e) => (e.startDate ?? 0) >= args.dateFrom!);
        }
        if (args.dateTo) {
          filtered = filtered.filter((e) => (e.startDate ?? 0) <= args.dateTo!);
        }

        // Client-side pagination
        const page = args.page ?? 1;
        const limit = Math.min(args.limit ?? 50, 500);
        const startIndex = (page - 1) * limit;
        const endIndex = startIndex + limit;
        const items = filtered.slice(startIndex, endIndex);

        return {
          items,
          page,
          pageSize: items.length,
          totalItems: filtered.length,
          totalPages: Math.ceil(filtered.length / limit),
          hasMore: endIndex < filtered.length,
        };
      }),
    },

    holded_create_event: {
      description:
        'Create a CRM event (call, meeting, email, etc.). kind is free text — common values: meeting, call, email, lunch, dinner, flight.',
      inputSchema: {
        type: 'object' as const,
        properties: {
          name: {
            type: 'string',
            description: 'Event name/title (required)',
          },
          contactId: {
            type: 'string',
            description: 'Associated contact ID',
          },
          contactName: {
            type: 'string',
            description: 'Contact name (if no contactId)',
          },
          kind: {
            type: 'string',
            description: 'Event type: meeting, call, email, lunch, dinner, flight, or any text',
          },
          desc: {
            type: 'string',
            description: 'Event description',
          },
          startDate: {
            type: 'number',
            description: 'Start date as Unix timestamp (seconds)',
          },
          duration: {
            type: 'number',
            description: 'Duration in minutes',
          },
          status: {
            type: 'number',
            description: 'Event status',
          },
          tags: {
            type: 'array',
            items: { type: 'string' },
            description: 'Tags for the event',
          },
          locationDesc: {
            type: 'string',
            description: 'Location description',
          },
          leadId: {
            type: 'string',
            description: 'Associated lead ID',
          },
          funnelId: {
            type: 'string',
            description: 'Associated funnel ID',
          },
        },
        required: ['name'],
      },
      destructiveHint: true,
      handler: withValidation(createEventSchema, async (args) => {
        return client.post('/events', args, 'crm');
      }),
    },
  };
}
```

```typescript
import { describe, it, expect, beforeEach, vi } from 'vitest';
import { createMockClient } from './mock-client.js';
import { getEventTools } from '../tools/events.js';

const mockEvents = [
  { id: 'evt-1', name: 'Call with Acme', kind: 'call', startDate: 1700000000 },
  { id: 'evt-2', name: 'Meeting Beta Corp', kind: 'meeting', startDate: 1700100000 },
  { id: 'evt-3', name: 'Email follow-up', kind: 'email', startDate: 1700200000 },
  { id: 'evt-4', name: 'Lunch with client', kind: 'lunch', startDate: 1700300000 },
];

describe('Event Tools', () => {
  let client: ReturnType<typeof createMockClient>;
  let tools: ReturnType<typeof getEventTools>;

  beforeEach(() => {
    vi.clearAllMocks();
    client = createMockClient();
    tools = getEventTools(client);
  });

  describe('holded_list_events', () => {
    it('should list all events using CRM module', async () => {
      client.get = vi.fn().mockResolvedValue(mockEvents);

      const result = await tools.holded_list_events.handler({}) as any;

      expect(client.get).toHaveBeenCalledWith('/events', undefined, 'crm');
      expect(result.items).toHaveLength(4);
      expect(result.totalItems).toBe(4);
    });

    it('should filter by dateFrom', async () => {
      client.get = vi.fn().mockResolvedValue(mockEvents);

      const result = await tools.holded_list_events.handler({ dateFrom: 1700150000 }) as any;

      expect(result.items).toHaveLength(2); // evt-3 and evt-4
      expect(result.items[0].id).toBe('evt-3');
    });

    it('should filter by dateTo', async () => {
      client.get = vi.fn().mockResolvedValue(mockEvents);

      const result = await tools.holded_list_events.handler({ dateTo: 1700100000 }) as any;

      expect(result.items).toHaveLength(2); // evt-1 and evt-2
    });

    it('should filter by date range (dateFrom + dateTo)', async () => {
      client.get = vi.fn().mockResolvedValue(mockEvents);

      const result = await tools.holded_list_events.handler({
        dateFrom: 1700050000,
        dateTo: 1700250000,
      }) as any;

      expect(result.items).toHaveLength(2); // evt-2 and evt-3
    });

    it('should paginate results', async () => {
      client.get = vi.fn().mockResolvedValue(mockEvents);

      const result = await tools.holded_list_events.handler({ page: 1, limit: 2 }) as any;

      expect(result.items).toHaveLength(2);
      expect(result.hasMore).toBe(true);
      expect(result.totalPages).toBe(2);
    });

    it('should return empty when no events match date filter', async () => {
      client.get = vi.fn().mockResolvedValue(mockEvents);

      const result = await tools.holded_list_events.handler({ dateFrom: 9999999999 }) as any;

      expect(result.items).toHaveLength(0);
      expect(result.totalItems).toBe(0);
    });

    it('should handle events without startDate in date filtering', async () => {
      const eventsWithMissing = [
        ...mockEvents,
        { id: 'evt-5', name: 'No date event', kind: 'other' }, // no startDate
      ];
      client.get = vi.fn().mockResolvedValue(eventsWithMissing);

      // dateFrom filter: events without startDate have startDate=0, so they fail dateFrom > 0
      const result = await tools.holded_list_events.handler({ dateFrom: 1700000000 }) as any;

      expect(result.items).toHaveLength(4); // only events with startDate >= 1700000000
    });
  });

  describe('holded_create_event', () => {
    it('should create an event with required fields', async () => {
      await tools.holded_create_event.handler({ name: 'Team standup' });

      expect(client.post).toHaveBeenCalledWith('/events', { name: 'Team standup' }, 'crm');
    });

    it('should create an event with all optional fields', async () => {
      const args = {
        name: 'Client meeting',
        contactId: 'contact-123',
        kind: 'meeting',
        desc: 'Quarterly review',
        startDate: 1700000000,
        duration: 60,
        locationDesc: 'Office',
        leadId: 'lead-456',
        funnelId: 'funnel-789',
        tags: ['important', 'quarterly'],
      };

      await tools.holded_create_event.handler(args);

      expect(client.post).toHaveBeenCalledWith('/events', args, 'crm');
    });

    it('should accept any string as kind (free text)', async () => {
      await tools.holded_create_event.handler({
        name: 'Custom event',
        kind: 'workshop',
      });

      expect(client.post).toHaveBeenCalledWith(
        '/events',
        { name: 'Custom event', kind: 'workshop' },
        'crm'
      );
    });

    it('should reject empty name', async () => {
      await expect(
        tools.holded_create_event.handler({ name: '' })
      ).rejects.toThrow();
    });
  });
});
```
