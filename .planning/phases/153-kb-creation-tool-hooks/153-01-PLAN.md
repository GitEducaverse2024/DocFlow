    ```typescript
    // Phase 153 — Append-only audit file for syncResource hook failures.
    // Distinct from `_audit_stale.md` (fully regenerated by kb-sync.cjs --audit-stale).
    import * as fs from 'fs';
    import * as path from 'path';

    const DEFAULT_KB_ROOT = path.resolve(__dirname, '../../../../.docflow-kb');

    function getKbRoot(): string {
      return process['env']['KB_ROOT'] || DEFAULT_KB_ROOT;
    }

    export interface StaleEntry {
      entity: string;    // DB table name, e.g. 'cat_paws'
      db_id: string;
      error: string;
    }

    export type StaleReason =
      | 'create-sync-failed'
      | 'update-sync-failed'
      | 'delete-sync-failed'
      | 'markDeprecated-failed';

    /**
     * Append a Phase 153 hook failure to `.docflow-kb/_sync_failures.md`.
     * Never throws. First call lazily writes the frontmatter header.
     * Single-line entries (<200 bytes) are atomic per fs.appendFileSync on
     * Linux (PIPE_BUF 4096). Cluster mode may interleave lines rarely but
     * each line remains self-contained.
     *
     * Schema-valid against .docflow-kb/_schema/frontmatter.schema.json:
     * - type: audit, subtype: null, ttl: never, source_of_truth: null,
     *   title/summary strings (lang: es).
     * Excluded from validate-kb.cjs via EXCLUDED_FILENAMES so the append
     * format does not need strict schema reparse.
     */
    export function markStale(
      kbRelPath: string,
      reason: StaleReason,
      details?: StaleEntry,
    ): void {
      try {
        const kbRoot = getKbRoot();
        const filePath = path.join(kbRoot, '_sync_failures.md');

        if (!fs.existsSync(filePath)) {
          const iso = new Date().toISOString();
          const header = [
            '---',
            'id: sync-failures',
            'type: audit',
            'subtype: null',
            'lang: es',
            'title: Sync Failures Log',
            'summary: Append-only log of Phase 153 hook failures for reconciliation.',
            'tags: [ops]',
            'audience: [developer]',
            'status: active',
            `created_at: ${iso}`,
            'created_by: kb-audit',
            'version: 1.0.0',
            `updated_at: ${iso}`,
            'updated_by: kb-audit',
            'source_of_truth: null',
            'ttl: never',
            '---',
            '',
            '# Sync Failures Log',
            '',
            'Entries appended by Phase 153 hooks when `syncResource()` throws. Run',
            '`node scripts/kb-sync.cjs --full-rebuild --source db` to reconcile.',
            '',
            '| Timestamp | Reason | Entity | DB ID | KB Path | Error |',
            '|-----------|--------|--------|-------|---------|-------|',
            '',
          ].join('\n');
          fs.mkdirSync(path.dirname(filePath), { recursive: true });
          fs.writeFileSync(filePath, header);
        }

        const ts = new Date().toISOString();
        const errRaw = details?.error ?? '';
        const errTrunc = errRaw.replace(/\|/g, '\u2502').slice(0, 100);
        const line =
          `| ${ts} | ${reason} | ${details?.entity ?? '?'} | ${details?.db_id ?? '?'} | ${kbRelPath} | ${errTrunc} |\n`;
        fs.appendFileSync(filePath, line);
      } catch {
        // Last-ditch fallback: swallow. logger.error() at the call site is the
        // canonical record; `_sync_failures.md` being unwritable is not fatal.
      }
    }
    ```
