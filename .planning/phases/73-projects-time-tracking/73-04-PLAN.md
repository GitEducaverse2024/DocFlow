```typescript
/**
 * Date/time conversion utilities for Holded API.
 * Holded uses Unix timestamps in seconds everywhere.
 * JavaScript uses milliseconds.
 */

/**
 * Convert JS Date or ms timestamp to Holded Unix timestamp (seconds).
 * Smart detection: if input > ~year 2100 in seconds, assumes it's milliseconds.
 */
export function toHoldedTimestamp(date: Date | number): number {
  if (date instanceof Date) {
    return Math.floor(date.getTime() / 1000);
  }
  // Heuristic: timestamps above 4102444800 (year 2100 in seconds) are likely milliseconds
  if (date > 4102444800) {
    return Math.floor(date / 1000);
  }
  return date;
}

/**
 * Convert Holded Unix timestamp (seconds) to JS Date.
 */
export function fromHoldedTimestamp(timestamp: number): Date {
  return new Date(timestamp * 1000);
}

/**
 * Convert Holded duration (seconds) to human-readable string "Xh Ym".
 */
export function formatDuration(seconds: number): string {
  const hours = Math.floor(seconds / 3600);
  const minutes = Math.floor((seconds % 3600) / 60);
  return `${hours}h ${minutes}m`;
}

/**
 * Convert hours and optional minutes to Holded duration (seconds).
 */
export function toDurationSeconds(hours: number, minutes: number = 0): number {
  return (hours * 3600) + (minutes * 60);
}

/**
 * Calculate total cost from duration (seconds) and hourly rate.
 * Mirrors Holded's server-side calculation: (duration / 3600) * costHour.
 */
export function calculateTotal(durationSeconds: number, costHour: number): number {
  return (durationSeconds / 3600) * costHour;
}
```

```typescript
import { describe, it, expect } from 'vitest';
import {
  toHoldedTimestamp,
  fromHoldedTimestamp,
  formatDuration,
  toDurationSeconds,
  calculateTotal,
} from '../utils/date-helpers.js';

describe('Date Helpers', () => {
  describe('toHoldedTimestamp', () => {
    it('should convert Date object to seconds', () => {
      const date = new Date('2024-01-15T12:00:00Z');
      const result = toHoldedTimestamp(date);
      expect(result).toBe(Math.floor(date.getTime() / 1000));
    });

    it('should convert milliseconds to seconds', () => {
      const ms = 1705320000000; // 2024-01-15T12:00:00Z in ms
      const result = toHoldedTimestamp(ms);
      expect(result).toBe(1705320000);
    });

    it('should pass through seconds unchanged', () => {
      const seconds = 1705320000; // already in seconds
      const result = toHoldedTimestamp(seconds);
      expect(result).toBe(1705320000);
    });

    it('should handle edge case near the heuristic boundary', () => {
      // Value below 4102444800 should be treated as seconds
      const seconds = 4102444799;
      expect(toHoldedTimestamp(seconds)).toBe(4102444799);

      // Value above 4102444800 should be treated as milliseconds
      const ms = 4102444801000;
      expect(toHoldedTimestamp(ms)).toBe(4102444801);
    });
  });

  describe('fromHoldedTimestamp', () => {
    it('should convert seconds to Date', () => {
      const date = fromHoldedTimestamp(1705320000);
      expect(date.getTime()).toBe(1705320000000);
    });

    it('should return valid Date for timestamp 0', () => {
      const date = fromHoldedTimestamp(0);
      expect(date.getTime()).toBe(0);
    });
  });

  describe('formatDuration', () => {
    it('should format 1 hour', () => {
      expect(formatDuration(3600)).toBe('1h 0m');
    });

    it('should format 1.5 hours', () => {
      expect(formatDuration(5400)).toBe('1h 30m');
    });

    it('should format 0 seconds', () => {
      expect(formatDuration(0)).toBe('0h 0m');
    });

    it('should round down sub-minute values', () => {
      expect(formatDuration(45)).toBe('0h 0m');
      expect(formatDuration(90)).toBe('0h 1m');
    });

    it('should format large durations', () => {
      expect(formatDuration(45300)).toBe('12h 35m');
    });
  });

  describe('toDurationSeconds', () => {
    it('should convert hours to seconds', () => {
      expect(toDurationSeconds(2)).toBe(7200);
    });

    it('should convert hours and minutes to seconds', () => {
      expect(toDurationSeconds(1, 30)).toBe(5400);
    });

    it('should handle 0 hours', () => {
      expect(toDurationSeconds(0, 45)).toBe(2700);
    });
  });

  describe('calculateTotal', () => {
    it('should calculate 1 hour at 35/hr', () => {
      expect(calculateTotal(3600, 35)).toBe(35);
    });

    it('should calculate 1.5 hours at 40/hr', () => {
      expect(calculateTotal(5400, 40)).toBe(60);
    });

    it('should match Holded API example (45300s at 35/hr)', () => {
      const result = calculateTotal(45300, 35);
      expect(result).toBeCloseTo(440.4167, 3);
    });

    it('should return 0 for 0 duration', () => {
      expect(calculateTotal(0, 35)).toBe(0);
    });

    it('should return 0 for 0 costHour', () => {
      expect(calculateTotal(3600, 0)).toBe(0);
    });
  });
});
```
