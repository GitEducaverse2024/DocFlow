- GET /projects/{projectId}/summary — financial/progress summary

**2. Create `src/tools/projects.ts`:**

```typescript
import { HoldedClient } from '../holded-client.js';
import {
  projectIdSchema,
  listProjectsSchema,
  createProjectSchema,
  updateProjectSchema,
  withValidation,
} from '../validation.js';

interface HoldedProject {
  id: string;
  name: string;
  [key: string]: unknown;
}

export function getProjectTools(client: HoldedClient) {
  return {
    holded_list_projects: {
      description:
        'List all Holded projects with client-side pagination. Returns project name, status, task counts, and financial summary.',
      inputSchema: {
        type: 'object' as const,
        properties: {
          page: {
            type: 'number',
            description: 'Page number (default: 1)',
          },
          limit: {
            type: 'number',
            description: 'Items per page (default: 50, max: 500)',
          },
        },
        required: [],
      },
      readOnlyHint: true,
      handler: withValidation(listProjectsSchema, async (args) => {
        const allProjects = await client.get<HoldedProject[]>('/projects', undefined, 'projects');

        const page = args.page ?? 1;
        const limit = Math.min(args.limit ?? 50, 500);
        const startIndex = (page - 1) * limit;
        const endIndex = startIndex + limit;
        const items = allProjects.slice(startIndex, endIndex);

        return {
          items,
          page,
          pageSize: items.length,
          totalItems: allProjects.length,
          totalPages: Math.ceil(allProjects.length / limit),
          hasMore: endIndex < allProjects.length,
        };
      }),
    },

    holded_get_project: {
      description:
        'Get a specific Holded project by ID, including its lists (kanban columns), labels, and task/financial summary.',
      inputSchema: {
        type: 'object' as const,
        properties: {
          projectId: {
            type: 'string',
            description: 'Project ID',
          },
        },
        required: ['projectId'],
      },
      readOnlyHint: true,
      handler: withValidation(projectIdSchema, async (args) => {
        return client.get(`/projects/${args.projectId}`, undefined, 'projects');
      }),
    },

    holded_create_project: {
      description:
        'Create a new Holded project. Only name is required on creation. Use holded_update_project to set description, dates, tags, etc.',
      inputSchema: {
        type: 'object' as const,
        properties: {
          name: {
            type: 'string',
            description: 'Project name (required)',
          },
        },
        required: ['name'],
      },
      destructiveHint: true,
      handler: withValidation(createProjectSchema, async (args) => {
        return client.post('/projects', args, 'projects');
      }),
    },

    holded_update_project: {
      description:
        'Update a Holded project. Partial update — only included fields are changed. Dates are Unix timestamps in seconds.',
      inputSchema: {
        type: 'object' as const,
        properties: {
          projectId: {
            type: 'string',
            description: 'Project ID (required)',
          },
          name: {
            type: 'string',
            description: 'Project name',
          },
          desc: {
            type: 'string',
            description: 'Project description',
          },
          tags: {
            type: 'array',
            items: { type: 'string' },
            description: 'Tags array',
          },
          contactName: {
            type: 'string',
            description: 'Associated contact name',
          },
          date: {
            type: 'number',
            description: 'Start date (Unix timestamp seconds)',
          },
          dueDate: {
            type: 'number',
            description: 'Due date (Unix timestamp seconds)',
          },
          status: {
            type: 'number',
            description: 'Project status',
          },
          billable: {
            type: 'number',
            description: '0 = not billable, 1 = billable',
          },
          price: {
            type: 'number',
            description: 'Project price',
          },
        },
        required: ['projectId'],
      },
      destructiveHint: true,
      handler: withValidation(updateProjectSchema, async (args) => {
        const { projectId, ...body } = args;
        return client.put(`/projects/${projectId}`, body, 'projects');
      }),
    },

    holded_delete_project: {
      description: 'Delete a Holded project by ID. This is irreversible.',
      inputSchema: {
        type: 'object' as const,
        properties: {
          projectId: {
            type: 'string',
            description: 'Project ID to delete',
          },
        },
        required: ['projectId'],
      },
      destructiveHint: true,
      handler: withValidation(projectIdSchema, async (args) => {
        return client.delete(`/projects/${args.projectId}`, undefined, 'projects');
      }),
    },

    holded_get_project_summary: {
      description:
        'Get financial and progress summary for a Holded project: profitability, sales, expenses, task completion, and economic status.',
      inputSchema: {
        type: 'object' as const,
        properties: {
          projectId: {
            type: 'string',
            description: 'Project ID',
          },
        },
        required: ['projectId'],
      },
      readOnlyHint: true,
      handler: withValidation(projectIdSchema, async (args) => {
        return client.get(`/projects/${args.projectId}/summary`, undefined, 'projects');
      }),
    },
  };
}
```

```typescript
import { describe, it, expect, beforeEach, vi } from 'vitest';
import { createMockClient } from './mock-client.js';
import { getProjectTools } from '../tools/projects.js';

const mockProjects = [
  {
    id: 'proj-1',
    name: 'Website Redesign',
    status: 0,
    numberOfTasks: 10,
    completedTasks: 5,
    lists: [{ listId: 'list-1', key: 'pending', name: 'Pending', desc: '' }],
  },
  {
    id: 'proj-2',
    name: 'Mobile App',
    status: 1,
    numberOfTasks: 8,
    completedTasks: 8,
    lists: [],
  },
  {
    id: 'proj-3',
    name: 'API Overhaul',
    status: 2,
    numberOfTasks: 15,
    completedTasks: 3,
    lists: [],
  },
];

describe('Project Tools', () => {
  let client: ReturnType<typeof createMockClient>;
  let tools: ReturnType<typeof getProjectTools>;

  beforeEach(() => {
    vi.clearAllMocks();
    client = createMockClient();
    tools = getProjectTools(client);
  });

  describe('holded_list_projects', () => {
    it('should list all projects using projects module', async () => {
      client.get = vi.fn().mockResolvedValue(mockProjects);

      const result = (await tools.holded_list_projects.handler({})) as any;

      expect(client.get).toHaveBeenCalledWith('/projects', undefined, 'projects');
      expect(result.items).toHaveLength(3);
      expect(result.totalItems).toBe(3);
    });

    it('should paginate results', async () => {
      client.get = vi.fn().mockResolvedValue(mockProjects);

      const result = (await tools.holded_list_projects.handler({ page: 1, limit: 2 })) as any;

      expect(result.items).toHaveLength(2);
      expect(result.hasMore).toBe(true);
      expect(result.totalPages).toBe(2);
    });

    it('should return empty array when no projects', async () => {
      client.get = vi.fn().mockResolvedValue([]);

      const result = (await tools.holded_list_projects.handler({})) as any;

      expect(result.items).toHaveLength(0);
      expect(result.totalItems).toBe(0);
    });
  });

  describe('holded_get_project', () => {
    it('should get a project by ID', async () => {
      const mockProject = { id: 'proj-1', name: 'Website Redesign', lists: [], labels: [] };
      client.get = vi.fn().mockResolvedValue(mockProject);

      const result = await tools.holded_get_project.handler({ projectId: 'proj-1' });

      expect(client.get).toHaveBeenCalledWith('/projects/proj-1', undefined, 'projects');
      expect(result).toEqual(mockProject);
    });

    it('should reject empty projectId', async () => {
      await expect(tools.holded_get_project.handler({ projectId: '' })).rejects.toThrow();
    });
  });

  describe('holded_create_project', () => {
    it('should create a project with name', async () => {
      client.post = vi.fn().mockResolvedValue({ status: 1, info: 'Created', id: 'new-proj' });

      const result = await tools.holded_create_project.handler({ name: 'New Project' });

      expect(client.post).toHaveBeenCalledWith('/projects', { name: 'New Project' }, 'projects');
      expect(result).toEqual({ status: 1, info: 'Created', id: 'new-proj' });
    });

    it('should reject empty name', async () => {
      await expect(tools.holded_create_project.handler({ name: '' })).rejects.toThrow();
    });
  });

  describe('holded_update_project', () => {
    it('should update project fields', async () => {
      client.put = vi.fn().mockResolvedValue({ status: 1, info: 'Updated', id: 'proj-1' });

      await tools.holded_update_project.handler({
        projectId: 'proj-1',
        name: 'Updated Name',
        desc: 'New description',
        billable: 1,
      });

      expect(client.put).toHaveBeenCalledWith(
        '/projects/proj-1',
        { name: 'Updated Name', desc: 'New description', billable: 1 },
        'projects'
      );
    });

    it('should send only provided fields (partial update)', async () => {
      client.put = vi.fn().mockResolvedValue({ status: 1 });

      await tools.holded_update_project.handler({
        projectId: 'proj-1',
        status: 2,
      });

      expect(client.put).toHaveBeenCalledWith('/projects/proj-1', { status: 2 }, 'projects');
    });
  });

  describe('holded_delete_project', () => {
    it('should delete a project by ID', async () => {
      client.delete = vi.fn().mockResolvedValue({ status: 1, info: 'Successfully deleted' });

      await tools.holded_delete_project.handler({ projectId: 'proj-1' });

      expect(client.delete).toHaveBeenCalledWith('/projects/proj-1', undefined, 'projects');
    });
  });

  describe('holded_get_project_summary', () => {
    it('should get project summary', async () => {
      const mockSummary = {
        name: 'Website Redesign',
        projectEvolution: { tasks: { total: 10, completed: 5 }, dueDate: 0 },
        profitability: { sales: 5000, expenses: { documents: 0, personnel: 2000, total: 2000 }, profit: 3000 },
        economicStatus: { sales: 5000, quoted: 10000, difference: -5000 },
      };
      client.get = vi.fn().mockResolvedValue(mockSummary);

      const result = await tools.holded_get_project_summary.handler({ projectId: 'proj-1' });

      expect(client.get).toHaveBeenCalledWith('/projects/proj-1/summary', undefined, 'projects');
      expect(result).toEqual(mockSummary);
    });
  });
});
```
