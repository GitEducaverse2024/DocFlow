**2. Create `src/tools/project-tasks.ts`:**

```typescript
import { HoldedClient } from '../holded-client.js';
import {
  projectTaskIdSchema,
  listProjectTasksSchema,
  createProjectTaskSchema,
  withValidation,
} from '../validation.js';

interface ProjectTask {
  id: string;
  projectId: string;
  listId: string;
  name: string;
  [key: string]: unknown;
}

export function getProjectTaskTools(client: HoldedClient) {
  return {
    holded_list_project_tasks: {
      description:
        'List project tasks with optional projectId filter. Returns tasks across all projects by default. Each task includes projectId, listId, name, status, comments.',
      inputSchema: {
        type: 'object' as const,
        properties: {
          projectId: {
            type: 'string',
            description: 'Filter by project ID (optional — returns all tasks if omitted)',
          },
          page: {
            type: 'number',
            description: 'Page number (default: 1)',
          },
          limit: {
            type: 'number',
            description: 'Items per page (default: 50, max: 500)',
          },
        },
        required: [],
      },
      readOnlyHint: true,
      handler: withValidation(listProjectTasksSchema, async (args) => {
        const allTasks = await client.get<ProjectTask[]>('/tasks', undefined, 'projects');

        // Client-side filter by projectId
        let filtered = allTasks;
        if (args.projectId) {
          filtered = allTasks.filter((t) => t.projectId === args.projectId);
        }

        // Client-side pagination
        const page = args.page ?? 1;
        const limit = Math.min(args.limit ?? 50, 500);
        const startIndex = (page - 1) * limit;
        const endIndex = startIndex + limit;
        const items = filtered.slice(startIndex, endIndex);

        return {
          items,
          page,
          pageSize: items.length,
          totalItems: filtered.length,
          totalPages: Math.ceil(filtered.length / limit),
          hasMore: endIndex < filtered.length,
        };
      }),
    },

    holded_get_project_task: {
      description:
        'Get a specific project task by ID, including comments, labels, and assignment details.',
      inputSchema: {
        type: 'object' as const,
        properties: {
          taskId: {
            type: 'string',
            description: 'Task ID',
          },
        },
        required: ['taskId'],
      },
      readOnlyHint: true,
      handler: withValidation(projectTaskIdSchema, async (args) => {
        return client.get(`/tasks/${args.taskId}`, undefined, 'projects');
      }),
    },

    holded_create_project_task: {
      description:
        'Create a new task in a Holded project. Requires projectId and listId (kanban column). Use holded_get_project first to see available lists.',
      inputSchema: {
        type: 'object' as const,
        properties: {
          projectId: {
            type: 'string',
            description: 'Parent project ID (required)',
          },
          listId: {
            type: 'string',
            description: 'List/column ID within the project (required — use holded_get_project to find valid list IDs)',
          },
          name: {
            type: 'string',
            description: 'Task name (required)',
          },
          desc: {
            type: 'string',
            description: 'Task description',
          },
          labels: {
            type: 'array',
            items: { type: 'string' },
            description: 'Label IDs to apply',
          },
          date: {
            type: 'number',
            description: 'Task date (Unix timestamp seconds)',
          },
          dueDate: {
            type: 'number',
            description: 'Due date (Unix timestamp seconds)',
          },
          userId: {
            type: 'string',
            description: 'Assigned user ID',
          },
          status: {
            type: 'number',
            description: '0 = open, 1 = completed',
          },
          billable: {
            type: 'number',
            description: '0 = not billable, 1 = billable',
          },
          featured: {
            type: 'number',
            description: '0 = normal, 1 = featured',
          },
        },
        required: ['projectId', 'listId', 'name'],
      },
      destructiveHint: true,
      handler: withValidation(createProjectTaskSchema, async (args) => {
        return client.post('/tasks', args, 'projects');
      }),
    },

    holded_delete_project_task: {
      description: 'Delete a project task by ID. This is irreversible.',
      inputSchema: {
        type: 'object' as const,
        properties: {
          taskId: {
            type: 'string',
            description: 'Task ID to delete',
          },
        },
        required: ['taskId'],
      },
      destructiveHint: true,
      handler: withValidation(projectTaskIdSchema, async (args) => {
        return client.delete(`/tasks/${args.taskId}`, undefined, 'projects');
      }),
    },
  };
}
```

import { describe, it, expect, beforeEach, vi } from 'vitest';
import { createMockClient } from './mock-client.js';
import { getProjectTaskTools } from '../tools/project-tasks.js';

const mockTasks = [
  {
    id: 'task-1',
    projectId: 'proj-1',
    listId: 'list-pending',
    name: 'Design mockups',
    status: 0,
    comments: [],
  },
  {
    id: 'task-2',
    projectId: 'proj-1',
    listId: 'list-done',
    name: 'Setup repo',
    status: 1,
    comments: [{ commentId: 'c1', message: 'Done!', createdAt: 1700000000, userId: 'u1' }],
  },
  {
    id: 'task-3',
    projectId: 'proj-2',
    listId: 'list-pending',
    name: 'Write API docs',
    status: 0,
    comments: [],
  },
];

describe('Project Task Tools', () => {
  let client: ReturnType<typeof createMockClient>;
  let tools: ReturnType<typeof getProjectTaskTools>;

  beforeEach(() => {
    vi.clearAllMocks();
    client = createMockClient();
    tools = getProjectTaskTools(client);
  });

  describe('holded_list_project_tasks', () => {
    it('should list all tasks using projects module', async () => {
      client.get = vi.fn().mockResolvedValue(mockTasks);

      const result = (await tools.holded_list_project_tasks.handler({})) as any;

      expect(client.get).toHaveBeenCalledWith('/tasks', undefined, 'projects');
      expect(result.items).toHaveLength(3);
      expect(result.totalItems).toBe(3);
    });

    it('should filter by projectId', async () => {
      client.get = vi.fn().mockResolvedValue(mockTasks);

      const result = (await tools.holded_list_project_tasks.handler({ projectId: 'proj-1' })) as any;

      expect(result.items).toHaveLength(2);
      expect(result.items.every((t: any) => t.projectId === 'proj-1')).toBe(true);
    });

    it('should paginate results', async () => {
      client.get = vi.fn().mockResolvedValue(mockTasks);

      const result = (await tools.holded_list_project_tasks.handler({ page: 1, limit: 2 })) as any;

      expect(result.items).toHaveLength(2);
      expect(result.hasMore).toBe(true);
    });

    it('should return empty when no tasks match filter', async () => {
      client.get = vi.fn().mockResolvedValue(mockTasks);

      const result = (await tools.holded_list_project_tasks.handler({ projectId: 'nonexistent' })) as any;

      expect(result.items).toHaveLength(0);
      expect(result.totalItems).toBe(0);
    });
  });

  describe('holded_get_project_task', () => {
    it('should get a task by ID', async () => {
      const mockTask = mockTasks[1];
      client.get = vi.fn().mockResolvedValue(mockTask);

      const result = await tools.holded_get_project_task.handler({ taskId: 'task-2' });

      expect(client.get).toHaveBeenCalledWith('/tasks/task-2', undefined, 'projects');
      expect(result).toEqual(mockTask);
    });

    it('should reject empty taskId', async () => {
      await expect(tools.holded_get_project_task.handler({ taskId: '' })).rejects.toThrow();
    });
  });

  describe('holded_create_project_task', () => {
    it('should create a task with required fields', async () => {
      client.post = vi.fn().mockResolvedValue({ status: 1, info: 'Created', id: 'new-task' });

      await tools.holded_create_project_task.handler({
        projectId: 'proj-1',
        listId: 'list-pending',
        name: 'New Task',
      });

      expect(client.post).toHaveBeenCalledWith(
        '/tasks',
        { projectId: 'proj-1', listId: 'list-pending', name: 'New Task' },
        'projects'
      );
    });

    it('should create a task with all optional fields', async () => {
      client.post = vi.fn().mockResolvedValue({ status: 1, info: 'Created', id: 'new-task' });

      const args = {
        projectId: 'proj-1',
        listId: 'list-pending',
        name: 'Full Task',
        desc: 'Detailed description',
        labels: ['label-1', 'label-2'],
        date: 1700000000,
        dueDate: 1700500000,
        userId: 'user-1',
        status: 0,
        billable: 1,
        featured: 0,
      };

      await tools.holded_create_project_task.handler(args);

      expect(client.post).toHaveBeenCalledWith('/tasks', args, 'projects');
    });

    it('should reject missing required fields', async () => {
      await expect(
        tools.holded_create_project_task.handler({ projectId: 'proj-1', listId: '', name: 'Test' } as any)
      ).rejects.toThrow();

      await expect(
        tools.holded_create_project_task.handler({ projectId: 'proj-1', listId: 'list-1', name: '' } as any)
      ).rejects.toThrow();
    });
  });

  describe('holded_delete_project_task', () => {
    it('should delete a task by ID', async () => {
      client.delete = vi.fn().mockResolvedValue({ status: 1, info: 'Successfully deleted' });

      await tools.holded_delete_project_task.handler({ taskId: 'task-1' });

      expect(client.delete).toHaveBeenCalledWith('/tasks/task-1', undefined, 'projects');
    });
  });
});
