```typescript
export type HoldedModule = 'invoicing' | 'crm' | 'projects' | 'team';
export class HoldedClient {
  async get<T>(endpoint: string, queryParams?: Record<string, string | number>, module?: HoldedModule): Promise<T>;
  async post<T>(endpoint: string, body?: unknown, module?: HoldedModule): Promise<T>;
  async put<T>(endpoint: string, body?: unknown, module?: HoldedModule): Promise<T>;
  async delete<T>(endpoint: string, body?: unknown, module?: HoldedModule): Promise<T>;
}
```

API details from RESEARCH.md:
- GET /projects/{projectId}/times — list time entries for a project (flat array)
- GET /projects/{projectId}/times/{timeTrackingId} — single time entry
- POST /projects/{projectId}/times — body: {duration (required, seconds), costHour (required), desc, userId, taskId}
- PUT /projects/{projectId}/times/{timeTrackingId} — partial update (same fields as create)
- DELETE /projects/{projectId}/times/{timeTrackingId} — standard mutation response
- GET /projects/times?start=X&end=Y&archived=bool — cross-project listing, grouped by project

QUIRK: duration is in SECONDS (1 hour = 3600). total = (duration / 3600) * costHour (server-calculated, read-only).
QUIRK: GET /projects/times returns [{id, name, timeTracking: [...]}] — grouped by project, NOT a flat array. Default 18-month window.
QUIRK: No batch/bulk operations. Each time entry created individually with 150ms rate limit delay.

```typescript
import { HoldedClient } from '../holded-client.js';
import {
  timeEntryIdSchema,
  listTimeEntriesSchema,
  createTimeEntrySchema,
  updateTimeEntrySchema,
  listAllTimesSchema,
  withValidation,
} from '../validation.js';

export function getTimeTrackingTools(client: HoldedClient) {
  return {
    holded_list_time_entries: {
      description:
        'List time tracking entries for a specific Holded project. Returns duration (seconds), costHour, total (calculated), desc, userId, taskId.',
      inputSchema: {
        type: 'object' as const,
        properties: {
          projectId: {
            type: 'string',
            description: 'Project ID (required)',
          },
        },
        required: ['projectId'],
      },
      readOnlyHint: true,
      handler: withValidation(listTimeEntriesSchema, async (args) => {
        return client.get(`/projects/${args.projectId}/times`, undefined, 'projects');
      }),
    },

    holded_list_all_time_entries: {
      description:
        'List time tracking entries across ALL projects, grouped by project. Supports date range filtering. Default: non-archived projects from last 18 months. Response: [{id, name, timeTracking: [...]}].',
      inputSchema: {
        type: 'object' as const,
        properties: {
          start: {
            type: 'number',
            description: 'Start date filter (Unix timestamp seconds)',
          },
          end: {
            type: 'number',
            description: 'End date filter (Unix timestamp seconds)',
          },
          archived: {
            type: 'boolean',
            description: 'Include archived projects (default: false)',
          },
        },
        required: [],
      },
      readOnlyHint: true,
      handler: withValidation(listAllTimesSchema, async (args) => {
        const queryParams: Record<string, string | number> = {};
        if (args.start !== undefined) queryParams.start = args.start;
        if (args.end !== undefined) queryParams.end = args.end;
        if (args.archived !== undefined) queryParams.archived = args.archived ? 1 : 0;

        const hasParams = Object.keys(queryParams).length > 0;
        return client.get(
          '/projects/times',
          hasParams ? queryParams : undefined,
          'projects'
        );
      }),
    },

    holded_get_time_entry: {
      description:
        'Get a specific time tracking entry by project ID and time tracking ID.',
      inputSchema: {
        type: 'object' as const,
        properties: {
          projectId: {
            type: 'string',
            description: 'Project ID',
          },
          timeTrackingId: {
            type: 'string',
            description: 'Time tracking entry ID',
          },
        },
        required: ['projectId', 'timeTrackingId'],
      },
      readOnlyHint: true,
      handler: withValidation(timeEntryIdSchema, async (args) => {
        return client.get(
          `/projects/${args.projectId}/times/${args.timeTrackingId}`,
          undefined,
          'projects'
        );
      }),
    },

    holded_create_time_entry: {
      description:
        'Create a time tracking entry on a Holded project. Duration is in SECONDS (1h = 3600, 30min = 1800). costHour is the hourly rate. Total is auto-calculated as (duration/3600)*costHour.',
      inputSchema: {
        type: 'object' as const,
        properties: {
          projectId: {
            type: 'string',
            description: 'Project ID (required)',
          },
          duration: {
            type: 'number',
            description: 'Duration in seconds (required). Examples: 3600 = 1h, 1800 = 30min, 7200 = 2h',
          },
          costHour: {
            type: 'number',
            description: 'Cost per hour rate (required)',
          },
          desc: {
            type: 'string',
            description: 'Description of the work done',
          },
          userId: {
            type: 'string',
            description: 'User who performed the work',
          },
          taskId: {
            type: 'string',
            description: 'Associated project task ID (optional)',
          },
        },
        required: ['projectId', 'duration', 'costHour'],
      },
      destructiveHint: true,
      handler: withValidation(createTimeEntrySchema, async (args) => {
        const { projectId, ...body } = args;
        return client.post(`/projects/${projectId}/times`, body, 'projects');
      }),
    },

    holded_update_time_entry: {
      description:
        'Update an existing time tracking entry. Partial update — only included fields are changed. Duration in seconds.',
      inputSchema: {
        type: 'object' as const,
        properties: {
          projectId: {
            type: 'string',
            description: 'Project ID (required)',
          },
          timeTrackingId: {
            type: 'string',
            description: 'Time tracking entry ID (required)',
          },
          duration: {
            type: 'number',
            description: 'Updated duration in seconds',
          },
          costHour: {
            type: 'number',
            description: 'Updated cost per hour rate',
          },
          desc: {
            type: 'string',
            description: 'Updated description',
          },
          userId: {
            type: 'string',
            description: 'Updated user ID',
          },
          taskId: {
            type: 'string',
            description: 'Updated task ID',
          },
        },
        required: ['projectId', 'timeTrackingId'],
      },
      destructiveHint: true,
      handler: withValidation(updateTimeEntrySchema, async (args) => {
        const { projectId, timeTrackingId, ...body } = args;
        return client.put(
          `/projects/${projectId}/times/${timeTrackingId}`,
          body,
          'projects'
        );
      }),
    },

    holded_delete_time_entry: {
      description: 'Delete a time tracking entry from a Holded project. This is irreversible.',
      inputSchema: {
        type: 'object' as const,
        properties: {
          projectId: {
            type: 'string',
            description: 'Project ID',
          },
          timeTrackingId: {
            type: 'string',
            description: 'Time tracking entry ID to delete',
          },
        },
        required: ['projectId', 'timeTrackingId'],
      },
      destructiveHint: true,
      handler: withValidation(timeEntryIdSchema, async (args) => {
        return client.delete(
          `/projects/${args.projectId}/times/${args.timeTrackingId}`,
          undefined,
          'projects'
        );
      }),
    },
  };
}
```

```typescript
import { describe, it, expect, beforeEach, vi } from 'vitest';
import { createMockClient } from './mock-client.js';
import { getTimeTrackingTools } from '../tools/time-tracking.js';

const mockTimeEntries = [
  {
    timeId: 'time-1',
    duration: 3600,
    desc: 'Backend work',
    costHour: 35,
    userId: 'user-1',
    taskId: 'task-1',
    total: 35,
  },
  {
    timeId: 'time-2',
    duration: 7200,
    desc: 'Frontend work',
    costHour: 40,
    userId: 'user-2',
    taskId: 'task-2',
    total: 80,
  },
];

const mockCrossProjectTimes = [
  {
    id: 'proj-1',
    name: 'Website Redesign',
    timeTracking: [mockTimeEntries[0]],
  },
  {
    id: 'proj-2',
    name: 'Mobile App',
    timeTracking: [mockTimeEntries[1]],
  },
];

describe('Time Tracking Tools', () => {
  let client: ReturnType<typeof createMockClient>;
  let tools: ReturnType<typeof getTimeTrackingTools>;

  beforeEach(() => {
    vi.clearAllMocks();
    client = createMockClient();
    tools = getTimeTrackingTools(client);
  });

  describe('holded_list_time_entries', () => {
    it('should list time entries for a project', async () => {
      client.get = vi.fn().mockResolvedValue(mockTimeEntries);

      const result = await tools.holded_list_time_entries.handler({ projectId: 'proj-1' });

      expect(client.get).toHaveBeenCalledWith('/projects/proj-1/times', undefined, 'projects');
      expect(result).toEqual(mockTimeEntries);
    });

    it('should reject empty projectId', async () => {
      await expect(tools.holded_list_time_entries.handler({ projectId: '' })).rejects.toThrow();
    });
  });

  describe('holded_list_all_time_entries', () => {
    it('should list all time entries across projects', async () => {
      client.get = vi.fn().mockResolvedValue(mockCrossProjectTimes);

      const result = await tools.holded_list_all_time_entries.handler({});

      expect(client.get).toHaveBeenCalledWith('/projects/times', undefined, 'projects');
      expect(result).toEqual(mockCrossProjectTimes);
    });

    it('should pass start/end query params', async () => {
      client.get = vi.fn().mockResolvedValue(mockCrossProjectTimes);

      await tools.holded_list_all_time_entries.handler({
        start: 1700000000,
        end: 1700500000,
      });

      expect(client.get).toHaveBeenCalledWith(
        '/projects/times',
        { start: 1700000000, end: 1700500000 },
        'projects'
      );
    });

    it('should pass archived param as 0/1', async () => {
      client.get = vi.fn().mockResolvedValue([]);

      await tools.holded_list_all_time_entries.handler({ archived: true });

      expect(client.get).toHaveBeenCalledWith(
        '/projects/times',
        { archived: 1 },
        'projects'
      );
    });

    it('should pass archived=false as 0', async () => {
      client.get = vi.fn().mockResolvedValue([]);

      await tools.holded_list_all_time_entries.handler({ archived: false });

      expect(client.get).toHaveBeenCalledWith(
        '/projects/times',
        { archived: 0 },
        'projects'
      );
    });
  });

  describe('holded_get_time_entry', () => {
    it('should get a time entry by project and time ID', async () => {
      client.get = vi.fn().mockResolvedValue(mockTimeEntries[0]);

      const result = await tools.holded_get_time_entry.handler({
        projectId: 'proj-1',
        timeTrackingId: 'time-1',
      });

      expect(client.get).toHaveBeenCalledWith('/projects/proj-1/times/time-1', undefined, 'projects');
      expect(result).toEqual(mockTimeEntries[0]);
    });

    it('should reject empty IDs', async () => {
      await expect(
        tools.holded_get_time_entry.handler({ projectId: '', timeTrackingId: 'time-1' })
      ).rejects.toThrow();
      await expect(
        tools.holded_get_time_entry.handler({ projectId: 'proj-1', timeTrackingId: '' })
      ).rejects.toThrow();
    });
  });

  describe('holded_create_time_entry', () => {
    it('should create a time entry with required fields', async () => {
      client.post = vi.fn().mockResolvedValue({ status: 1, info: 'Created', id: 'new-time' });

      await tools.holded_create_time_entry.handler({
        projectId: 'proj-1',
        duration: 3600,
        costHour: 35,
      });

      expect(client.post).toHaveBeenCalledWith(
        '/projects/proj-1/times',
        { duration: 3600, costHour: 35 },
        'projects'
      );
    });

    it('should create a time entry with all fields', async () => {
      client.post = vi.fn().mockResolvedValue({ status: 1, info: 'Created', id: 'new-time' });

      await tools.holded_create_time_entry.handler({
        projectId: 'proj-1',
        duration: 5400,
        costHour: 40,
        desc: '1.5 hours of API development',
        userId: 'user-1',
        taskId: 'task-1',
      });

      expect(client.post).toHaveBeenCalledWith(
        '/projects/proj-1/times',
        {
          duration: 5400,
          costHour: 40,
          desc: '1.5 hours of API development',
          userId: 'user-1',
          taskId: 'task-1',
        },
        'projects'
      );
    });

    it('should reject zero or negative duration', async () => {
      await expect(
        tools.holded_create_time_entry.handler({ projectId: 'proj-1', duration: 0, costHour: 35 })
      ).rejects.toThrow();
      await expect(
        tools.holded_create_time_entry.handler({ projectId: 'proj-1', duration: -100, costHour: 35 })
      ).rejects.toThrow();
    });

    it('should reject negative costHour', async () => {
      await expect(
        tools.holded_create_time_entry.handler({ projectId: 'proj-1', duration: 3600, costHour: -5 })
      ).rejects.toThrow();
    });
  });

  describe('holded_update_time_entry', () => {
    it('should update a time entry with partial fields', async () => {
      client.put = vi.fn().mockResolvedValue({ status: 1, info: 'Updated' });

      await tools.holded_update_time_entry.handler({
        projectId: 'proj-1',
        timeTrackingId: 'time-1',
        duration: 7200,
        desc: 'Updated description',
      });

      expect(client.put).toHaveBeenCalledWith(
        '/projects/proj-1/times/time-1',
        { duration: 7200, desc: 'Updated description' },
        'projects'
      );
    });

    it('should send only update fields (not projectId/timeTrackingId)', async () => {
      client.put = vi.fn().mockResolvedValue({ status: 1 });

      await tools.holded_update_time_entry.handler({
        projectId: 'proj-1',
        timeTrackingId: 'time-1',
        costHour: 50,
      });

      expect(client.put).toHaveBeenCalledWith(
        '/projects/proj-1/times/time-1',
        { costHour: 50 },
        'projects'
      );
    });
  });

  describe('holded_delete_time_entry', () => {
    it('should delete a time entry', async () => {
      client.delete = vi.fn().mockResolvedValue({ status: 1, info: 'Successfully deleted' });

      await tools.holded_delete_time_entry.handler({
        projectId: 'proj-1',
        timeTrackingId: 'time-1',
      });

      expect(client.delete).toHaveBeenCalledWith(
        '/projects/proj-1/times/time-1',
        undefined,
        'projects'
      );
    });
  });
});
```
