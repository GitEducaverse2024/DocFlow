    ```typescript
    case 'search_kb': {
      try {
        const params = args as {
          type?: string; subtype?: string; tags?: string[]; audience?: string;
          status?: string; search?: string; limit?: number;
        };
        const result = searchKb(params);
        return { name, result };
      } catch (err: unknown) {
        const message = err instanceof Error ? err.message : String(err);
        return { name, result: { error: message } };
      }
    }

    case 'get_kb_entry': {
      try {
        const id = args.id as string | undefined;
        if (!id || typeof id !== 'string' || id.trim() === '') {
          return { name, result: { error: 'id es obligatorio (string)' } };
        }
        const entry = getKbEntry(id);
        if (!entry) {
          return { name, result: { error: 'NOT_FOUND', id } };
        }
        return { name, result: entry };
      } catch (err: unknown) {
        const message = err instanceof Error ? err.message : String(err);
        return { name, result: { error: message } };
      }
    }
    ```
