```ts
export function resolveCatPawsForJob(jobId: string, catpaws: Array<{name:string, description?:string, system_prompt:string, mode?:string}>): { created: string[] } {
  const job = getIntentJob(jobId);
  if (!job) throw new Error('Job not found');
  if (job.pipeline_phase !== 'awaiting_user') throw new Error(`Job in phase ${job.pipeline_phase}, cannot approve catpaws`);
  const created: string[] = [];
  for (const cp of catpaws) {
    const id = generateId();
    db.prepare('INSERT INTO cat_paws (id, name, description, mode, system_prompt, is_active, created_at, updated_at) VALUES (?, ?, ?, ?, ?, 1, datetime(\'now\'), datetime(\'now\'))')
      .run(id, cp.name, cp.description ?? '', cp.mode ?? 'agent', cp.system_prompt);
    created.push(id);
  }
  const prev = JSON.parse(job.progress_message || '{}');
  updateIntentJob(jobId, {
    pipeline_phase: 'architect_retry',
    progressMessage: { ...prev, cat_paws_resolved: true, cat_paws_created: created, message: 'CatPaws creados, reanudando architect' },
  });
  return { created };
}
```

    ```ts
      if (job.pipeline_phase !== 'awaiting_approval') {

      if (!job.canvas_id) {

      updateIntentJob(job.id, { pipeline_phase: 'running', status: 'running' });

      const baseUrl = process['env']['INTERNAL_BASE_URL'] || 'http://localhost:3000';
      fetch(`${baseUrl}/api/canvas/${job.canvas_id}/execute`, { method: 'POST' }).catch(() => {
        // executor tick + alert service handle stuck cases
      });

      return NextResponse.json({ ok: true, canvas_id: job.canvas_id });
    }
    ```

    ```ts
    import { NextResponse } from 'next/server';
    import { getIntentJob, updateIntentJob } from '@/lib/catbot-db';
    import db from '@/lib/db';
    import { generateId } from '@/lib/utils';

    export const dynamic = 'force-dynamic';

    type CatPawInput = { name: string; description?: string; system_prompt: string; mode?: string };

    export async function POST(
      request: Request,
      { params }: { params: { id: string } },
    ): Promise<NextResponse> {
      const job = getIntentJob(params.id);
      if (!job) return NextResponse.json({ error: 'Job not found' }, { status: 404 });
      if (job.pipeline_phase !== 'awaiting_user') {
        return NextResponse.json({ error: `Job in phase ${job.pipeline_phase}, cannot approve catpaws` }, { status: 400 });
      }

      let body: { catpaws?: CatPawInput[] } = {};
      try { body = await request.json(); } catch { /* empty body ok */ }

      const prev = JSON.parse(job.progress_message || '{}') as { cat_paws_needed?: CatPawInput[] };
      const list = body.catpaws ?? prev.cat_paws_needed ?? [];
      if (!list.length) {
        return NextResponse.json({ error: 'No catpaws to create' }, { status: 400 });
      }

      const created: string[] = [];
      const stmt = db.prepare(
        "INSERT INTO cat_paws (id, name, description, mode, system_prompt, is_active, created_at, updated_at) VALUES (?, ?, ?, ?, ?, 1, datetime('now'), datetime('now'))",
      );
      for (const cp of list) {
        const id = generateId();
        stmt.run(id, cp.name, cp.description ?? '', cp.mode ?? 'agent', cp.system_prompt);
        created.push(id);
      }

      updateIntentJob(job.id, {
        pipeline_phase: 'architect_retry',
        progressMessage: { ...prev, cat_paws_resolved: true, cat_paws_created: created, message: 'CatPaws creados, reanudando architect' },
      });

      return NextResponse.json({ ok: true, created });
    }
    ```
