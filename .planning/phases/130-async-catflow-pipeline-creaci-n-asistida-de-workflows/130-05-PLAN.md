checkStuckPipelines pattern:
```ts
const STUCK_PIPELINE_THRESHOLD_MIN = 30;
static async checkStuckPipelines(): Promise<void> {
  const row = catbotDb.prepare(
    `SELECT COUNT(*) AS cnt FROM intent_jobs
     WHERE status = 'running' AND updated_at < datetime('now', '-${STUCK_PIPELINE_THRESHOLD_MIN} minutes')`
  ).get() as { cnt: number };
  if (row.cnt > 0) {
    this.insertAlert(
      'execution', 'pipelines_stuck', 'Pipelines atascados',
      `Hay ${row.cnt} intent_jobs en estado running sin actualizarse en >${STUCK_PIPELINE_THRESHOLD_MIN} minutos`,
      'warning', JSON.stringify({ count: row.cnt, threshold_min: STUCK_PIPELINE_THRESHOLD_MIN }),
    );
  }
}
```

    ```ts
    import { NextResponse } from 'next/server';
    import { listJobsByUser } from '@/lib/catbot-db';
    // Si existe auth helper: import { getServerSession } from '@/lib/auth';

    export const dynamic = 'force-dynamic';

    export async function GET(request: Request): Promise<NextResponse> {
      const url = new URL(request.url);
      let userId: string | null = null;

      // Intentar obtener de session (si existe auth)
      try {
        // const session = await getServerSession();
        // userId = session?.user?.id ?? null;
      } catch { /* dev mode sin auth */ }

      // Fallback dev: query param
      if (!userId) userId = url.searchParams.get('user_id');
      if (!userId) return NextResponse.json({ error: 'Unauthenticated' }, { status: 401 });

      const rows = listJobsByUser(userId, { limit: 50 });
      const jobs = rows.map(r => ({
        id: r.id,
        pipeline_phase: r.pipeline_phase,
        status: r.status,
        progress_message: (() => { try { return JSON.parse(r.progress_message || '{}'); } catch { return {}; } })(),
        canvas_id: r.canvas_id,
        channel: r.channel,
        created_at: r.created_at,
        updated_at: r.updated_at,
        completed_at: r.completed_at,
      }));
      return NextResponse.json({ jobs });
    }
    ```

    ```tsx
    'use client';
    import { useEffect, useState } from 'react';
    import { Card, CardContent, CardHeader, CardTitle } from '@/components/ui/card';
    import { Badge } from '@/components/ui/badge';
    import { Button } from '@/components/ui/button';
    import Link from 'next/link';

    interface Job {
      id: string;
      pipeline_phase: string;
      status: string;
      progress_message: { message?: string; goal?: string; [k: string]: unknown };
      canvas_id: string | null;
      channel: string;
      created_at: string;
      updated_at: string;
    }

    const PHASE_COLORS: Record<string, string> = {
      pending: 'bg-slate-200', strategist: 'bg-blue-200', decomposer: 'bg-indigo-200',
      architect: 'bg-violet-200', architect_retry: 'bg-violet-300',
      awaiting_approval: 'bg-amber-200', awaiting_user: 'bg-orange-200',
      running: 'bg-emerald-200', completed: 'bg-green-300', failed: 'bg-red-200', cancelled: 'bg-zinc-200',
    };

    export function TabPipelines(): JSX.Element {
      const [jobs, setJobs] = useState<Job[]>([]);
      const [loading, setLoading] = useState(true);
      const [error, setError] = useState<string | null>(null);

      const load = async (): Promise<void> => {
        try {
          const res = await fetch('/api/intent-jobs');
          if (!res.ok) throw new Error(`HTTP ${res.status}`);
          const data = await res.json() as { jobs: Job[] };
          setJobs(data.jobs);
          setError(null);
        } catch (err) {
          setError((err as Error).message);
        } finally {
          setLoading(false);
        }
      };

      useEffect(() => {
        void load();
        const interval = setInterval(() => void load(), 10_000);
        return () => clearInterval(interval);
      }, []);

      if (loading) return <div className="p-4 text-muted-foreground">Cargando pipelines...</div>;
      if (error) return <div className="p-4 text-red-600">Error: {error}</div>;
      if (jobs.length === 0) return <div className="p-4 text-muted-foreground">No hay pipelines activos ni recientes.</div>;

      return (
        <div className="space-y-3">
          {jobs.map(job => (
            <Card key={job.id}>
              <CardHeader className="pb-2 flex flex-row items-center justify-between">
                <CardTitle className="text-sm font-medium">
                  {String(job.progress_message.goal ?? 'Pipeline ' + job.id.slice(0, 8))}
                </CardTitle>
                <Badge className={PHASE_COLORS[job.pipeline_phase] ?? 'bg-gray-200'}>
                  {job.pipeline_phase}
                </Badge>
              </CardHeader>
              <CardContent className="text-sm space-y-2">
                <p className="text-muted-foreground">{String(job.progress_message.message ?? '—')}</p>
                <p className="text-xs text-muted-foreground">
                  {job.channel} • creado {new Date(job.created_at).toLocaleString()}
                </p>
                {job.pipeline_phase === 'awaiting_approval' && job.canvas_id && (
                  <Button asChild size="sm" variant="outline">
                    <Link href={`/catflow/${job.canvas_id}`}>Ver propuesta</Link>
                  </Button>
                )}
              </CardContent>
            </Card>
          ))}
        </div>
      );
    }
    ```
