2. Leer el flow_data actual, parsear JSON, localizar los nodos Normalizador y Respondedor por label, actualizar sus instructions con las mismas clausulas anadidas en Parte A, y escribir el flow_data actualizado:

```bash
docker exec docflow-app node -e "
const Database = require('better-sqlite3');
const db = new Database('/app/data/docflow.db');
const canvas = db.prepare(\"SELECT id, flow_data FROM canvases WHERE name LIKE '%Classifier%Pilot%' LIMIT 1\").get();
if (!canvas) { console.log('ERROR: no pilot canvas'); process.exit(1); }
const flow = JSON.parse(canvas.flow_data);
for (const node of flow.nodes) {
  if (node.data?.label === 'Normalizador') {
    node.data.instructions += '\nResponde SOLO JSON puro sin markdown, sin triple backtick, sin texto adicional. Tu respuesta completa debe ser parseable con JSON.parse() directamente.';
    console.log('Normalizador patched');
  }
  if (node.data?.label === 'Respondedor') {
    node.data.instructions = node.data.instructions.replace(
      'Maximo 200 palabras',
      'IMPORTANTE: Procesa CADA email del array individualmente. Genera una respuesta separada para CADA email no-spam. Usa SOLO datos reales del input (nombre del remitente, empresa, asunto) — NUNCA inventes nombres, empresas ni datos que no esten en el input.\nMaximo 200 palabras'
    );
    console.log('Respondedor patched');
  }
}
db.prepare('UPDATE canvases SET flow_data = ?, updated_at = ? WHERE id = ?').run(JSON.stringify(flow), new Date().toISOString(), canvas.id);
console.log('Canvas updated:', canvas.id);
db.close();
"
```
