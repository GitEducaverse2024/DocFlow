**Paso 2: Aplicar parche con append incondicional**

SOLO si las clausulas no estan presentes (para idempotencia), anadir con +=:

```bash
cd /home/deskmath/docflow/app && node -e "
const Database = require('better-sqlite3');
const db = new Database('/home/deskmath/docflow-data/docflow.db');
const canvas = db.prepare(\"SELECT id, flow_data FROM canvases WHERE id LIKE 'af469bf2%' LIMIT 1\").get();
if (!canvas) { console.log('ERROR: canvas not found'); process.exit(1); }
const flow = JSON.parse(canvas.flow_data);
let patched = 0;
for (const node of flow.nodes) {
  if (node.data?.label === 'Normalizador' && !node.data.instructions.includes('SOLO JSON puro')) {
    node.data.instructions += '\nResponde SOLO JSON puro sin markdown, sin triple backtick, sin texto adicional. Tu respuesta completa debe ser parseable con JSON.parse() directamente.';
    patched++;
    console.log('Normalizador: PATCHED');
  }
  if (node.data?.label === 'Respondedor' && !node.data.instructions.includes('CADA email')) {
    node.data.instructions += '\nIMPORTANTE: Procesa CADA email del array individualmente. Genera una respuesta separada para CADA email no-spam. Usa SOLO datos reales del input (nombre del remitente, empresa, asunto) — NUNCA inventes nombres, empresas ni datos que no esten en el input.';
    patched++;
    console.log('Respondedor: PATCHED');
  }
}
if (patched > 0) {
  db.prepare('UPDATE canvases SET flow_data = ?, updated_at = ? WHERE id = ?').run(JSON.stringify(flow), new Date().toISOString(), canvas.id);
  console.log('Canvas updated. Nodes patched:', patched);
} else {
  console.log('No patches needed — clausulas ya presentes');
}
db.close();
"
```
