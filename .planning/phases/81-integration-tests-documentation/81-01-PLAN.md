export interface PendingDelete {
  token: string;
  toolName: string;
  resourceType: string;
  resourceId: string;
  resourceLabel: string;
  deleteEndpoint: string;
  deleteModule?: HoldedModule;
  deleteBody?: unknown;
  createdAt: number;
  status: 'pending' | 'confirmed' | 'cancelled' | 'expired' | 'error';

export function createPendingDelete(

**Test 1: List employees**
```
it('should list employees with id and holdedUserId fields', async () => {
  const employees = await client.get<Array<{ id: string; holdedUserId?: string; name?: string }>>('/employees', undefined, 'team');
  expect(Array.isArray(employees)).toBe(true);
  expect(employees.length).toBeGreaterThan(0);
  // Phase 77 fix: holdedUserId must exist on employees
  const first = employees[0];
  expect(first).toHaveProperty('id');
  expect(typeof first.id).toBe('string');
  // holdedUserId may be empty string for some employees, but the field should exist
  expect('holdedUserId' in first).toBe(true);
});
```
