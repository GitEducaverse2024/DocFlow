### Task 4: Handle scheduler 'waiting' in pollStatus (canvas-editor.tsx)
**File**: `app/src/components/canvas/canvas-editor.tsx`

In the `pollStatus` function, the existing checkpoint waiting detection block (around line 303-318) currently finds any node in 'waiting' status and opens the checkpoint dialog. Scheduler nodes also enter 'waiting' status but should NOT trigger the checkpoint dialog. Add a type check:

In the block that detects `data.status === 'waiting'` (line 303), before opening the checkpoint dialog, check the node type. The `nodes` state variable holds the current nodes:

```typescript
// Inside the if (data.status === 'waiting') block:
if (waitingEntry) {
  const [waitingNodeId] = waitingEntry;
  // Find the node type to distinguish checkpoint vs scheduler
  const waitingNode = prev.find((n: Node) => n.id === waitingNodeId);
  if (waitingNode?.type === 'checkpoint') {
    const predecessorOutput = waitingEntry[1].output ?? null;
    setCheckpointDialog({ nodeId: waitingNodeId, predecessorOutput });
    // Do NOT schedule next poll — polling resumes after approve/reject
  } else {
    // Scheduler or other waiting node — keep polling, signal comes externally
    schedulePoll(rid);
  }
}
```

This requires accessing `nodes` state inside `pollStatus`. Since `pollStatus` already uses `setNodes`, wrap the waiting detection inside a `setNodes` callback to access current nodes:

```typescript
if (data.status === 'waiting') {
  const waitingEntry = Object.entries(nodeStates).find(([, ns]) => ns.status === 'waiting');
  if (waitingEntry) {
    const [waitingNodeId] = waitingEntry;
    // Check node type — need current nodes
    setNodes(prev => {
      const waitingNode = prev.find(n => n.id === waitingNodeId);
      if (waitingNode?.type === 'checkpoint') {
        const predecessorOutput = waitingEntry[1].output ?? null;
        setCheckpointDialog({ nodeId: waitingNodeId, predecessorOutput });
      } else {
        // Scheduler listen mode — continue polling, signal resolves externally
        schedulePoll(rid);
      }
      return prev; // No state change, just reading
    });
  } else {
    schedulePoll(rid);
  }
  return;
}
```
