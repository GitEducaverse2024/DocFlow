  export const registerTimeSchema = z.object({
    projectId: z.string().min(1).describe('Project ID'),
    hours: z.number().positive().describe('Hours to register (converted to seconds internally)'),
    costHour: z.number().min(0).optional().describe('Cost per hour rate (default: 0)'),
    desc: z.string().optional().describe('Description of the work'),
    employeeId: z.string().optional().describe('Employee ID (holdedUserId will be resolved)'),
    taskId: z.string().optional().describe('Associated project task ID'),
  });

  export const batchRegisterTimesSchema = z.object({
    projectId: z.string().min(1).describe('Project ID'),
    entries: z.array(z.object({
      hours: z.number().positive().describe('Hours to register'),
      desc: z.string().optional().describe('Description'),
      taskId: z.string().optional().describe('Task ID'),
      costHour: z.number().min(0).optional().describe('Cost per hour (overrides top-level)'),
    })).min(1).describe('Array of time entries to register'),
    employeeId: z.string().optional().describe('Employee ID (resolved once for all entries)'),
    costHour: z.number().min(0).optional().describe('Default cost per hour for all entries (default: 0)'),

  **2. Create `~/holded-mcp/src/tools/project-times.ts`:**

  ```typescript
  import { HoldedClient } from '../holded-client.js';
  import { registerTimeSchema, batchRegisterTimesSchema, withValidation } from '../validation.js';

  interface HoldedEmployee {
    id: string;
    holdedUserId?: string;
    name?: string;
    [key: string]: unknown;
  }

  async function resolveHoldedUserId(client: HoldedClient, employeeId: string): Promise<string | undefined> {
    const employee = await client.get<HoldedEmployee>(`/employees/${employeeId}`, undefined, 'team');
    return employee?.holdedUserId || undefined;
  }

  export function getProjectTimeTools(client: HoldedClient) {
    return {
      holded_register_time: {
        description: 'Register time on a Holded project. Accepts hours (converted to seconds internally). Resolves employee holdedUserId automatically. costHour defaults to 0.',
        inputSchema: { ... },  // Mirror registerTimeSchema as JSON Schema
        destructiveHint: true,
        handler: withValidation(registerTimeSchema, async (args) => {
          const duration = Math.round(args.hours * 3600);

          let userId: string | undefined;
          if (args.employeeId) {
            userId = await resolveHoldedUserId(client, args.employeeId);
            // If holdedUserId is empty, omit userId (PFIX-04)
          }

          const body: Record<string, unknown> = {
            duration,
            costHour: args.costHour ?? 0,
          };
          if (args.desc) body.desc = args.desc;
          if (userId) body.userId = userId;
          if (args.taskId) body.taskId = args.taskId;

          return client.post(`/projects/${args.projectId}/times`, body, 'projects');
        }),
      },

      holded_batch_register_times: {
        description: 'Register multiple time entries on a Holded project in one call. Resolves employee holdedUserId once. Each entry: hours (converted to seconds), optional desc/taskId/costHour override.',
        inputSchema: { ... },  // Mirror batchRegisterTimesSchema as JSON Schema
        destructiveHint: true,
        handler: withValidation(batchRegisterTimesSchema, async (args) => {
          // Resolve holdedUserId ONCE before loop (PFIX-06)
          let userId: string | undefined;
          if (args.employeeId) {
            userId = await resolveHoldedUserId(client, args.employeeId);
          }

          const defaultCostHour = args.costHour ?? 0;
          const results = [];

          for (const entry of args.entries) {
            const duration = Math.round(entry.hours * 3600);
            const body: Record<string, unknown> = {
              duration,
              costHour: entry.costHour ?? defaultCostHour,
            };
            if (entry.desc) body.desc = entry.desc;
            if (userId) body.userId = userId;
            if (entry.taskId) body.taskId = entry.taskId;

            const result = await client.post(`/projects/${args.projectId}/times`, body, 'projects');
            results.push(result);
          }

          return { registered: results.length, results };
        }),
      },
    };
  }
  ```

  The inputSchema for each tool must be JSON Schema format (not Zod) matching the Zod schemas, following the same pattern as `time-tracking.ts`.
