        </div>

        {listeningCatflows.length === 0 ? (

                        ? triggerTargets.filter(t => t.id !== cf.id)

                      update({ trigger_targets: newTargets });
                    }}
                  />
                  <span className="text-xs text-zinc-300">{cf.name}</span>
                </label>
              );
            })}
          </div>
        )}
      </div>
    </div>
  );
