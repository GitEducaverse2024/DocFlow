API contracts created by Plan 01:

GET /api/catbot/knowledge/entries?validated=true|false
Response: LearnedRow[] (id, knowledge_path, category, content, learned_from, confidence, validated, access_count, created_at, updated_at)

PATCH /api/catbot/knowledge/entries
Body: { id: string, action: 'validate' | 'reject' }

GET /api/catbot/knowledge/gaps?resolved=true|false and area=string
Response: KnowledgeGapRow[] (id, knowledge_path, query, context, reported_at, resolved, resolved_at)

PATCH /api/catbot/knowledge/gaps
Body: { id: string, action: 'resolve' }

GET /api/catbot/knowledge/stats
Response: { total: number, staging: number, validated: number, avgAccessCount: number }

GET /api/catbot/knowledge/tree
Response: { id, name, path, updated_at, counts: { tools, concepts, howto, dont, common_errors, endpoints, sources }, completeness: number(0-1) }[]
