    /**
     * Format an ISO or SQL-style datetime as a Spanish relative time string.
     * Examples:
     *   'hoy', 'ayer', 'hace 5 días', 'hace 2 meses', or a localized date for >= 1 year.
     * Tolerant of both ISO 8601 (`2026-04-20T10:00:00Z`) and SQL format (`2026-04-20 10:00:00`).
     */
    export function formatRelativeTime(updated: string): string {
      const safe = updated.includes('T') ? updated : updated.replace(' ', 'T') + 'Z';
      const parsed = new Date(safe).getTime();
      if (Number.isNaN(parsed)) return updated;
      const diff = Date.now() - parsed;
      const days = Math.floor(diff / 86_400_000);
      if (days <= 0) return 'hoy';
      if (days === 1) return 'ayer';
      if (days < 30) return `hace ${days} días`;
      if (days < 365) return `hace ${Math.floor(days / 30)} meses`;
      return new Date(safe).toLocaleDateString('es-ES');
    }
    ```

    /**
     * Aggregate KbIndex.header.last_changes[] by day (YYYY-MM-DD).
     * Accepts ISO 8601 and SQL datetime formats. Invalid dates are skipped.
     * Returned array is sorted ascending by day.
     */
    export interface KbChange {
      id: string;
      updated: string;
    }

    export interface TimelineDay {
      day: string;
      count: number;
    }

    export function aggregateChangesByDay(changes: readonly KbChange[]): TimelineDay[] {
      const byDay = new Map<string, number>();
      for (const c of changes) {
        const safe = c.updated.includes('T') ? c.updated : c.updated.replace(' ', 'T') + 'Z';
        const day = safe.slice(0, 10);
        if (!/^\d{4}-\d{2}-\d{2}$/.test(day)) continue;
        byDay.set(day, (byDay.get(day) ?? 0) + 1);
      }
      return [...byDay.entries()]
        .sort(([a], [b]) => a.localeCompare(b))
        .map(([day, count]) => ({ day, count }));
    }
    ```

    ### Step 2.7 — Write `app/src/lib/kb-filters.ts` (GREEN)

    ```typescript
    /**
     * Pure TypeScript filter helpers for the Phase 154 KB dashboard.
     * Consumed from client components (`KnowledgeTable`, `KnowledgeFilters`) in Plan 02.
     * Unit-testable in vitest env=node — no React imports.
     */
    import type { KbIndexEntry } from '@/lib/services/kb-index-cache';

    export interface KbFilterState {
      type?: string;
      subtype?: string;
      status?: string;
      audience?: string;
      tags: string[];
      search: string;
    }

    export function applyKbFilters(entries: readonly KbIndexEntry[], f: KbFilterState): KbIndexEntry[] {
      const q = (f.search ?? '').trim().toLowerCase();
      const wantTags = f.tags ?? [];
      return entries.filter((e) => {
        if (f.type && e.type !== f.type) return false;
        if (f.subtype && e.subtype !== f.subtype) return false;
        if (f.status && e.status !== f.status) return false;
        if (f.audience && !e.audience.includes(f.audience)) return false;
        if (wantTags.length && !wantTags.every((t) => e.tags.includes(t))) return false;
        if (q) {
          const hay = `${e.title} ${e.summary}`.toLowerCase();
          if (!hay.includes(q)) return false;
        }
        return true;
      });
    }

    export function collectDistinctTypes(entries: readonly KbIndexEntry[]): string[] {
      return [...new Set(entries.map((e) => e.type))].sort();
    }

    export function collectDistinctSubtypes(
      entries: readonly KbIndexEntry[],
      type?: string,
    ): string[] {
      const scoped = type ? entries.filter((e) => e.type === type) : entries;
      return [...new Set(scoped.map((e) => e.subtype).filter((s): s is string => !!s))].sort();
    }

    export function collectDistinctTags(entries: readonly KbIndexEntry[], max = 50): string[] {
      const count = new Map<string, number>();
      for (const e of entries) {
        for (const t of e.tags) count.set(t, (count.get(t) ?? 0) + 1);
      }
      return [...count.entries()]
        .sort((a, b) => b[1] - a[1] || a[0].localeCompare(b[0]))
        .slice(0, max)
        .map(([t]) => t);
    }
    ```
