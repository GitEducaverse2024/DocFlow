# Plan 60-03: Schedule Toggle UI + Cycle Progress Display

## Goal
Add schedule activation/deactivation controls and cycle progress display to the task detail page.

## Context
- Plans 60-01/02 deliver the scheduler backend, executeTaskWithCycles(), and next_run_at management
- Task detail page (`app/src/app/tasks/[id]/page.tsx`) currently shows task name, status, steps, result — no v15.0 fields

- Need a PATCH endpoint specifically for toggling schedule is_active

## Tasks

### Task 1: Create schedule toggle API route

**File**: `app/src/app/api/tasks/[id]/schedule/route.ts` (NEW)

PATCH endpoint:
1. Read task, verify execution_mode='scheduled'
2. Accept `{ is_active: boolean }`
3. **Activate**: parse schedule_config → `calculateNextExecution(config)` → update task_schedules (is_active=1, next_run_at) + tasks.next_run_at
4. **Deactivate**: update task_schedules (is_active=0, next_run_at=NULL) + tasks.next_run_at=NULL
5. Return `{ is_active, next_run_at }`

```typescript
import { NextResponse } from 'next/server';
import db from '@/lib/db';
import { calculateNextExecution } from '@/lib/schedule-utils';
import { logger } from '@/lib/logger';

export const dynamic = 'force-dynamic';

export async function PATCH(request: Request, { params }: { params: { id: string } }) {
  try {
    const task = db.prepare('SELECT * FROM tasks WHERE id = ?').get(params.id) as Record<string, unknown> | undefined;
    if (!task) return NextResponse.json({ error: 'Tarea no encontrada' }, { status: 404 });
    if (task.execution_mode !== 'scheduled') {
      return NextResponse.json({ error: 'La tarea no es de tipo programado' }, { status: 400 });
    }

    const { is_active } = await request.json();
    const now = new Date().toISOString();

    if (is_active) {
      const config = task.schedule_config ? JSON.parse(task.schedule_config as string) : null;
      if (!config) return NextResponse.json({ error: 'Sin configuracion de horario' }, { status: 400 });

      const nextRun = calculateNextExecution(config);
      db.prepare('UPDATE task_schedules SET is_active = 1, next_run_at = ?, updated_at = ? WHERE task_id = ?')
        .run(nextRun?.toISOString() || null, now, params.id);
      db.prepare('UPDATE tasks SET next_run_at = ?, updated_at = ? WHERE id = ?')
        .run(nextRun?.toISOString() || null, now, params.id);

      return NextResponse.json({ is_active: true, next_run_at: nextRun?.toISOString() || null });
    } else {
      db.prepare('UPDATE task_schedules SET is_active = 0, next_run_at = NULL, updated_at = ? WHERE task_id = ?')
        .run(now, params.id);
      db.prepare('UPDATE tasks SET next_run_at = NULL, updated_at = ? WHERE id = ?')
        .run(now, params.id);

      return NextResponse.json({ is_active: false, next_run_at: null });
    }
  } catch (error) {
    logger.error('tasks', 'Error toggling schedule', { error: (error as Error).message });
    return NextResponse.json({ error: 'Error interno' }, { status: 500 });
  }
}
```

```typescript
const [scheduleActive, setScheduleActive] = useState(false);

useEffect(() => {
  if (task?.schedule_config) {
    try {
      const config = JSON.parse(task.schedule_config);
      setScheduleActive(config.is_active ?? false);
    } catch { /* ignore parse errors */ }
  }
}, [task]);

const handleScheduleToggle = async (active: boolean) => {
  const res = await fetch(`/api/tasks/${task!.id}/schedule`, {
    method: 'PATCH',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ is_active: active }),
  });
  if (res.ok) {
    setScheduleActive(active);
    fetchTask(); // refresh to update next_run_at display
    toast.success(active ? t('taskDetail.scheduleActivated') : t('taskDetail.scheduleDeactivated'));
  } else {
    toast.error(t('taskDetail.scheduleToggleError'));
  }
};
```
