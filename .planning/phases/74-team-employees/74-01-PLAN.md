```typescript
import * as fs from 'node:fs';
import * as path from 'node:path';
import * as os from 'node:os';
import { HoldedClient } from '../holded-client.js';
import {
  employeeIdSchema,
  listEmployeesSchema,
  searchEmployeeSchema,
  setMyEmployeeIdSchema,
  getMyEmployeeIdSchema,
  withValidation,
} from '../validation.js';

interface HoldedEmployee {
  id: string;
  name: string;
  lastName?: string;
  email?: string;
  [key: string]: unknown;
}

interface MpcConfig {
  myEmployeeId?: string;
}

const CONFIG_DIR = path.join(os.homedir(), '.config', 'holded-mcp');
const CONFIG_FILE = path.join(CONFIG_DIR, 'config.json');

function readConfig(): MpcConfig {
  try {
    return JSON.parse(fs.readFileSync(CONFIG_FILE, 'utf-8'));
  } catch {
    return {};
  }
}

function writeConfig(config: MpcConfig): void {
  fs.mkdirSync(CONFIG_DIR, { recursive: true });
  fs.writeFileSync(CONFIG_FILE, JSON.stringify(config, null, 2));
}

export function getEmployeeTools(client: HoldedClient) {
  return {
    holded_list_employees: {
      description:
        'List all employees in Holded with client-side pagination. Returns employee name, lastName, email, and other fields.',
      inputSchema: {
        type: 'object' as const,
        properties: {
          page: {
            type: 'number',
            description: 'Page number (default: 1)',
          },
          limit: {
            type: 'number',
            description: 'Items per page (default: 50, max: 500)',
          },
        },
        required: [],
      },
      readOnlyHint: true,
      handler: withValidation(listEmployeesSchema, async (args) => {
        const allEmployees = await client.get<HoldedEmployee[]>('/employees', undefined, 'team');

        const page = args.page ?? 1;
        const limit = Math.min(args.limit ?? 50, 500);
        const startIndex = (page - 1) * limit;
        const endIndex = startIndex + limit;
        const items = allEmployees.slice(startIndex, endIndex);

        return {
          items,
          page,
          pageSize: items.length,
          totalItems: allEmployees.length,
          totalPages: Math.ceil(allEmployees.length / limit),
          hasMore: endIndex < allEmployees.length,
        };
      }),
    },

    holded_get_employee: {
      description:
        'Get a specific Holded employee by ID, including full detail (contact info, address, teams, etc.).',
      inputSchema: {
        type: 'object' as const,
        properties: {
          employeeId: {
            type: 'string',
            description: 'Employee ID',
          },
        },
        required: ['employeeId'],
      },
      readOnlyHint: true,
      handler: withValidation(employeeIdSchema, async (args) => {
        return client.get(`/employees/${args.employeeId}`, undefined, 'team');
      }),
    },

    holded_search_employee: {
      description:
        'Search employees by name, lastName, or email (case-insensitive, client-side filter). Returns matching employees with pagination.',
      inputSchema: {
        type: 'object' as const,
        properties: {
          query: {
            type: 'string',
            description: 'Search query (matches name, lastName, or email)',
          },
          page: {
            type: 'number',
            description: 'Page number (default: 1)',
          },
          limit: {
            type: 'number',
            description: 'Items per page (default: 50, max: 500)',
          },
        },
        required: ['query'],
      },
      readOnlyHint: true,
      handler: withValidation(searchEmployeeSchema, async (args) => {
        const allEmployees = await client.get<HoldedEmployee[]>('/employees', undefined, 'team');
        const q = args.query.toLowerCase();

        const filtered = allEmployees.filter((emp) => {
          const nameMatch = emp.name?.toLowerCase().includes(q);
          const lastNameMatch = emp.lastName?.toLowerCase().includes(q);
          const emailMatch = emp.email?.toLowerCase().includes(q);
          return nameMatch || lastNameMatch || emailMatch;
        });

        const page = args.page ?? 1;
        const limit = Math.min(args.limit ?? 50, 500);
        const startIndex = (page - 1) * limit;
        const endIndex = startIndex + limit;
        const items = filtered.slice(startIndex, endIndex);

        return {
          items,
          page,
          pageSize: items.length,
          totalItems: filtered.length,
          totalPages: Math.ceil(filtered.length / limit),
          hasMore: endIndex < filtered.length,
        };
      }),
    },

    holded_set_my_employee_id: {
      description:
        'Save your Holded employee ID locally for use with clock-in/out and timesheet tools. Stores in ~/.config/holded-mcp/config.json.',
      inputSchema: {
        type: 'object' as const,
        properties: {
          employeeId: {
            type: 'string',
            description: 'Your Holded employee ID (use holded_search_employee to find it)',
          },
        },
        required: ['employeeId'],
      },
      destructiveHint: true,
      handler: withValidation(setMyEmployeeIdSchema, async (args) => {
        const config = readConfig();
        config.myEmployeeId = args.employeeId;
        writeConfig(config);
        return {
          status: 'ok',
          message: `Employee ID ${args.employeeId} saved to ${CONFIG_FILE}`,
          myEmployeeId: args.employeeId,
        };
      }),
    },

    holded_get_my_employee_id: {
      description:
        'Get your saved Holded employee ID from local config. Use holded_set_my_employee_id first to save it.',
      inputSchema: {
        type: 'object' as const,
        properties: {},
        required: [],
      },
      readOnlyHint: true,
      handler: withValidation(getMyEmployeeIdSchema, async () => {
        const config = readConfig();
        if (!config.myEmployeeId) {
          return {
            status: 'not_set',
            message:
              'No employee ID configured. Use holded_set_my_employee_id to save your ID.',
            myEmployeeId: null,
          };
        }
        return {
          status: 'ok',
          myEmployeeId: config.myEmployeeId,
        };
      }),
    },
  };
}
```

```typescript
import { describe, it, expect, beforeEach, vi, afterEach } from 'vitest';
import * as fs from 'node:fs';
import * as path from 'node:path';
import * as os from 'node:os';
import { createMockClient } from './mock-client.js';
import { getEmployeeTools } from '../tools/employees.js';

const mockEmployees = [
  {
    id: 'emp-1',
    name: 'Juan',
    lastName: 'Garcia',
    email: 'juan@example.com',
  },
  {
    id: 'emp-2',
    name: 'Maria',
    lastName: 'Lopez',
    email: 'maria@example.com',
  },
  {
    id: 'emp-3',
    name: 'Carlos',
    lastName: 'Martinez',
    email: 'carlos@example.com',
  },
];

// Mock config file path for tests
const TEST_CONFIG_DIR = path.join(os.tmpdir(), 'holded-mcp-test-' + process.pid);
const TEST_CONFIG_FILE = path.join(TEST_CONFIG_DIR, 'config.json');

describe('Employee Tools', () => {
  let client: ReturnType<typeof createMockClient>;
  let tools: ReturnType<typeof getEmployeeTools>;

  beforeEach(() => {
    vi.clearAllMocks();
    client = createMockClient();
    tools = getEmployeeTools(client);
  });

  describe('holded_list_employees', () => {
    it('should list all employees using team module', async () => {
      client.get = vi.fn().mockResolvedValue(mockEmployees);

      const result = (await tools.holded_list_employees.handler({})) as any;

      expect(client.get).toHaveBeenCalledWith('/employees', undefined, 'team');
      expect(result.items).toHaveLength(3);
      expect(result.totalItems).toBe(3);
    });

    it('should paginate results', async () => {
      client.get = vi.fn().mockResolvedValue(mockEmployees);

      const result = (await tools.holded_list_employees.handler({ page: 1, limit: 2 })) as any;

      expect(result.items).toHaveLength(2);
      expect(result.hasMore).toBe(true);
      expect(result.totalPages).toBe(2);
    });

    it('should return page 2', async () => {
      client.get = vi.fn().mockResolvedValue(mockEmployees);

      const result = (await tools.holded_list_employees.handler({ page: 2, limit: 2 })) as any;

      expect(result.items).toHaveLength(1);
      expect(result.items[0].id).toBe('emp-3');
      expect(result.hasMore).toBe(false);
    });

    it('should return empty array when no employees', async () => {
      client.get = vi.fn().mockResolvedValue([]);

      const result = (await tools.holded_list_employees.handler({})) as any;

      expect(result.items).toHaveLength(0);
      expect(result.totalItems).toBe(0);
    });
  });

  describe('holded_get_employee', () => {
    it('should get an employee by ID', async () => {
      const mockEmployee = { id: 'emp-1', name: 'Juan', lastName: 'Garcia', email: 'juan@example.com' };
      client.get = vi.fn().mockResolvedValue(mockEmployee);

      const result = await tools.holded_get_employee.handler({ employeeId: 'emp-1' });

      expect(client.get).toHaveBeenCalledWith('/employees/emp-1', undefined, 'team');
      expect(result).toEqual(mockEmployee);
    });

    it('should reject empty employeeId', async () => {
      await expect(tools.holded_get_employee.handler({ employeeId: '' })).rejects.toThrow();
    });
  });

  describe('holded_search_employee', () => {
    it('should search by name (case-insensitive)', async () => {
      client.get = vi.fn().mockResolvedValue(mockEmployees);

      const result = (await tools.holded_search_employee.handler({ query: 'juan' })) as any;

      expect(client.get).toHaveBeenCalledWith('/employees', undefined, 'team');
      expect(result.items).toHaveLength(1);
      expect(result.items[0].id).toBe('emp-1');
    });

    it('should search by lastName', async () => {
      client.get = vi.fn().mockResolvedValue(mockEmployees);

      const result = (await tools.holded_search_employee.handler({ query: 'martinez' })) as any;

      expect(result.items).toHaveLength(1);
      expect(result.items[0].id).toBe('emp-3');
    });

    it('should search by email', async () => {
      client.get = vi.fn().mockResolvedValue(mockEmployees);

      const result = (await tools.holded_search_employee.handler({ query: 'maria@' })) as any;

      expect(result.items).toHaveLength(1);
      expect(result.items[0].id).toBe('emp-2');
    });

    it('should return empty when no match', async () => {
      client.get = vi.fn().mockResolvedValue(mockEmployees);

      const result = (await tools.holded_search_employee.handler({ query: 'nonexistent' })) as any;

      expect(result.items).toHaveLength(0);
      expect(result.totalItems).toBe(0);
    });

    it('should paginate search results', async () => {
      client.get = vi.fn().mockResolvedValue(mockEmployees);

      // 'a' matches Juan (name), Maria (name+lastName), Carlos (name) = 3 results
      const result = (await tools.holded_search_employee.handler({ query: 'a', limit: 2 })) as any;

      expect(result.items).toHaveLength(2);
      expect(result.hasMore).toBe(true);
      expect(result.totalItems).toBe(3);
    });

    it('should reject empty query', async () => {
      await expect(tools.holded_search_employee.handler({ query: '' })).rejects.toThrow();
    });
  });

  describe('holded_set_my_employee_id', () => {
    it('should save employeeId to config file', async () => {
      const result = (await tools.holded_set_my_employee_id.handler({ employeeId: 'emp-123' })) as any;

      expect(result.status).toBe('ok');
      expect(result.myEmployeeId).toBe('emp-123');
    });

    it('should reject empty employeeId', async () => {
      await expect(tools.holded_set_my_employee_id.handler({ employeeId: '' })).rejects.toThrow();
    });
  });

  describe('holded_get_my_employee_id', () => {
    it('should return not_set when no config exists', async () => {
      // Use a fresh tools instance that reads from the real (empty/missing) config
      // The handler reads from ~/.config/holded-mcp/config.json
      // In test environment, this may or may not exist
      const result = (await tools.holded_get_my_employee_id.handler({})) as any;

      // Result is either 'ok' (if config exists from previous test) or 'not_set'
      expect(['ok', 'not_set']).toContain(result.status);
    });

    it('should return employeeId after set', async () => {
      // Set first
      await tools.holded_set_my_employee_id.handler({ employeeId: 'emp-456' });

      // Then get
      const result = (await tools.holded_get_my_employee_id.handler({})) as any;

      expect(result.status).toBe('ok');
      expect(result.myEmployeeId).toBe('emp-456');
    });
  });
});
```
