API details from RESEARCH.md:
- GET /employees/times — list ALL employee time records (paginated 500/page, flat array)
- GET /employees/{employeeId}/times — list specific employee's time records (paginated 500/page)
- POST /employees/{employeeId}/times — create time record, body: {startTmp, endTmp} (both required, strings)
- POST /employees/{employeeId}/times/clockin — body: {location?} (string), returns {status, info, id} HTTP 201
- POST /employees/{employeeId}/times/clockout — body: {latitude?, longitude?} (strings), returns {status, info, id} HTTP 201
- POST /employees/{employeeId}/times/pause — body: {latitude?, longitude?} (strings), returns {status, info, id} HTTP 201
- POST /employees/{employeeId}/times/unpause — body: {latitude?, longitude?} (strings), returns {status, info, id} HTTP 201

QUIRK: startTmp/endTmp are strings per API spec. Accept as Unix timestamp (seconds) as number from user input, convert to string via `String(value)` before sending to API.
QUIRK: Clock actions (clockin/clockout/pause/unpause) return HTTP 201 with {status, info, id}.
QUIRK: "timesheet" naming for employee/team records (this plan) vs "time_entry" for project records (phase 73). Different APIs, different modules.

import { HoldedClient } from '../holded-client.js';
import {
  listTimesheetsSchema,
  createTimesheetSchema,
  clockInSchema,
  clockOutSchema,
  clockPauseSchema,
  clockUnpauseSchema,
  weeklyTimesheetSummarySchema,
  withValidation,
} from '../validation.js';

interface TimesheetRecord {
  id: string;
  startTmp?: string;
  endTmp?: string;
  [key: string]: unknown;
}

export function getEmployeeTimesheetTools(client: HoldedClient) {
  return {
    holded_list_timesheets: {
      description:
        'List employee timesheet (work schedule) records from Holded. If employeeId is provided, returns that employee\'s records. If omitted, returns ALL employee time records. These are employee clock/schedule records (team module), NOT project time tracking entries. Supports client-side pagination.',
      inputSchema: {
        type: 'object' as const,
        properties: {
          employeeId: {
            type: 'string',
            description: 'Employee ID. If omitted, lists ALL employee time records.',
          },
          page: {
            type: 'number',
            description: 'Page number (default: 1)',
          },
          limit: {
            type: 'number',
            description: 'Items per page (default: 50, max: 500)',
          },
        },
        required: [],
      },
      readOnlyHint: true,
      handler: withValidation(listTimesheetsSchema, async (args) => {
        const endpoint = args.employeeId
          ? `/employees/${args.employeeId}/times`
          : '/employees/times';
        const all = await client.get<TimesheetRecord[]>(endpoint, undefined, 'team');
        const page = args.page ?? 1;
        const limit = Math.min(args.limit ?? 50, 500);
        const start = (page - 1) * limit;
        const items = all.slice(start, start + limit);
        return {
          items,
          page,
          pageSize: items.length,
          totalItems: all.length,
          totalPages: Math.ceil(all.length / limit),
          hasMore: start + limit < all.length,
        };
      }),
    },

    holded_create_timesheet: {
      description:
        'Create a manual timesheet entry for an employee in Holded. Used for retroactive/corrective time entries. startTmp and endTmp are Unix timestamps in seconds (e.g., 1711900800 for 2024-03-31 12:00 UTC). These are employee schedule records (team module), NOT project time entries.',
      inputSchema: {
        type: 'object' as const,
        properties: {
          employeeId: {
            type: 'string',
            description: 'Employee ID (required)',
          },
          startTmp: {
            type: 'number',
            description: 'Start time as Unix timestamp in seconds (required). Example: 1711900800',
          },
          endTmp: {
            type: 'number',
            description: 'End time as Unix timestamp in seconds (required). Example: 1711929600',
          },
        },
        required: ['employeeId', 'startTmp', 'endTmp'],
      },
      destructiveHint: true,
      handler: withValidation(createTimesheetSchema, async (args) => {
        return client.post(
          `/employees/${args.employeeId}/times`,
          { startTmp: String(args.startTmp), endTmp: String(args.endTmp) },
          'team'
        );
      }),
    },

    holded_clock_in: {
      description:
        'Clock in (start work tracking) for an employee in Holded. Use holded_get_my_employee_id first to get your own ID. Returns {status, info, id}.',
      inputSchema: {
        type: 'object' as const,
        properties: {
          employeeId: {
            type: 'string',
            description: 'Employee ID (required)',
          },
          location: {
            type: 'string',
            description: 'Optional location description',
          },
        },
        required: ['employeeId'],
      },
      destructiveHint: true,
      handler: withValidation(clockInSchema, async (args) => {
        const body: Record<string, string> = {};
        if (args.location) body.location = args.location;
        return client.post(`/employees/${args.employeeId}/times/clockin`, body, 'team');
      }),
    },

    holded_clock_out: {
      description:
        'Clock out (stop work tracking) for an employee in Holded. Optionally provide GPS coordinates. Returns {status, info, id}.',
      inputSchema: {
        type: 'object' as const,
        properties: {
          employeeId: {
            type: 'string',
            description: 'Employee ID (required)',
          },
          latitude: {
            type: 'string',
            description: 'Optional latitude coordinate',
          },
          longitude: {
            type: 'string',
            description: 'Optional longitude coordinate',
          },
        },
        required: ['employeeId'],
      },
      destructiveHint: true,
      handler: withValidation(clockOutSchema, async (args) => {
        const body: Record<string, string> = {};
        if (args.latitude) body.latitude = args.latitude;
        if (args.longitude) body.longitude = args.longitude;
        return client.post(`/employees/${args.employeeId}/times/clockout`, body, 'team');
      }),
    },

    holded_clock_pause: {
      description:
        'Pause work tracking for an employee in Holded (e.g., lunch break). Optionally provide GPS coordinates. Returns {status, info, id}.',
      inputSchema: {
        type: 'object' as const,
        properties: {
          employeeId: {
            type: 'string',
            description: 'Employee ID (required)',
          },
          latitude: {
            type: 'string',
            description: 'Optional latitude coordinate',
          },
          longitude: {
            type: 'string',
            description: 'Optional longitude coordinate',
          },
        },
        required: ['employeeId'],
      },
      destructiveHint: true,
      handler: withValidation(clockPauseSchema, async (args) => {
        const body: Record<string, string> = {};
        if (args.latitude) body.latitude = args.latitude;
        if (args.longitude) body.longitude = args.longitude;
        return client.post(`/employees/${args.employeeId}/times/pause`, body, 'team');
      }),
    },

    holded_clock_unpause: {
      description:
        'Resume work tracking for an employee in Holded (after a pause). Optionally provide GPS coordinates. Returns {status, info, id}.',
      inputSchema: {
        type: 'object' as const,
        properties: {
          employeeId: {
            type: 'string',
            description: 'Employee ID (required)',
          },
          latitude: {
            type: 'string',
            description: 'Optional latitude coordinate',
          },
          longitude: {
            type: 'string',
            description: 'Optional longitude coordinate',
          },
        },
        required: ['employeeId'],
      },
      destructiveHint: true,
      handler: withValidation(clockUnpauseSchema, async (args) => {
        const body: Record<string, string> = {};
        if (args.latitude) body.latitude = args.latitude;
        if (args.longitude) body.longitude = args.longitude;
        return client.post(`/employees/${args.employeeId}/times/unpause`, body, 'team');
      }),
    },

    holded_weekly_timesheet_summary: {
      description:
        'Get a weekly timesheet summary for an employee. Fetches all time records, filters to the specified week, and aggregates total hours per day. weekOffset: 0 = current week (Mon-Sun), -1 = last week, etc. Returns {employeeId, weekStart, weekEnd, days: [{date, totalHours, entries}], totalWeekHours}.',
      inputSchema: {
        type: 'object' as const,
        properties: {
          employeeId: {
            type: 'string',
            description: 'Employee ID (required)',
          },
          weekOffset: {
            type: 'number',
            description: 'Week offset from current week. 0 = current (default), -1 = last week, 1 = next week.',
          },
        },
        required: ['employeeId'],
      },
      readOnlyHint: true,
      handler: withValidation(weeklyTimesheetSummarySchema, async (args) => {
        // Calculate week boundaries (Monday 00:00 to Sunday 23:59:59)
        const now = new Date();
        const offset = args.weekOffset ?? 0;
        const currentDay = now.getDay(); // 0=Sun, 1=Mon, ...
        const mondayOffset = currentDay === 0 ? -6 : 1 - currentDay; // days to Monday
        const monday = new Date(now);
        monday.setDate(now.getDate() + mondayOffset + offset * 7);
        monday.setHours(0, 0, 0, 0);
        const sunday = new Date(monday);
        sunday.setDate(monday.getDate() + 6);
        sunday.setHours(23, 59, 59, 999);

        const weekStartTs = Math.floor(monday.getTime() / 1000);
        const weekEndTs = Math.floor(sunday.getTime() / 1000);

        // Fetch all time records for this employee
        const records = await client.get<TimesheetRecord[]>(
          `/employees/${args.employeeId}/times`,
          undefined,
          'team'
        );

        // Filter records within the week window
        const weekRecords = records.filter((r) => {
          const start = r.startTmp ? parseInt(r.startTmp, 10) : 0;
          return start >= weekStartTs && start <= weekEndTs;
        });

        // Aggregate by day (Mon=0, Tue=1, ..., Sun=6)
        const dayNames = ['Monday', 'Tuesday', 'Wednesday', 'Thursday', 'Friday', 'Saturday', 'Sunday'];
        const days = dayNames.map((name, i) => {
          const dayDate = new Date(monday);
          dayDate.setDate(monday.getDate() + i);
          const dayStart = Math.floor(dayDate.getTime() / 1000);
          const dayEnd = dayStart + 86400 - 1; // 23:59:59

          const dayEntries = weekRecords.filter((r) => {
            const start = r.startTmp ? parseInt(r.startTmp, 10) : 0;
            return start >= dayStart && start <= dayEnd;
          });

          let totalSeconds = 0;
          for (const entry of dayEntries) {
            const start = entry.startTmp ? parseInt(entry.startTmp, 10) : 0;
            const end = entry.endTmp ? parseInt(entry.endTmp, 10) : 0;
            if (start > 0 && end > start) {
              totalSeconds += end - start;
            }
          }

          return {
            date: dayDate.toISOString().split('T')[0],
            dayName: name,
            totalHours: Math.round((totalSeconds / 3600) * 100) / 100,
            entries: dayEntries.length,
          };
        });

        const totalWeekHours = Math.round(days.reduce((sum, d) => sum + d.totalHours, 0) * 100) / 100;

        return {
          employeeId: args.employeeId,
          weekStart: monday.toISOString().split('T')[0],
          weekEnd: sunday.toISOString().split('T')[0],
          days,
          totalWeekHours,
          totalEntries: weekRecords.length,
        };
      }),
    },
  };
}
