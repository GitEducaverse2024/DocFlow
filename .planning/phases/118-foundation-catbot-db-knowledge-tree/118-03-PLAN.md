    ```typescript
    // Estado: conversation ID activo
    const [conversationId, setConversationId] = useState<string | null>(null);

    // Cargar mensajes desde DB API (fallback a localStorage)
    async function loadMessagesFromDB(): Promise<Message[]> {
      try {
        const res = await fetch('/api/catbot/conversations');
        if (!res.ok) throw new Error('API error');
        const convs = await res.json();
        if (convs.length > 0) {
          const latest = convs[0]; // mas reciente
          setConversationId(latest.id);
          return JSON.parse(latest.messages);
        }
        return [];
      } catch {
        // Fallback a localStorage si API no disponible
        return loadMessagesFromLocalStorage();
      }
    }

    // Guardar mensajes en DB API (fallback a localStorage)
    async function saveMessagesToDB(messages: Message[]) {
      try {
        const method = conversationId ? 'POST' : 'POST';
        const res = await fetch('/api/catbot/conversations', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({ messages: messages.slice(-MAX_STORED) }),
        });
        if (res.ok) {
          const data = await res.json();
          if (!conversationId) setConversationId(data.id);
        }
      } catch {
        // Fallback: guardar en localStorage
        saveMessagesToLocalStorage(messages);
      }
    }
    ```

    **Migracion localStorage (una sola vez):**
    ```typescript
    async function migrateLocalStorageOnce() {
      const MIGRATED_KEY = 'docatflow_catbot_migrated';
      if (localStorage.getItem(MIGRATED_KEY)) return;

      const stored = localStorage.getItem(STORAGE_KEY);
      if (!stored) {
        localStorage.setItem(MIGRATED_KEY, 'true');
        return;
      }

      try {
        const messages = JSON.parse(stored);
        if (!Array.isArray(messages) || messages.length === 0) return;

        const res = await fetch('/api/catbot/conversations/migrate', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({ messages }),
        });

        if (res.ok) {
          localStorage.removeItem(STORAGE_KEY);
          localStorage.setItem(MIGRATED_KEY, 'true');
        }
      } catch {
        // Migration failed, retry next load (don't set MIGRATED_KEY)
      }
    }
    ```
