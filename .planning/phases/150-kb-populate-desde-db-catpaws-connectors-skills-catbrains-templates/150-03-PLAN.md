    ```javascript
    function bumpVersion(current, level) {
      const [M, m, p] = String(current || '1.0.0').split('.').map(Number);
      if (level === 'major') return `${M + 1}.0.0`;
      if (level === 'minor') return `${M}.${m + 1}.0`;
      if (level === 'patch') return `${M}.${m}.${p + 1}`;
      return current;
    }

    function detectBumpLevel(currentFm, newFm) {
      // Compare the stable (non-volatile) projection
      const curStable = stripVolatile(currentFm);
      const newStable = stripVolatile(newFm);

      // MAJOR triggers
      if (curStable.subtype !== newStable.subtype) return 'major';
      if (curStable.status === 'active' && newStable.status === 'deprecated') return 'major';
      if (curStable.mode !== newStable.mode && curStable.mode !== undefined && newStable.mode !== undefined) return 'major';

      // MINOR triggers
      // Detect system_prompt change by scanning source_of_truth.fields_from_db snapshot OR body;
      // For simplicity here, compare a "structural hash" of the file's content fields.
      // The populateFromDb context passes us the DB row, so we reconstruct:
      // (Task 2 reads the existing file's `sync_snapshot` or body-embedded system_prompt to compare.)
      if (stringify(curStable.related) !== stringify(newStable.related)) return 'minor';
      // connectors_linked / skills_linked shifts → captured by related delta

      // PATCH
      if (JSON.stringify(curStable) !== JSON.stringify(newStable)) return 'patch';

      // No change detected
      return null;
    }

    function stringify(v) { return JSON.stringify(v === undefined ? null : v); }
    ```

    const VOLATILE_UPDATE_KEYS = new Set(['updated_at', 'updated_by', 'version', 'change_log', 'sync_snapshot']);

    function stripVolatile(fm) {

      for (const [k, v] of Object.entries(fm)) {
        if (!VOLATILE_UPDATE_KEYS.has(k)) out[k] = v;
      }
      return out;
    }

    function parseFile(filePath) {
      const raw = fs.readFileSync(filePath, 'utf8');
      // Reuse parseYAML from the copied serializer block
      const match = raw.match(/^---\n([\s\S]*?)\n---\n?([\s\S]*)$/);
      if (!match) return { frontmatter: {}, body: raw };
      return { frontmatter: parseYAML(match[1]), body: match[2] };
    }

    function writeResourceFile(kbRoot, subtype, shortIdSlug, fm, body, dryRun, verbose) {
      const targetDir = path.join(kbRoot, SUBTYPE_SUBDIR[subtype]);
      const filePath = path.join(targetDir, `${shortIdSlug}.md`);

      if (fs.existsSync(filePath)) {
        // Idempotence path
        const { frontmatter: curFm, body: curBody } = parseFile(filePath);
        const bump = detectBumpLevel(curFm, fm);

        if (bump === null && curBody.trimEnd() === body.trimEnd()) {
          if (verbose) console.log(`UNCHANGED ${filePath}`);
          return { path: filePath, action: 'unchanged', subtype, id: shortIdSlug };
        }

        // Real change: bump version, preserve change_log + created_at, refresh updated_at
        const newVersion = bumpVersion(curFm.version || '1.0.0', bump || 'patch');
        const now = new Date().toISOString();
        const prevLog = Array.isArray(curFm.change_log) ? curFm.change_log : [];
        const newEntry = {
          version: newVersion,
          date: now.slice(0, 10),
          author: 'kb-sync-bootstrap',
          change: `Auto-sync ${bump || 'patch'} bump from DB`,

        // Truncate to last 5 per Phase 149 convention
        const changeLog = [...prevLog, newEntry].slice(-5);

        // Preserve created_at from the existing file (never re-set from DB — that's the point)
        const mergedFm = {
          ...fm,
          created_at: curFm.created_at || fm.created_at,
          created_by: curFm.created_by || fm.created_by,
          version: newVersion,
          updated_at: now,
          updated_by: 'kb-sync-bootstrap',
          change_log: changeLog,

        // If reactivating (deprecated → active), clear deprecated_* per RESEARCH Pitfall 8
        if (curFm.status === 'deprecated' && mergedFm.status === 'active') {

          delete mergedFm.deprecated_by;

        return { path: filePath, action: 'update', subtype, id: shortIdSlug, bump };
      }

      if (dryRun) {

      }

      fs.writeFileSync(filePath, renderFile(fm, body), 'utf8');
      if (verbose) console.log(`CREATE ${filePath}`);
      return { path: filePath, action: 'create', subtype, id: shortIdSlug };
    }
