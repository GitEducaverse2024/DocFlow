    ```javascript
    function loadCatPawRelations(db) {
      const byPaw = new Map();
      const pushRel = (pawId, rel) => {
        if (!byPaw.has(pawId)) byPaw.set(pawId, []);
        byPaw.get(pawId).push(rel);
      };
      for (const r of db.prepare(`SELECT paw_id, catbrain_id FROM cat_paw_catbrains WHERE is_active = 1`).all()) {
        pushRel(r.paw_id, { id: r.catbrain_id, subtype: 'catbrain' });

        pushRel(r.paw_id, { id: r.connector_id, subtype: 'connector' });
      }
      for (const r of db.prepare(`SELECT paw_id, skill_id FROM cat_paw_skills WHERE is_active = 1`).all()) {
        pushRel(r.paw_id, { id: r.skill_id, subtype: 'skill' });
      }
      return byPaw;
    }

    function loadCatbrainRelations(db) {
      const byBrain = new Map();
      for (const r of db.prepare(`SELECT catbrain_id, connector_id FROM catbrain_connectors WHERE is_active = 1`).all()) {
        if (!byBrain.has(r.catbrain_id)) byBrain.set(r.catbrain_id, []);
    ```

    **Section 5 — slugify + collision resolver (per RESEARCH §"Architecture Pattern 3"):**
    ```javascript
    function slugify(name) {
      return String(name || 'unnamed')
        .normalize('NFKD').replace(/[\u0300-\u036f]/g, '')  // strip diacritics
        .toLowerCase()
        .replace(/[^a-z0-9]+/g, '-')
        .replace(/^-+|-+$/g, '')
        .slice(0, 50) || 'unnamed';
    }

    // typeMap: Map<row.id, short-id-slug>. Mutates typeMap.
    function resolveShortIdSlug(fullId, slug, typeMap) {
      const claimed = new Set(typeMap.values());
      for (const len of [8, 12, 16, fullId.length]) {
        const candidate = `${fullId.slice(0, len)}-${slug}`;
        if (!claimed.has(candidate)) return candidate;
      }
      // Extreme fallback: append index
      let i = 2;
      while (claimed.has(`${fullId}-${slug}-${i}`)) i++;
      return `${fullId}-${slug}-${i}`;
    }
    ```

    **Section 6 — Pass 1 (buildIdMap):**
    ```javascript
    // Selects all rows for all subtypes once, returns:
    //   { rows: { subtype: row[] }, maps: { subtype: Map<row.id, short-id-slug> } }
    function buildIdMap(db, subtypesFilter) {
      const rows = {};
      const maps = {};
      for (const sub of SUBTYPES) {
        if (subtypesFilter && !subtypesFilter.includes(sub)) {
          rows[sub] = [];
          maps[sub] = new Map();
          continue;
        }
        rows[sub] = db.prepare(SELECTS[sub]).all();
        maps[sub] = new Map();
        for (const row of rows[sub]) {
          if (!row.id || !row.name) continue;  // per CONTEXT "Claude's Discretion": skip + WARN
          const slug = slugify(row.name);
          const shortIdSlug = resolveShortIdSlug(row.id, slug, maps[sub]);
          maps[sub].set(row.id, shortIdSlug);
        }
      }
      return { rows, maps };
    }
    ```

    ```javascript
    const DEPARTMENT_MAP = {
      'negocio': 'business', 'finanzas': 'finance', 'producción': 'production', 'produccion': 'production', 'otro': 'other',
    };
    const CONNECTOR_TYPE_MAP = {
      'mcp_server': 'mcp', 'http_api': 'http', 'gmail': 'gmail', 'drive': 'drive',

      'smtp': 'smtp', 'n8n': 'n8n',
    };

    // Loaded lazily from .docflow-kb/_schema/tag-taxonomy.json for cross_cutting / roles check
    let _taxonomy = null;
    function taxonomy(kbRoot) {
      if (_taxonomy) return _taxonomy;
      const raw = fs.readFileSync(path.join(kbRoot, '_schema', 'tag-taxonomy.json'), 'utf8');
      _taxonomy = JSON.parse(raw);
      return _taxonomy;
    }

    function deriveTags(subtype, row, kbRoot, relations) {
      const tax = taxonomy(kbRoot);
      const out = new Set([subtype === 'email-template' ? 'template' : subtype]);  // floor: entity tag

      if (subtype === 'catpaw') {
        if (row.mode && tax.modes.includes(row.mode)) out.add(row.mode);
        const rawDept = (row.department_tags || '').split(',').map((s) => s.trim()).filter(Boolean);
        for (const d of rawDept) {
          const translated = DEPARTMENT_MAP[d.toLowerCase()];
          if (translated && tax.departments.includes(translated)) out.add(translated);
        }
        // Add connector types from relations
        for (const rel of (relations || [])) {
          if (rel.subtype !== 'connector') continue;
          const connType = rel.connector_type_raw;
          const mapped = connType ? CONNECTOR_TYPE_MAP[connType] : null;
          if (mapped && tax.connectors.includes(mapped)) out.add(mapped);
        }
      } else if (subtype === 'connector') {
        const mapped = CONNECTOR_TYPE_MAP[row.type];
        if (mapped && tax.connectors.includes(mapped)) out.add(mapped);
        /* also inject implicit domain — gmail/smtp → email, holded → crm, drive → storage */
        const TYPE_TO_DOMAIN = { gmail: 'email', smtp: 'email', holded: 'crm', drive: 'storage' };
        const domain = TYPE_TO_DOMAIN[mapped];
        if (domain && tax.domains.includes(domain)) out.add(domain);
      } else if (subtype === 'skill') {
        if (row.category && tax.roles.includes(row.category)) out.add(row.category);
        try {
          const parsed = JSON.parse(row.tags || '[]');
          if (Array.isArray(parsed)) {
            for (const t of parsed) {
              const s = String(t).toLowerCase();
              if (tax.cross_cutting.includes(s)) out.add(s);
            }
          }
        } catch { /* malformed JSON — ignore */ }
      } else if (subtype === 'catbrain') {
        // Heuristic: rag_enabled=1 → chat; has catbrain_connectors → processor; else hybrid
        const hasConnectors = relations && relations.length > 0;
        const mode = row.rag_enabled ? 'chat' : hasConnectors ? 'processor' : 'hybrid';
        if (tax.modes.includes(mode)) out.add(mode);
      } else if (subtype === 'email-template') {
        if (row.category && tax.domains.includes(row.category)) out.add(row.category);
        out.add('email');  /* always */
      } else if (subtype === 'canvas') {
        if (row.mode && tax.modes.includes(row.mode)) out.add(row.mode);
        const rawTags = String(row.tags || '').split(',').map((s) => s.trim()).filter(Boolean);
        for (const t of rawTags) {
          const s = t.toLowerCase();
          // Search all taxonomy buckets
          for (const bucket of Object.values(tax)) {
            if (Array.isArray(bucket) && bucket.includes(s)) { out.add(s); break; }
          }
        }
      }
      return Array.from(out);
    }
    ```

    ```javascript
    // Returns a plain object in insertion order that the inline YAML serializer will render
    // with keys in this exact sequence: id, type, subtype, lang, title, summary, tags,
    // audience, status, created_at, created_by, version, updated_at, updated_by,
    // source_of_truth, ttl, related, change_log.
    function buildFrontmatter(subtype, row, kbRoot, maps, relations) {
      const shortIdSlug = maps[subtype].get(row.id);
      const derivedTags = deriveTags(subtype, row, kbRoot, relations);

      // Compute status
      let status;
      if (subtype === 'canvas' && row.status === 'archived') status = 'deprecated';
      else if (subtype === 'catbrain' && row.status === 'draft') status = 'draft';
      else if (row.is_active === 0 || row.is_active === false) status = 'deprecated';
      else status = 'active';

      // created_at/updated_at: DB ground truth (idempotence-stable)
      const createdAt = row.created_at || row.updated_at || new Date().toISOString();
      const updatedAt = row.updated_at || row.created_at || new Date().toISOString();

      // audience
      const audience = subtype === 'skill' ? ['catbot', 'developer'] : ['catbot', 'architect'];

      // summary truncation
      const rawSummary = String(row.description || `${subtype} sin descripción`);
      const summary = rawSummary.length > 200 ? rawSummary.slice(0, 197) + '...' : rawSummary;

      // source_of_truth
      const fieldsByType = {
        catpaw: ['name','description','mode','model','system_prompt','tone','department_tags','is_active','times_used','temperature','max_tokens','output_format'],
        connector: ['name','description','type','is_active','times_used','test_status'],
        skill: ['name','description','category','tags','instructions','source','version','author','times_used'],
        catbrain: ['name','description','purpose','tech_stack','status','agent_id','rag_enabled','rag_collection'],
        'email-template': ['name','description','category','is_active','times_used'],
        canvas: ['name','description','mode','status','tags','is_template'],
      };
      const sourceOfTruth = [{ db: 'sqlite', table: SUBTYPE_TABLE[subtype], id: row.id, fields_from_db: fieldsByType[subtype] }];

      // related (Pass 2: relations is already resolved via maps)
      const related = [];
      for (const rel of relations || []) {
        const targetMap = maps[rel.subtype];
        if (!targetMap) continue;
        const targetShortId = targetMap.get(rel.id);
        if (!targetShortId) {
          // Orphan join row — log WARN, skip
          if (process.env.KB_SYNC_VERBOSE) console.warn(`WARN orphan related ${rel.subtype}/${rel.id} referenced by ${subtype}/${row.id}`);
          continue;
        }
        related.push({ type: rel.subtype, id: targetShortId });
      }

      // change_log — always 1 initial entry (Plan 03's idempotence layer merges with existing on update)
      const changeLog = [{
        version: '1.0.0',
        date: updatedAt.slice(0, 10),
        author: 'kb-sync-bootstrap',
        change: 'Initial population from DB via Phase 150',
      }];

      const fm = {
        id: shortIdSlug,
        type: 'resource',
        subtype,
        lang: 'es',
        title: String(row.name),
        summary,
        tags: derivedTags,
        audience,
        status,
        created_at: createdAt,
        created_by: 'kb-sync-bootstrap',
        version: '1.0.0',
        updated_at: updatedAt,
        updated_by: 'kb-sync-bootstrap',
        source_of_truth: sourceOfTruth,
        ttl: 'never',
      };
      if (related.length > 0) fm.related = related;
      if (status === 'deprecated') {
        fm.deprecated_at = updatedAt;
        fm.deprecated_by = 'kb-sync-bootstrap';
        fm.deprecated_reason = subtype === 'canvas' && row.status === 'archived'
          ? 'status=archived at first population'
          : 'is_active=0 at first population';
      }
      fm.change_log = changeLog;
      return fm;
    }
    ```

    function buildBody(subtype, row) {
      const lines = [];
      lines.push('');
      lines.push('## Descripción');
      lines.push('');
      lines.push(String(row.description || '_(sin descripción)_'));
      lines.push('');
      lines.push('## Configuración');
      lines.push('');

      if (subtype === 'catpaw') {
        lines.push(`- **Mode:** ${row.mode || 'chat'}`);
        lines.push(`- **Model:** ${row.model || 'gemini-main'}`);
        if (row.temperature != null) lines.push(`- **Temperatura:** ${row.temperature}`);
        if (row.max_tokens != null) lines.push(`- **Max tokens:** ${row.max_tokens}`);
        if (row.output_format) lines.push(`- **Output format:** ${row.output_format}`);
        if (row.tone) lines.push(`- **Tone:** ${row.tone}`);
        if (row.department_tags) lines.push(`- **Department tags:** ${row.department_tags}`);
        lines.push(`- **times_used:** ${row.times_used ?? 0}`);
        if (row.system_prompt) {
          lines.push('');
          lines.push('## System prompt');
          lines.push('');
          lines.push('```');
          lines.push(String(row.system_prompt).slice(0, 1000));
          lines.push('```');
        }
      } else if (subtype === 'connector') {
        lines.push(`- **Type:** ${row.type}`);
        lines.push(`- **test_status:** ${row.test_status || 'untested'}`);
        lines.push(`- **times_used:** ${row.times_used ?? 0}`);
        /* NEVER render row.config */
      } else if (subtype === 'skill') {
        lines.push(`- **Category:** ${row.category || '-'}`);
        lines.push(`- **Source:** ${row.source || '-'}`);
        lines.push(`- **Version:** ${row.version || '-'}`);
        lines.push(`- **Author:** ${row.author || '-'}`);
        lines.push(`- **times_used:** ${row.times_used ?? 0}`);
        if (row.instructions) {
          lines.push('');
          lines.push('## Instrucciones');
          lines.push('');
          lines.push(String(row.instructions).slice(0, 2000));
        }
      } else if (subtype === 'catbrain') {
        lines.push(`- **Purpose:** ${row.purpose || '-'}`);
        lines.push(`- **Tech stack:** ${row.tech_stack || '-'}`);
        lines.push(`- **Status:** ${row.status || '-'}`);
        lines.push(`- **RAG enabled:** ${row.rag_enabled ? 'yes' : 'no'}`);
        if (row.rag_collection) lines.push(`- **RAG collection:** ${row.rag_collection}`);
      } else if (subtype === 'email-template') {
        lines.push(`- **Category:** ${row.category || '-'}`);
        lines.push(`- **Ref code:** ${row.ref_code || '-'}`);
        lines.push(`- **times_used:** ${row.times_used ?? 0}`);
        /* NEVER render row.structure or row.html_preview */
      } else if (subtype === 'canvas') {
        lines.push(`- **Mode:** ${row.mode || '-'}`);
        lines.push(`- **Status (DB):** ${row.status || '-'}`);
        lines.push(`- **Is template:** ${row.is_template ? 'yes' : 'no'}`);
        if (row.tags) lines.push(`- **Tags (raw):** ${row.tags}`);
        /* NEVER render row.flow_data or row.thumbnail */
      }

      lines.push('');
      return lines.join('\n');
    }

    ```javascript
    function writeResourceFile(kbRoot, subtype, shortIdSlug, fm, body, dryRun, verbose) {
      const targetDir = path.join(kbRoot, SUBTYPE_SUBDIR[subtype]);
      const filePath = path.join(targetDir, `${shortIdSlug}.md`);
      const content = renderFile(fm, body);
      if (dryRun) {
        if (verbose) console.log(`DRY ${filePath}`);
        return { path: filePath, action: 'would-create' };
      }
      fs.mkdirSync(targetDir, { recursive: true });
      const action = fs.existsSync(filePath) ? 'overwrite' : 'create';  /* Plan 03 refines to 'unchanged'/'update' */
      fs.writeFileSync(filePath, content, 'utf8');
      if (verbose) console.log(`${action.toUpperCase()} ${filePath}`);
      return { path: filePath, action, subtype, id: shortIdSlug };
    }

    function populateFromDb({ kbRoot = DEFAULT_KB_ROOT, dbPath, subtypes, dryRun = false, verbose = false } = {}) {
      const effective = subtypes && subtypes.length ? subtypes : null;
      if (verbose) process.env.KB_SYNC_VERBOSE = '1';  /* toggle buildFrontmatter WARNs */
      const db = openDb(dbPath);
      const report = { created: 0, updated: 0, unchanged: 0, orphans: 0, skipped: 0, files: [] };
      try {
        const { rows, maps } = buildIdMap(db, effective);

        // Eager-load relations indexed by source row id
        const pawRels = loadCatPawRelations(db);
        const brainRels = loadCatbrainRelations(db);

        for (const sub of SUBTYPES) {
          if (effective && !effective.includes(sub)) continue;
          for (const row of rows[sub]) {
            if (!row.id || !row.name) { report.skipped++; continue; }
            let relations = [];
            if (sub === 'catpaw') relations = pawRels.get(row.id) || [];
            else if (sub === 'catbrain') relations = brainRels.get(row.id) || [];
            const shortIdSlug = maps[sub].get(row.id);
            const fm = buildFrontmatter(sub, row, kbRoot, maps, relations);
            const body = buildBody(sub, row);
            const res = writeResourceFile(kbRoot, sub, shortIdSlug, fm, body, dryRun, verbose);
            report.files.push(res);
            if (res.action === 'create' || res.action === 'would-create') report.created++;
            else if (res.action === 'overwrite') report.updated++;
          }
        }
        return report;
      } finally {
        db.close();
      }
    }
    ```
