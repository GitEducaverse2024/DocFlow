# Plan 61-01: Bundle Generator Service + Export API

## Goal
Install archiver, implement the bundle generator service that collects task resources and produces a ZIP, and create the POST export API route.

## Context
- `task_bundles` table exists (id, task_id, bundle_name, bundle_path, manifest, created_at)

- Tasks reference agents via `task_steps.agent_id` → `cat_paws` table
- Tasks reference canvases via `task_steps.canvas_id` → `canvases` table
- Tasks reference skills via `worker_skills` junction (agent_id → skill_id)

**Implementation details:**

1. **Collect task + steps**:
   ```sql
   SELECT * FROM tasks WHERE id = ?
   SELECT * FROM task_steps WHERE task_id = ? ORDER BY order_index
   ```

2. **Collect referenced agents** (cat_paws):
   - Get unique `agent_id` values from task_steps WHERE agent_id IS NOT NULL
   - For each: `SELECT * FROM cat_paws WHERE id = ?`
   - Also get their CatBrain associations: `SELECT * FROM cat_paw_catbrains WHERE paw_id = ?`
   - Also get their connector associations: `SELECT * FROM cat_paw_connectors WHERE paw_id = ?`

3. **Collect referenced canvases**:
   - Get unique `canvas_id` values from task_steps WHERE canvas_id IS NOT NULL
   - For each: `SELECT * FROM canvases WHERE id = ?`
   - Also scan canvas `flow_data` nodes for agent references (node.data.agentId) and include those agents too

4. **Collect referenced skills**:
   - For each collected agent: `SELECT s.* FROM skills s JOIN worker_skills ws ON ws.skill_id = s.id WHERE ws.agent_id = ?`

5. **Detect required Docker services**:
   ```typescript
   function detectRequiredServices(steps: StepRow[], agents: CatPaw[]): string[] {
     const services = ['docflow', 'litellm']; // always needed

     // Qdrant: any step uses RAG context
     const needsQdrant = steps.some(s =>
       s.context_mode === 'rag' || s.use_project_rag === 1
     );
     if (needsQdrant) services.push('qdrant');

     // Ollama: any agent uses local model
     const needsOllama = agents.some(a =>
       a.model?.startsWith('ollama/') || a.model?.startsWith('local/')
     );
     if (needsOllama) services.push('ollama');

     return services;
   }
   ```

6. **Detect credentials needed**:
   ```typescript
   function detectCredentials(agents: CatPaw[], services: string[]): string[] {
     const creds: Set<string> = new Set();
     creds.add('LITELLM_API_KEY'); // always

     const models = agents.map(a => a.model || '');
     if (models.some(m => m.includes('openai') || m.includes('gpt'))) creds.add('OPENAI_API_KEY');
     if (models.some(m => m.includes('anthropic') || m.includes('claude'))) creds.add('ANTHROPIC_API_KEY');
     if (models.some(m => m.includes('gemini'))) creds.add('GOOGLE_API_KEY');

     return Array.from(creds);
   }
   ```

7. **Build manifest.json** — populate BundleManifest struct

8. **Create ZIP with archiver**:
   ```typescript
   // Ensure exports dir exists
   fs.mkdirSync(EXPORTS_DIR, { recursive: true });

   const slug = task.name.toLowerCase().replace(/[^a-z0-9]+/g, '-').slice(0, 50);
   const bundleId = uuidv4();
   const filename = `${slug}-${Date.now()}.zip`;
   const bundlePath = path.join(EXPORTS_DIR, filename);

   const output = fs.createWriteStream(bundlePath);
   const archive = archiver('zip', { zlib: { level: 9 } });
   archive.pipe(output);

   const prefix = `${slug}-bundle`;

   // manifest.json
   archive.append(JSON.stringify(manifest, null, 2), { name: `${prefix}/manifest.json` });

   // config/task.json
   archive.append(JSON.stringify({ ...task, steps }, null, 2), { name: `${prefix}/config/task.json` });

   // config/canvases/*.json
   for (const canvas of canvases) {
     archive.append(JSON.stringify(canvas, null, 2), { name: `${prefix}/config/canvases/${canvas.id}.json` });
   }

   // config/agents/*.json
   for (const agent of agents) {
     archive.append(JSON.stringify(agent, null, 2), { name: `${prefix}/config/agents/${agent.id}.json` });
   }

   // config/skills/*.json
   for (const skill of skills) {
     archive.append(JSON.stringify(skill, null, 2), { name: `${prefix}/config/skills/${skill.id}.json` });
   }

   // Placeholder directories for plans 61-03 (docker/, runner/, install/)
   // These will be filled by plan 61-03 with actual templates

   await archive.finalize();
   await new Promise<void>((resolve, reject) => {
     output.on('close', resolve);
     output.on('error', reject);
   });
   ```

### Task 3: Create export API route

**File**: `app/src/app/api/tasks/[id]/export/route.ts`

```typescript
import { NextResponse } from 'next/server';
import db from '@/lib/db';
import { generateBundle } from '@/lib/services/bundle-generator';

export const dynamic = 'force-dynamic';

// POST /api/tasks/[id]/export — Generate a new export bundle
export async function POST(
  request: Request,
  { params }: { params: { id: string } }
) {
  try {
    // Verify task exists
    const task = db.prepare('SELECT id, name, status FROM tasks WHERE id = ?').get(params.id);
    if (!task) {
      return NextResponse.json({ error: 'Task not found' }, { status: 404 });
    }

    const result = await generateBundle(params.id);

    return NextResponse.json({
      id: result.bundleId,
      bundle_path: result.bundlePath,
      manifest: result.manifest,
    }, { status: 201 });
  } catch (error) {
    console.error('Export error:', error);
    return NextResponse.json(
      { error: 'Failed to generate export bundle' },
      { status: 500 }
    );
  }
}
```
