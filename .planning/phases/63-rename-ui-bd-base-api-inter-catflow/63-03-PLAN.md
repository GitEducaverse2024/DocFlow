```typescript
import { NextResponse } from 'next/server';
import db from '@/lib/db';

export const dynamic = 'force-dynamic';

export async function GET() {
  try {
    const tasks = db.prepare(
      'SELECT id, name, description, status FROM tasks WHERE listen_mode = 1'
    ).all();
    return NextResponse.json(tasks);
  } catch (error) {
    console.error('Error fetching listening catflows:', error);
    return NextResponse.json({ error: 'Internal Server Error' }, { status: 500 });
  }
}
```

This is a simple query endpoint. Returns task id, name, description, status for all tasks with listen_mode=1. Used by the MultiAgent node config panel (phase 67) to populate its CatFlow selector.

1. Create app/src/app/api/catflow-triggers/route.ts with POST handler:

Request body: `{ source_task_id, source_run_id?, source_node_id?, target_task_id, payload? }`

Steps:
  a. Validate target_task_id exists and has listen_mode=1. Return 400 if not listening.
  b. Generate trigger ID with generateId() (import from @/lib/utils).
  c. INSERT into catflow_triggers with status='pending'.
  d. UPDATE tasks SET external_input = payload WHERE id = target_task_id. The payload is the JSON string from the request body (store as-is). If payload is an object, JSON.stringify it before storing.
  e. UPDATE catflow_triggers SET status='running' WHERE id = trigger_id.
  f. Fire-and-forget: import executeTaskWithCycles from '@/lib/services/task-executor' and call it with target_task_id. Wrap in .catch() to log errors.
  g. Return 201 with { id: trigger_id, status: 'running' }.

IMPORTANT: Use generateId() from @/lib/utils, NOT crypto.randomUUID() (not available in HTTP context). Use bracket notation for any process.env access. Export `dynamic = 'force-dynamic'`.

2. Create app/src/app/api/catflow-triggers/[id]/route.ts with GET handler:

  a. Query catflow_triggers by id.
  b. If not found, return 404.
  c. Return the full trigger record as JSON.

3. Create app/src/app/api/catflow-triggers/[id]/complete/route.ts with POST handler:

Request body: `{ response? }`

  a. Query catflow_triggers by id. Return 404 if not found.
  b. If trigger.status is already 'completed' or 'failed', return 400 (already finalized).
  c. UPDATE catflow_triggers SET status='completed', response=?, completed_at=datetime('now') WHERE id=?.
  d. Return 200 with updated trigger.

Follow existing API route patterns:
- try/catch wrapping entire handler
- console.error for error logging (or logger.info/error if logger is imported)
- NextResponse.json for all responses
- Type assertions with `as` for DB query results
