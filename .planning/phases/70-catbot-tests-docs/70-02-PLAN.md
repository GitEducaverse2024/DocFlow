```typescript
import { Page, Locator } from '@playwright/test';

export class CatFlowPOM {
  readonly page: Page;
  readonly pageHeading: Locator;
  readonly newButton: Locator;
  readonly sidebarLink: Locator;
  readonly cardGrid: Locator;
  readonly emptyState: Locator;
  readonly filterButtons: Locator;

  constructor(page: Page) {
    this.page = page;
    this.pageHeading = page.getByRole('heading', { name: /catflow/i });
    this.newButton = page.getByRole('link', { name: /nuevo catflow/i });
    this.sidebarLink = page.getByRole('link', { name: /catflow/i }).first();
    this.cardGrid = page.locator('[data-testid="catflow-grid"]').or(page.locator('.grid'));
    this.emptyState = page.getByText(/no hay catflows/i).or(page.getByText(/crea tu primer/i));
    this.filterButtons = page.locator('button[data-filter]').or(page.getByRole('button', { name: /todos|activos|programados|escucha|borradores/i }));
  }

  async goto() {
    await this.page.goto('/catflow');
    await this.page.waitForLoadState('networkidle');
  }
}
```
