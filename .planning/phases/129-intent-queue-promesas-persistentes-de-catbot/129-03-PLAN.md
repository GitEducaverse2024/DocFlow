          private static readonly UNRESOLVED_INTENTS_THRESHOLD = 5;

          static async checkIntentsUnresolved(): Promise<void> {
            const row = catbotDb.prepare(
              `SELECT COUNT(*) AS cnt FROM intents WHERE status IN ('failed','abandoned')
               AND (completed_at IS NULL OR completed_at > datetime('now', '-7 days'))`
            ).get() as { cnt: number };

            if (row.cnt > this.UNRESOLVED_INTENTS_THRESHOLD) {
              this.insertAlert(
                'execution',
                'intents_unresolved',
                'Intents sin resolver acumulados',
                `Hay ${row.cnt} intents en estado failed/abandoned sin resolver (umbral: ${this.UNRESOLVED_INTENTS_THRESHOLD})`,
                'warning',
                JSON.stringify({ count: row.cnt, threshold: this.UNRESOLVED_INTENTS_THRESHOLD }),
              );
            }
          }
