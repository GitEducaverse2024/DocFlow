<!-- Available APIs -->
POST /api/tasks — creates task {name, description, expected_output} → returns task with id
POST /api/tasks/{id}/steps — creates step {type, name, agent_id, agent_name, agent_model, instructions, context_mode, context_manual, use_project_rag, skill_ids}
POST /api/tasks/{id}/steps/reorder — {step_ids: string[]}
PATCH /api/tasks/{id} — updates task {linked_projects, status, etc}
POST /api/tasks/{id}/execute — launches execution
GET /api/tasks/templates — lists templates
POST /api/tasks/from-template — {template_id, name, description, expected_output, linked_projects}
GET /api/projects?limit=100 — returns {data: Project[], pagination: {...}}
GET /api/projects/{id}/rag/info — returns {enabled, vectorCount, ...} or {enabled: false}
GET /api/agents — returns Agent[] (both openclaw and custom merged)
GET /api/skills — returns Skill[]
