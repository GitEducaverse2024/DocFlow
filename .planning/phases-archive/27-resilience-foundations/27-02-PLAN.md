async healthCheck() {

Add TTL caching to all GET handlers. The pattern is identical for every route:

```typescript
import { cacheGet, cacheSet, cacheInvalidate } from '@/lib/cache';

// At the top of GET handler:
const CACHE_KEY = 'agents';  // or 'dashboard:summary', 'health', etc.
const CACHE_TTL = 30_000;    // milliseconds

export async function GET() {
  const cached = cacheGet<ReturnType>(CACHE_KEY);
  if (cached) return NextResponse.json(cached);

  // ... existing logic ...

  // ONLY on success, before returning:
  const responseData = { ... };
  cacheSet(CACHE_KEY, responseData, CACHE_TTL);
  return NextResponse.json(responseData);
}
```
