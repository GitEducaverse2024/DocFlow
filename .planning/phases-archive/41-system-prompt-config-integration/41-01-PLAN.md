   ```typescript
   export interface CatBrainInput {
     query: string;
     context?: string;       // predecessor output or manual context
     mode?: 'rag' | 'connector' | 'both';  // defaults to 'both'
   }

   export interface CatBrainOutput {
     answer: string;
     sources?: string[];      // RAG chunk texts used
     connector_data?: { connector_name: string; success: boolean; data: unknown }[];
     catbrain_id: string;
     catbrain_name: string;
     tokens?: number;
     input_tokens?: number;
     output_tokens?: number;
     duration_ms?: number;
   }
   ```

2. Create `app/src/lib/services/execute-catbrain.ts`:
   - Import db, logger, qdrant, ollama, catbrain-connector-executor, litellm, logUsage
   - Export async function `executeCatBrain(catbrainId: string, input: CatBrainInput): Promise<CatBrainOutput>`
   - Steps:
     a. Load catbrain record: `SELECT id, name, system_prompt, default_model, rag_enabled, rag_collection FROM catbrains WHERE id = ?`
     b. If not found, throw Error('CatBrain no encontrado')
     c. Based on `input.mode` (default 'both'):
        - If mode includes 'rag' and rag_enabled: query Qdrant (reuse pattern from canvas-executor getRagContext — get embedding via ollama, search via qdrant, extract text)
        - If mode includes 'connector': call `executeCatBrainConnectors(catbrainId, input.query, mode)`
     d. Build system prompt:
        - Start with catbrain.system_prompt if set, otherwise default: `Eres el asistente experto del CatBrain "${catbrain.name}".`
        - Append RAG context section if ragContext exists: `\n\nContexto de la base de conocimiento:\n${ragContext}`
        - Append connector results if any: `\n\nDatos de conectores:\n${formatConnectorResults(results)}`
     e. Build user message: if input.context exists, prepend it to input.query with separator
     f. Call LiteLLM via fetch (same pattern as task-executor callLLM):
        - Use `catbrain.default_model || process['env']['CHAT_MODEL'] || 'gemini-main'`
        - Validate model via `litellm.resolveModel()`
        - POST to `${litellmUrl}/v1/chat/completions` with system + user messages
     g. Log usage via logUsage()
     h. Return CatBrainOutput with all fields populated
   - Use `process['env']['LITELLM_URL']` and `process['env']['LITELLM_API_KEY']` bracket notation
   - Import litellm from `@/lib/services/litellm`
